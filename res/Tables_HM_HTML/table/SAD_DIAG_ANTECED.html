TRIGGER T_EVTHM_DIAG_ANTECED_CMS
AFTER INSERT OR UPDATE OR DELETE
ON SAD_DIAG_ANTECED
FOR EACH ROW
BEGIN
	DECLARE
		n_IdEvnt				SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd				SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml					SCL_EVT_DATA.EVND_VAL%TYPE;
		n_idApli				SCL_APPLI.APLI_ID%Type;
		v_idTrameOrigine		VARCHAR2(19);
		PAT						IDE_PATIENT%ROWTYPE;
		v_evtHL7				SCL_EVT.EVNT_EVTHL7%Type;
		v_action				VARCHAR2(1);
		v_srvNip				scl_appli.APLI_SRVIPP%type;
		v_patiStatut			bas_parametre.para_valeur%Type;
		v_evtIndNoIpp			Varchar2(1);
		v_idDant				SAD_DIAG_ANTECED.DANT_ID_DANT%Type;

	BEGIN
		-- #Version# 1.17 du 01/12/2017 #Version#

		-- noPerroquet
		If Not deleting Then
			If   (:new.DANT_ID_ADIA = :old.DANT_ID_ADIA or (:new.DANT_ID_ADIA is null and :old.DANT_ID_ADIA is null)) and
				 (:new.DANT_ID_PATI = :old.DANT_ID_PATI or (:new.DANT_ID_PATI is null and :old.DANT_ID_PATI is null)) and
				 (:new.DANT_DAT_DEB = :old.DANT_DAT_DEB or (:new.DANT_DAT_DEB is null and :old.DANT_DAT_DEB is null)) and
				 (:new.DANT_DAT_FIN = :old.DANT_DAT_FIN or (:new.DANT_DAT_FIN is null and :old.DANT_DAT_FIN is null)
                 or (:new.DANT_DAT_FIN is not null and :old.DANT_DAT_FIN is null) or (:new.DANT_DAT_FIN is null and :old.DANT_DAT_FIN is not null)) and
				 (:new.DANT_IND_GEN_AUTO = :old.DANT_IND_GEN_AUTO or (:new.DANT_IND_GEN_AUTO is null and :old.DANT_IND_GEN_AUTO is null)) and
				 (:new.DANT_COMMENTAIRE = :old.DANT_COMMENTAIRE or (:new.DANT_COMMENTAIRE is null and :old.DANT_COMMENTAIRE is null)) and
				 (:new.DANT_ID_CAGE_PATHOLOGIE = :old.DANT_ID_CAGE_PATHOLOGIE or (:new.DANT_ID_CAGE_PATHOLOGIE is null and :old.DANT_ID_CAGE_PATHOLOGIE is null)) and
				 (:new.DANT_DAT_SUPPRESSION = :old.DANT_DAT_SUPPRESSION or (:new.DANT_DAT_SUPPRESSION is null and :old.DANT_DAT_SUPPRESSION is null))
			Then
				return;
			End If;
		End If;

		v_evtIndNoIpp := '-';

		if inserting then
			v_action := 'C';
		elsif updating then
			if (:new.DANT_DAT_SUPPRESSION is null) then
				-- Modif d'un antécedent non supprimé logiquement : action modification
				v_action := 'M';
			else
				if (:old.DANT_DAT_SUPPRESSION is null) then
					-- Modif d'un antécedent qui a été supprimé logiquement : action suppression
					v_action := 'S';
				else
					-- Modif d'un antécedent déjà supprimé logiquement : on sort
					return;
				end if;
			end if;
		else
			v_action := 'S';
		end if;

		If deleting Then
			-- Relecture du patient
			SELECT /* T_EVTHM_DIAG_ANTECED_CMS */ * INTO PAT FROM IDE_PATIENT
			WHERE PATI_ID_PATI = :old.DANT_ID_PATI;

			-- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
			pckTrigPfiHmM.PRO_appliExt(:old.DANT_DAT_MOD, :old.DANT_COL_COM, n_idApli, v_idTrameOrigine);

			v_idDant := :old.DANT_ID_DANT;
		Else
			-- Relecture du patient
			SELECT /* T_EVTHM_DIAG_ANTECED_CMS */ * INTO PAT FROM IDE_PATIENT
			WHERE PATI_ID_PATI = :new.DANT_ID_PATI;

			-- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
			pckTrigPfiHmM.PRO_appliExt(:new.DANT_DAT_MOD, :new.DANT_COL_COM, n_idApli, v_idTrameOrigine);

			v_idDant := :new.DANT_ID_DANT;
		End If;

		-- On ne transmet pas les évènements sur les diagnostics antécédents des patients fusionnés
		if PAT.PATI_ID_PATI_MAI Is Not Null then
			return;
		end if;

		v_patiStatut := pckPfiHmM.FUN_litValParam('NUM_PATI_STATUT', PAT.PATI_ID_ETAB, current_date);

		If v_patiStatut = 'J' Then
			v_srvNip := '0';
		Else
			Begin
				Select /* T_EVTHM_DIAG_ANTECED_CMS */ max(APLI_SRVIPP) Into v_srvNip
				From scl_appli
				Where APLI_TYPE in ('HM','OM')
				And APLI_ID_ETAB = PAT.PATI_ID_ETAB;

			Exception
				When others Then
					v_srvNip := '0';
			End;
		End If;

		-- HM est serveur d'IPP
		If v_srvNip = '1' Then
			If PAT.pati_nip is not null Then
				v_evtHl7 := 'A31';
			Else
				return;
			End If;
		-- HM n'est pas serveur d'IPP
		else
			If PAT.pati_nip is null Then
				v_evtIndNoIpp := 'D';
				v_evtHl7 := 'A31';
			else
				v_evtHl7 := 'A31';
			end if;
		end if;

		If v_evtHL7 is not null Then
			v_evtHL7 := v_evtHL7 || v_evtIndNoIpp || '--';
		End If;

		Select /* T_EVTHM_DIAG_ANTECED_CMS */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		If deleting Then
			INSERT /* T_EVTHM_DIAG_ANTECED_CMS */
			INTO SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB, EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ,
					EVNT_NUM_RSS, EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
	      			EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', :old.DANT_ID_ETAB_D, PAT.PATI_ID_PATI, PAT.PATI_NIP, NULL, NULL,
					NULL, 'DIAG_ANTECED', v_idDant, NULL, NULL, v_action, n_IdApli, v_idTrameOrigine, v_evtHl7,
	      			current_date, current_date, 0, :old.DANT_UTI_CRE, :old.DANT_UTI_MOD, :old.DANT_PRO_CRE, :old.DANT_PRO_MOD, :old.DANT_COL_COM);

		else
			INSERT /* T_EVTHM_DIAG_ANTECED_CMS */
			INTO SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB, EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ,
					EVNT_NUM_RSS, EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
	      			EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', :new.DANT_ID_ETAB_D, PAT.PATI_ID_PATI, PAT.PATI_NIP, NULL,NULL,
					NULL, 'DIAG_ANTECED', v_idDant, NULL, NULL, v_action, n_IdApli, v_idTrameOrigine, v_evtHl7,
	      			current_date, current_date, 0, :new.DANT_UTI_CRE, :new.DANT_UTI_MOD, :new.DANT_PRO_CRE, :new.DANT_PRO_MOD, :new.DANT_COL_COM);
		end if;

		-- Segment Patient
	    v_xml := null;
	    v_xml := v_xml || '<PATIENT>';
	    v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', PAT.PATI_ID_PATI);
	    v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', PAT.PATI_NOM_USU);
	    v_xml := v_xml || pckPfiHmM.val2xml('NOM_NAI', PAT.PATI_NOM_NAI);
	    v_xml := v_xml || pckPfiHmM.val2xml('PRENOM', PAT.PATI_PRENOM);
	    v_xml := v_xml || pckPfiHmM.val2xml('DAT_NAI', PAT.PATI_DAT_NAI);
	    v_xml := v_xml || pckPfiHmM.val2xml('SEXE', PAT.PATI_SEXE);
	    v_xml := v_xml || pckPfiHmM.val2xml('NIP', PAT.PATI_NIP);
	    v_xml := v_xml || '</PATIENT>';

		Select /* T_EVTHM_DIAG_ANTECED_CMS */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

	    INSERT /* T_EVTHM_DIAG_ANTECED_CMS */
	    INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
	    			EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd, n_IdEvnt, 1, 'PATIENT', 'I', v_Xml,
	      		current_date, current_date, 0, 3, 3, 2, 2, NULL);

	    v_xml := null;
	    v_xml := v_xml || '<DIAG_ANTECED>';

	    if deleting then
			v_xml := v_xml || pckPfiHmM.val2xml('ID_DANT', :old.DANT_ID_DANT);
			v_xml := v_xml || pckPfiHmM.val2xml('ID_ADIA', :old.DANT_ID_ADIA);
		    v_xml := v_xml || pckPfiHmM.val2xml('ID_CAGE_PATHOLOGIE', :old.DANT_ID_CAGE_PATHOLOGIE);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', :old.DANT_DAT_DEB);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_FIN', :old.DANT_DAT_FIN);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_SUPPRESSION', :old.DANT_DAT_SUPPRESSION);
		else
			v_xml := v_xml || pckPfiHmM.val2xml('ID_DANT', :new.DANT_ID_DANT);
			v_xml := v_xml || pckPfiHmM.val2xml('ID_ADIA', :new.DANT_ID_ADIA);
		    v_xml := v_xml || pckPfiHmM.val2xml('ID_CAGE_PATHOLOGIE', :new.DANT_ID_CAGE_PATHOLOGIE);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', :new.DANT_DAT_DEB);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_FIN', :new.DANT_DAT_FIN);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_SUPPRESSION', :new.DANT_DAT_SUPPRESSION);
		end if;

	    v_xml := v_xml || '</DIAG_ANTECED>';

	    INSERT /* T_EVTHM_DIAG_ANTECED_CMS */ INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
	    			EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
	    VALUES (n_IdEvnd+1, n_IdEvnt, 2, 'DIAG_ANTECED', v_Action, v_Xml,
	      		current_date, current_date, 0, 3, 3, 2, 2, NULL);

	EXCEPTION
		WHEN OTHERS THEN
			Begin
				pckTrigPfiHmM.PRO_traceScl(PAT.PATI_NIP, null, PAT.PATI_ID_ETAB, 'Diagnostics antécédents', 'T_EVTHM_DIAG_ANTECED_CMS', '0', SUBSTR(SQLERRM,1,2000));
			Exception
				When Others Then
					return;
			End;
	END;
END T_EVTHM_DIAG_ANTECED_CMS;
