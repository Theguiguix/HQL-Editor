TRIGGER "PCAP_XROUGE".T_EVTHM_UTIL_CM
AFTER UPDATE OR INSERT
ON BAS_UTILISATEUR
FOR EACH ROW
BEGIN
    DECLARE
        n_idApli            SCL_APPLI.APLI_ID%Type;
        v_idTrameOrigine    VARCHAR2(19);
        v_action            SCL_EVT.EVNT_ACTION%Type;
		n_IdEvnt			SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd			SCL_EVT_DATA.EVND_ID_EVND%TYPE;
        v_Xml               SCL_EVT_DATA.EVND_VAL%Type;
        v_evtHL7            VARCHAR2(10);

    BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#

		If updating Then
			If (:new.UTIL_ID_MEDE = :old.UTIL_ID_MEDE or (:new.UTIL_ID_MEDE is null and :old.UTIL_ID_MEDE is null)) and
				(:new.UTIL_NOM = :old.UTIL_NOM or (:new.UTIL_NOM is null and :old.UTIL_NOM is null)) and
				(:new.UTIL_LOGIN = :old.UTIL_LOGIN or (:new.UTIL_LOGIN is null and :old.UTIL_LOGIN is null)) and
				(:new.UTIL_PASSORD = :old.UTIL_PASSORD or (:new.UTIL_PASSORD is null and :old.UTIL_PASSORD is null)) and
				(:new.UTIL_DAT_ACT = :old.UTIL_DAT_ACT or (:new.UTIL_DAT_ACT is null and :old.UTIL_DAT_ACT is null)) and
				(:new.UTIL_DAT_FIN = :old.UTIL_DAT_FIN or (:new.UTIL_DAT_FIN is null and :old.UTIL_DAT_FIN is null)) and

				(:new.UTIL_IND_FIL_MEDICAL = :old.UTIL_IND_FIL_MEDICAL or (:new.UTIL_IND_FIL_MEDICAL is null and :old.UTIL_IND_FIL_MEDICAL is null)) and
				(:new.UTIL_NUM_BADGE = :old.UTIL_NUM_BADGE or (:new.UTIL_NUM_BADGE is null and :old.UTIL_NUM_BADGE is null)) and

				(:new.UTIL_CER_PK = :old.UTIL_CER_PK or (:new.UTIL_CER_PK is null and :old.UTIL_CER_PK is null)) and
				(:new.UTIL_NOM_MAJ = :old.UTIL_NOM_MAJ or (:new.UTIL_NOM_MAJ is null and :old.UTIL_NOM_MAJ is null)) and
				(nvl(substr(:new.UTIL_COL_COM,7,1),'0') = nvl(substr(:old.UTIL_COL_COM,7,1),'0'))
			   Then
				   return;
			End If;
		End If;


        If  :new.UTIL_ID_UTIL <= 100  then
       		return;
        End If;


        If Inserting Then
            v_Action := 'C';                              -- Action DE creation
            v_evtHL7 := 'PMU-B01';
        Else
          v_Action := 'M';                                -- Action de reactivation
          v_evtHL7 := 'PMU-B02';
           If (:new.UTIL_DAT_FIN is NOT null and :old.UTIL_DAT_FIN is null) Then
                v_Action := 'M';                          -- Action de désactivation
                v_evtHL7 := 'PMU-B05';
            else
                If (:new.UTIL_DAT_FIN is null and :old.UTIL_DAT_FIN is not null) Then
                    v_Action := 'M';                      -- Action de reactivation
                    v_evtHL7 := 'PMU-B04';
               End if ;
           End if ;
        End If;

       	pckTrigPfiHmM.PRO_appliExt(:new.UTIL_DAT_MOD, :new.UTIL_COL_COM, n_idApli, v_idTrameOrigine);

		Select /* T_EVTHM_UTIL_CM */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

                NULL, 'UTILISATEUR', :new.UTIL_ID_UTIL, null, null, v_action, n_IdApli, v_idTrameOrigine,  v_evtHL7,
                current_date, current_date, 0, :new.UTIL_UTI_CRE, :new.UTIL_UTI_MOD, :new.UTIL_PRO_CRE, :new.UTIL_PRO_MOD, :new.UTIL_COL_COM);

        -- Insert SCL_EVT_DATA pour UTILISATEUR
		v_xml := null;
        v_xml := v_xml || '<UTILISATEUR>';
        v_xml := v_xml || pckPfiHmM.val2xml('ID_UTIL', :new.UTIL_ID_UTIL);
        v_xml := v_xml || pckPfiHmM.val2xml('NOM', :new.UTIL_NOM);
		v_xml := v_xml || pckPfiHmM.val2xml('NOMMAJ', :new.UTIL_NOM_MAJ);
        v_xml := v_xml || pckPfiHmM.val2xml('LOGIN', :new.UTIL_LOGIN);
        v_xml := v_xml || pckPfiHmM.val2xml('D_ACTIV', :new.UTIL_DAT_ACT);
        v_xml := v_xml || pckPfiHmM.val2xml('D_FIN', :new.UTIL_DAT_FIN);
        v_xml := v_xml || pckPfiHmM.val2xml('ID_MEDE', :new.UTIL_ID_MEDE);
		v_xml := v_xml || pckPfiHmM.val2xml('D_DERCONNEX', :new.UTIL_DAT_DER_CONNEXION);
        v_xml := v_xml || '</UTILISATEUR>';

		Select /* T_EVTHM_UTIL_CM */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

        INSERT /* T_EVTHM_UTIL_CM */
        INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
                                  EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
        VALUES (n_IdEvnd, n_IdEvnt, 1, 'UTILISATEUR',v_action, v_Xml,
                current_date, current_date, 0, 3, 3, 2, 2, NULL);

    EXCEPTION
        WHEN OTHERS THEN
            Begin
                pckTrigPfiHmM.PRO_traceScl(NULL, NULL, NULL, 'Evt Utilisateur', 'T_EVTHM_UTIL_CM', '0', SUBSTR(SQLERRM,1,2000));
            Exception
                When Others Then
                    return;
            End;
    END;

END T_EVTHM_UTIL_CM;
