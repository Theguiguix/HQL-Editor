BEGIN
	DECLARE
		n_CodRet				NUMBER;
		n_IdEvnt				SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd				SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml					SCL_EVT_DATA.EVND_VAL%TYPE;
		RPatient				IDE_PATIENT%ROWTYPE;
		RSejour					IDE_SEJOUR%ROWTYPE;
		v_UfCode				BAS_UF.UNFO_CODE%TYPE;
		v_UfTypSej				BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		v_MedeCode				VARCHAR2(40);
		v_TypSejBase 			BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		v_TypSej				BAS_CATALOGUE_PERS.CAPE_CODE%TYPE;
		n_idApli				SCL_APPLI.APLI_ID%Type;
		v_idTrameOrigine		VARCHAR2(19);
		v_Action				VARCHAR2(1);
		v_evtHL7				SCL_EVT.EVNT_EVTHL7%Type;
		v_evtIndNoNda			Varchar2(1);
		v_evtTypMvt				Varchar2(1);
		v_evtIndRec				Varchar2(1);
		v_srvNip				scl_appli.APLI_SRVIPP%type;
		v_sejoStatut			bas_parametre.para_valeur%Type;

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#
		v_evtIndNoNda	:= '-';
		v_evtTypMvt := '-';
		v_evtIndRec := '-';

		If Updating Then
			-- noPerroquet
			If (:new.SEPP_ID_CAGE_TYPE = :old.SEPP_ID_CAGE_TYPE or (:new.SEPP_ID_CAGE_TYPE is null and :old.SEPP_ID_CAGE_TYPE is null)) and
				 (:new.SEPP_NOM = :old.SEPP_NOM or (:new.SEPP_NOM is null and :old.SEPP_NOM is null)) and
				 (:new.SEPP_PRENOM = :old.SEPP_PRENOM or (:new.SEPP_PRENOM is null and :old.SEPP_PRENOM is null)) and
				 (:new.SEPP_COMMENTAIRE = :old.SEPP_COMMENTAIRE or (:new.SEPP_COMMENTAIRE is null and :old.SEPP_COMMENTAIRE is null)) and
				 (:new.SEPP_LIEN = :old.SEPP_LIEN or (:new.SEPP_LIEN is null and :old.SEPP_LIEN is null)) and
				 (nvl(substr(:new.SEPP_COL_COM,7,1),'0') = nvl(substr(:old.SEPP_COL_COM,7,1),'0')) and
				 (:new.SEPP_ID_CAGE_CIV = :old.SEPP_ID_CAGE_CIV or (:new.SEPP_ID_CAGE_CIV is null and :old.SEPP_ID_CAGE_CIV is null)) and
				 (:new.SEPP_EMAIL = :old.SEPP_EMAIL or (:new.SEPP_EMAIL is null and :old.SEPP_EMAIL is null)) and
				 (:new.SEPP_IND_CONFIANCE = :old.SEPP_IND_CONFIANCE or (:new.SEPP_IND_CONFIANCE is null and :old.SEPP_IND_CONFIANCE is null)) and
				 (:new.SEPP_DAT_DEB_VAL = :old.SEPP_DAT_DEB_VAL or (:new.SEPP_DAT_DEB_VAL is null and :old.SEPP_DAT_DEB_VAL is null)) and
				 (:new.SEPP_DAT_FIN_VAL = :old.SEPP_DAT_FIN_VAL or (:new.SEPP_DAT_FIN_VAL is null and :old.SEPP_DAT_FIN_VAL is null)) and
				 (:new.SEPP_ID_TITR = :old.SEPP_ID_TITR or (:new.SEPP_ID_TITR is null and :old.SEPP_ID_TITR is null)) and
				 (:new.SEPP_ID_CAPE_PARENTE = :old.SEPP_ID_CAPE_PARENTE or (:new.SEPP_ID_CAPE_PARENTE is null and :old.SEPP_ID_CAPE_PARENTE is null))
			Then
				return;
			End If;
		End If;

		If Deleting Then
			n_CodRet := pckPfiHmM.FUN_LireSEJ(:old.SEPP_ID_SEJO, RSejour);
			IF SUBSTR(:old.SEPP_COL_COM, 4, 1) = '1' THEN
				If :old.SEPP_UTI_MOD = 3 Then
					pckTrigPfiHmM.PRO_appliExt(:old.SEPP_DAT_MOD, :old.SEPP_COL_COM, n_idApli, v_idTrameOrigine);
				End If;
			End If;
		Else
			n_CodRet := pckPfiHmM.FUN_LireSEJ(:new.SEPP_ID_SEJO, RSejour);
			If :new.SEPP_UTI_MOD = 3 Then
				pckTrigPfiHmM.PRO_appliExt(:new.SEPP_DAT_MOD, :new.SEPP_COL_COM, n_idApli, v_idTrameOrigine);
			End If;
		End If;

		If RSejour.sejo_dat_sup is not null Then
			return;
		End If;

		-- Séjour incomplet = séjour mal formé au sens HM = séjour inexploitable = Séjour en cours de saisie
		If (RSejour.SEJO_IND_INCOMPLET = '1') OR (RSejour.SEJO_IND_INCOMPLET IS NULL) Then
			return;
		End If;

		If RSejour.SEJO_NUM_SEJ_COMM is null Then
			v_evtIndNoNda := 'D';

			-- HM est il serveur de NDA ?
			v_sejoStatut := pckPfiHmM.FUN_litValParam('NUM_SEJO_STATUT', RSejour.SEJO_ID_ETAB_CON, current_date);

			If v_sejoStatut = 'J' Then
				v_srvNip := '0';
			Else
				Begin
					Select /* T_EVTHM_PAP_CMS */ Max(APLI_SRVIPP)
					  Into v_srvNip
					  From scl_appli
					 Where APLI_TYPE in ('HM','OM')
					   And APLI_ID_ETAB = RSejour.SEJO_ID_ETAB_CON;

				Exception
					When others Then
						v_srvNip := '0';
				End;
			End If;

			If v_srvNip = '1' Then
				return;
			End If;
		End If;

		-- Lecture type du séjour
		if RSejour.SEJO_ID_CAPE_VEN is not null then
			SELECT /* T_EVTHM_PAP_CMS */ CAGE_CODE, CAPE_CODE
				INTO v_TypSejBase, v_TypSej
				FROM BAS_CATALOGUE_SPE, BAS_CATALOGUE_GEN, BAS_CATALOGUE_PERS
			 WHERE CAPE_ID_CAPE = RSejour.SEJO_ID_CAPE_VEN
				 AND CASP_ID_CAPE = CAPE_ID_CAPE
				 AND CASP_ID_CAGE = CAGE_ID_CAGE
				 AND CAGE_ID_TYIN = 4;
		end if;

		n_CodRet := pckPfiHmM.FUN_LirePAT(RSejour.SEJO_ID_PATI, RPatient);

		v_evtHL7 := 'Z99';

		If v_TypSejBase in ('S','J','N') Then
			v_evtIndRec := 'R';
		End If;

		If v_evtHL7 is not null Then
			v_evtHL7 := v_evtHL7 || v_evtIndNoNda || v_evtTypMvt || v_evtIndRec;
		End If;

		If inserting Then
			-- Erreur volontaire (le séjour est pivot, pour l'heure)
			v_Action := 'M';
		ElsIf updating Then
			v_Action := 'M';
		else
			-- Erreur volontaire (le séjour est pivot, pour l'heure)
			v_Action := 'M';
		End If;

		-- Lecture du médecin responsable
		n_CodRet := pckPfiHmM.FUN_LireMED_VENUE(RSejour.SEJO_ID_SEJO, RSejour.SEJO_ID_ETAB_CON, RSejour.SEJO_DAT_DEB, v_MedeCode);

		Select /* T_EVTHM_PAP_CMS */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		If Deleting Then
			Insert /* T_EVTHM_PAP_CMS */
			Into SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB,
						EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ, EVNT_NUM_RSS,
						EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION,
						EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
	      				EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', RSejour.SEJO_ID_ETAB_CON,
					RSejour.SEJO_ID_PATI, RPatient.PATI_NIP, RSejour.SEJO_ID_SEJO, RSejour.SEJO_NUM_SEJ_COMM, NULL,
					'PAP', RSejour.SEJO_ID_SEJO, NULL, NULL, v_Action,
					n_IdApli,  v_idTrameOrigine, v_evtHl7,
	      			current_date, current_date, 0, :old.SEPP_UTI_CRE, :old.SEPP_UTI_MOD, :old.SEPP_PRO_CRE, :old.SEPP_PRO_MOD, :old.SEPP_COL_COM);

		else
			Insert /* T_EVTHM_PAP_CMS */
			Into SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB,
						EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ, EVNT_NUM_RSS,
						EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION,
						EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
	      				EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', RSejour.SEJO_ID_ETAB_CON,
					RSejour.SEJO_ID_PATI, RPatient.PATI_NIP, RSejour.SEJO_ID_SEJO, RSejour.SEJO_NUM_SEJ_COMM, NULL,
					'PAP', RSejour.SEJO_ID_SEJO, NULL, NULL, v_Action,
					n_IdApli,  v_idTrameOrigine, v_evtHl7,
	      			current_date, current_date, 0, :new.SEPP_UTI_CRE, :new.SEPP_UTI_MOD, :new.SEPP_PRO_CRE, :new.SEPP_PRO_MOD, :new.SEPP_COL_COM);
		end if;

		-- Segment Patient
		v_xml := null;
		v_xml := v_xml || '<PATIENT>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', RPatient.PATI_ID_PATI);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', RPatient.PATI_NOM_USU);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_NAI', RPatient.PATI_NOM_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('PRENOM', RPatient.PATI_PRENOM);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_NAI', RPatient.PATI_DAT_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('SEXE', RPatient.PATI_SEXE);
		v_xml := v_xml || pckPfiHmM.val2xml('NIP', RPatient.PATI_NIP);
		v_xml := v_xml || '</PATIENT>';

		Select /* T_EVTHM_PAP_CMS */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		INSERT /* T_EVTHM_PAP_CMS */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
						EVND_DAT_CRE,EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd, n_IdEvnt, 1, 'PATIENT', 'I', v_Xml,
				current_date, current_date, 0, 3, 3, 2, 2, NULL);

		v_xml := null;
		v_xml := v_xml || '<SEJOUR>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_SEJO', RSejour.SEJO_ID_SEJO);
		v_xml := v_xml || pckPfiHmM.val2xml('ID_UNFO', RSejour.SEJO_ID_UNFO);
		v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJBASE', v_TypSejBase);
		v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJ', v_TypSej);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_CODE', v_UfCode);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_TYPSEJ', v_UfTypSej);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', RSejour.SEJO_DAT_DEB);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_FIN', RSejour.SEJO_DAT_FIN);
		v_xml := v_xml || pckPfiHmM.val2xml('NUM_SEJ', RSejour.SEJO_NUM_SEJ_COMM);
		v_xml := v_xml || pckPfiHmM.val2xml('STATUT', RSejour.SEJO_STATUT);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_SUP', RSejour.SEJO_DAT_SUP);
		v_xml := v_xml || pckPfiHmM.val2xml('MED', v_MedeCode);
		v_xml := v_xml || '</SEJOUR>';

		INSERT /* T_EVTHM_PAP_CMS */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
						EVND_DAT_CRE,EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd+1, n_IdEvnt, 2, 'SEJOUR', v_Action, v_Xml,
				current_date, current_date, 0, 3, 3, 2, 2, NULL);


	EXCEPTION
		WHEN OTHERS THEN
			Begin
				pckTrigPfiHmM.PRO_traceScl(RPatient.PATI_NIP, RSejour.SEJO_NUM_SEJ_COMM, RSejour.SEJO_ID_ETAB_CON, 'Evt Sejour (PaP)', 'T_EVTHM_PAP_CMS', '0', SUBSTR(SQLERRM,1,2000));
			Exception
				When Others Then
					return;
			End;
	END;
END T_EVTHM_PAP_CMS;
