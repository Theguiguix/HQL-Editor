TRIGGER_NAME TRIGGER_TYPE TRIGGER_OWNER TRIGGERING_EVENT STATUS TABLE_NAME
T_EVTHM_CUEILLETTE_C AFTER EACH ROW PCAP_XROUGE UPDATE DISABLED PHA_CUEILLETTE_UF_WRK 



TRIGGER "PCAP_XROUGE".t_evthm_cueillette_c
   AFTER UPDATE
   ON pha_cueillette_uf_wrk
   FOR EACH ROW
BEGIN
   DECLARE
		n_IdEvnt					SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd					SCL_EVT_DATA.EVND_ID_EVND%TYPE;
      	v_xml              			scl_evt_data.evnd_val%TYPE;
      	n_idapli           			scl_appli.apli_id%TYPE;
      	v_idtrameorigine   			VARCHAR2 (19);
      	n_idcage_e         			bas_catalogue_gen.cage_id_cage%TYPE;
      	n_idcage_t         			bas_catalogue_gen.cage_id_cage%TYPE;
      	n_nb_envois        			NUMBER;
      	v_indCueilletteActive		bas_parametre.para_code%Type;

   BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#

		v_indCueilletteActive := pckPfiHmM.FUN_litValParam('CDC_CUEILLETTE_ANC', :new.PCUW_ID_ETAB, current_date);
		if v_indCueilletteActive is null or v_indCueilletteActive = '0' then
			Return;
		end if;

      SELECT /* t_evthm_cueillette_c */ a.cage_id_cage
        INTO n_idcage_e
        FROM bas_catalogue_gen a, bas_type_info b
       WHERE a.cage_id_tyin = b.tyin_id_tyin
         AND a.cage_code = 'DISP_ENVOYEE'
         AND b.tyin_code = 'CAT_ETAT_CUEILLETTE';

      -- On déclenche le trigger si l'état cueillette vient de passer à "envoyée" ou si col_com position 7 a changé
      If :NEW.pcuw_id_cage_eta_cueillette = :OLD.pcuw_id_cage_eta_cueillette Or
         SUBSTR(:NEW.pcuw_col_com, 7, 1) = SUBSTR(:OLD.pcuw_col_com, 7, 1)
      Then
      	Return;
      End If;

      If :NEW.pcuw_id_cage_eta_cueillette <> n_idcage_e Or :NEW.pcuw_id_cage_eta_cueillette is null Then
      	Return;
      End If;

      -- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
      pckTrigPfiHmM.pro_appliExt (:NEW.pcuw_dat_mod,:NEW.pcuw_col_com, n_idapli, v_idtrameorigine);

	  Select /* t_evthm_cueillette_c */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

      INSERT /* t_evthm_cueillette_c */
      INTO scl_evt
                  (evnt_id_evnt, evnt_date, evnt_version, evnt_id_etab,
                   evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej,
                   evnt_num_rss, evnt_code_evt, evnt_id_obj, evnt_code_pere,
                   evnt_id_pere, evnt_action, evnt_id_apli,
                   evnt_id_evt_externe
                 , EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
           VALUES (n_IdEvnt, current_date, '0', :NEW.pcuw_id_etab,
                   NULL, NULL, NULL, NULL,
                   NULL, 'STOCKMED', :NEW.pcuw_id_pcuw, NULL,
                   NULL, 'C', n_idapli,
                   v_idtrameorigine
                 , current_date, current_date, 0, :NEW.pcuw_UTI_CRE, :NEW.pcuw_UTI_MOD, :NEW.pcuw_PRO_CRE, :NEW.pcuw_PRO_MOD, :NEW.pcuw_COL_COM);

      v_xml := NULL;
      v_xml := v_xml || '<CUEILLETTE>';
      v_xml := v_xml || pckpfihmm.val2xml ('ID_UF', :NEW.pcuw_id_unfo);
      v_xml := v_xml || pckpfihmm.val2xml ('DATE_DEB', TO_CHAR (:NEW.pcuw_dat_debut, 'DD/MM/YYYYY HH24:MI:SS'));
      v_xml := v_xml || pckpfihmm.val2xml ('DATE_FIN', TO_CHAR (:NEW.pcuw_dat_fin, 'DD/MM/YYYYY HH24:MI:SS'));
      v_xml := v_xml || pckpfihmm.val2xml ('CAGE_DISP', :NEW.pcuw_id_cage_dispensation);
      v_xml := v_xml || pckpfihmm.val2xml ('CUEILLETTE', 'Ancienne');
      v_xml := v_xml || '</CUEILLETTE>';

	  Select /* t_evthm_cueillette_c */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

      INSERT /* t_evthm_cueillette_c */
      INTO scl_evt_data
                  (evnd_id_evnd,
                   evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val
                 , EVND_DAT_CRE , EVND_DAT_MOD , EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
           VALUES (n_IdEvnd,
           				 n_IdEvnt, 1, 'CUEILLETTE', 'C', v_xml
                 , current_date, current_date, 0, 3, 3, 2, 2, NULL);

	EXCEPTION
		WHEN OTHERS THEN
			Begin
		        pcktrigpfihmm.pro_traceScl (NULL, NULL, :NEW.pcuw_id_etab, 'Evt cueillette', 'T_EVTHM_CUEILLETTE_C', '0', SUBSTR (SQLERRM, 1, 2000));
			Exception
				When Others Then
					return;
			End;
   END;
END t_evthm_cueillette_c;
