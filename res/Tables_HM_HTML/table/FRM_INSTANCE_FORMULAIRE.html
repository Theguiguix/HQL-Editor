TRIGGER "PCAP_XROUGE".T_EVTHM_FRM_INSTANCE_M
AFTER INSERT OR UPDATE
ON FRM_INSTANCE_FORMULAIRE
REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
    RPatient           	IDE_PATIENT%RowType;
    RFormulaire        	FRM_FORMULAIRE%RowType;
	n_IdEvnt			SCL_EVT.EVNT_ID_EVNT%TYPE;
	n_IdEvnd			SCL_EVT_DATA.EVND_ID_EVND%TYPE;
    v_xml              	scl_evt_data.evnd_val%TYPE;
    n_idapli           	scl_appli.apli_id%TYPE;
    v_idtrameorigine   	VARCHAR2 (19);
    v_GoEvt			   	Varchar2(1);

BEGIN
	-- #Version# 1.16 du 01/03/2016 #Version#

	-- Validation médicale du formulaire ou invalidation : on poste
	if ((:old.FIFO_IND_VAL_MEDICALE is null or :old.FIFO_IND_VAL_MEDICALE = '0') And :new.FIFO_IND_VAL_MEDICALE = '1') Or
	   (:old.FIFO_DATE is null And :new.FIFO_DATE is not null)
	then
		-- Au moins un indicateur remarquable a été activé : on poste des évenements
		v_GoEvt := '1';

	else
		-- Aucun indicateur remarquable n'a bougé
		v_GoEvt := '0';
	end if;

	If (v_GoEvt = '0') Then
		return;
	End If;

	-- Gestion de l'écho
	pckTrigPfiHmM.PRO_appliExt(:new.FIFO_DAT_MOD, :new.FIFO_COL_COM, n_idApli, v_idTrameOrigine);

	-- Lecture du patient
	Select /* T_EVTHM_FRM_INSTANCE_M */ *
	Into RPatient
	From IDE_PATIENT
	Where PATI_ID_PATI = :NEW.FIFO_ID_PATI;

	-- Lecture du formulaire
	Select /* T_EVTHM_FRM_INSTANCE_M */ *
	Into RFormulaire
	From FRM_FORMULAIRE
		Where FORM_ID_FORM = :NEW.FIFO_ID_FORM;

	Select /* T_EVTHM_FRM_INSTANCE_M */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

	INSERT /* T_EVTHM_FRM_INSTANCE_M */
	INTO scl_evt
			  (evnt_id_evnt, evnt_date, evnt_version, evnt_id_etab,
			   evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej,
			   evnt_num_rss, evnt_code_evt, evnt_id_obj, evnt_code_pere,
			   evnt_id_pere, evnt_action, evnt_id_apli, evnt_id_evt_externe
			 , EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD
			 , EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM
	   )
	VALUES (n_IdEvnt, SYSDATE, '0', RFormulaire.FORM_ID_ETAB,
			   RPatient.PATI_ID_PATI, RPatient.PATI_NIP, :NEW.FIFO_ID_SEJO, NULL,
			   NULL, 'INSTANCEFORM', :NEW.fifo_id_fifo, NULL,
			   NULL, 'M', n_idapli, v_idtrameorigine
			 , SYSDATE, SYSDATE, 0, :NEW.fifo_UTI_CRE, :NEW.fifo_UTI_MOD
			 , :NEW.fifo_PRO_CRE, :NEW.fifo_PRO_MOD, :NEW.fifo_COL_COM
			 );

	v_xml := NULL;
	v_xml := v_xml || '<INSTANCEFORM>';
	v_xml := v_xml || pckpfihmm.val2xml ('NOMPAT', RPatient.pati_nom_usu);
	v_xml := v_xml || pckpfihmm.val2xml ('NOMNAI', RPatient.pati_nom_nai);
	v_xml := v_xml || pckpfihmm.val2xml ('PRENPAT',RPatient.pati_prenom);
	v_xml := v_xml || pckpfihmm.val2xml ('NIP', RPatient.pati_nip);

	v_xml := v_xml || pckpfihmm.val2xml ('IDFIFO', :NEW.fifo_id_fifo);
	v_xml := v_xml || pckpfihmm.val2xml ('IDFORM', :NEW.fifo_id_form);
	v_xml := v_xml || pckpfihmm.val2xml ('CODFORM', RFormulaire.form_code);
	v_xml := v_xml || pckpfihmm.val2xml ('LIBFORM', RFormulaire.form_libelle);
	v_xml := v_xml || pckpfihmm.val2xml ('IDTI', :NEW.fifo_id_tyin);
	v_xml := v_xml || pckpfihmm.val2xml ('IDOBJ', :NEW.fifo_id_objet);
	v_xml := v_xml || pckpfihmm.val2xml ('IDMEDEVAL', :NEW.fifo_id_mede_val_medicale);
	v_xml :=
		v_xml
	 || pckpfihmm.val2xml ('DATEVALMED',
						   TO_CHAR (:NEW.fifo_dat_val_medicale,
									'DD/MM/YYYYY HH24:MI:SS'
								   )
						  );
	v_xml :=
	v_xml
	   || pckpfihmm.val2xml ('DATEVAL',
				 TO_CHAR (:NEW.fifo_dat_valeur,
					  'DD/MM/YYYYY HH24:MI:SS'
					 )
	);

	v_xml :=
		  v_xml
			 || pckpfihmm.val2xml ('DATEVALINV',
					   TO_CHAR (:NEW.fifo_date,
							'DD/MM/YYYYY HH24:MI:SS'
						   )
	);

	v_xml := v_xml || '</INSTANCEFORM>';

	Select /* T_EVTHM_FRM_INSTANCE_M */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

	INSERT /* T_EVTHM_FRM_INSTANCE_M */
	INTO scl_evt_data
			  (evnd_id_evnd,
			   evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val
			 , EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD
			 , EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM
			 )
	VALUES (n_IdEvnd,
						n_IdEvnt, 1, 'INSTANCEFORM', 'M', v_xml
			 , SYSDATE, SYSDATE, 0, 3, 3
			 , 2, 2, NULL
			);

EXCEPTION
	WHEN OTHERS THEN
		Begin
			pckTrigPfiHmM.PRO_traceScl(RPatient.PATI_NIP, null, RPatient.PATI_ID_ETAB, 'Evt_instance_formulaire', 'T_EVTHM_FRM_INSTANCE_FORMULAIRE_M', '0', SUBSTR(SQLERRM,1,2000));
		Exception
			When Others Then
				return;
		End;
END T_EVTHM_FRM_INSTANCE_M;
