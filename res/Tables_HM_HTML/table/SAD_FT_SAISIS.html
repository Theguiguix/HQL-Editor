TRIGGER_NAME TRIGGER_TYPE TRIGGER_OWNER TRIGGERING_EVENT STATUS TABLE_NAME
T_EVT_SAD_FT_SAISIS_CMS BEFORE EACH ROW PCAP_XROUGE INSERT OR UPDATE OR DELETE DISABLED SAD_FT_SAISIS 



TRIGGER "PCAP_XROUGE".T_EVT_SAD_FT_SAISIS_CMS
Before INSERT or UPDATE or DELETE
ON SAD_FT_SAISIS
FOR EACH ROW
DECLARE
	RForfait						SAD_FT_SAISIS%RowType;
	RRecueil						SAD_RECUEIL%RowType;
	RDossier						SAD_DOSSIER_TECHNIQUE%RowType;
	REtape							SAD_ETAP_SAISIES%RowType;
	RPatient						IDE_PATIENT%RowType;
	RSejour							IDE_SEJOUR%RowType;
	RUfTec							BAS_UF%RowType;
	RAppareil						PAD_APPAREIL%RowType;
	v_xml							SCL_EVT_DATA.EVND_VAL%Type;
	n_IdEvnt						SCL_EVT.EVNT_ID_EVNT%TYPE;
	n_IdEvnd						SCL_EVT_DATA.EVND_ID_EVND%TYPE;
	v_codeAction					varchar2(1);
	n_idApli						SCL_APPLI.APLI_ID%Type;
	v_idTrameOrigine				VARCHAR2(19);

BEGIN
	-- #Version# 1.16 du 01/03/2016 #Version#

	-- Nb : on ne regarde pas l'état du Rendez vous car le Calcul des n° de FT est fait au plus tard dans la prise en charge,
	-- donc forcément Ok pour transmission
	if Deleting then
		RForfait.FTSA_ID_FTSA := :old.FTSA_ID_FTSA;
		RForfait.FTSA_ID_FOTE := :old.FTSA_ID_FOTE;
		RForfait.FTSA_ID_ETSA := :old.FTSA_ID_ETSA;
		RForfait.FTSA_ID_RECU := :old.FTSA_ID_RECU;
		RForfait.FTSA_IND_MIN_FORFAIT := :old.FTSA_IND_MIN_FORFAIT;
		RForfait.FTSA_LET_CLE := :old.FTSA_LET_CLE;
		RForfait.FTSA_NUM_FT := :old.FTSA_NUM_FT;
		RForfait.FTSA_VLR_TARIF := :old.FTSA_VLR_TARIF;
		RForfait.FTSA_UTI_CRE := :old.FTSA_UTI_CRE;
		RForfait.FTSA_UTI_MOD := :old.FTSA_UTI_MOD;
		RForfait.FTSA_PRO_CRE := :old.FTSA_PRO_CRE;
		RForfait.FTSA_PRO_MOD := :old.FTSA_PRO_MOD;
		RForfait.FTSA_COL_COM := :old.FTSA_COL_COM;

	else
		RForfait.FTSA_ID_FTSA := :new.FTSA_ID_FTSA;
		RForfait.FTSA_ID_FOTE := :new.FTSA_ID_FOTE;
		RForfait.FTSA_ID_ETSA := :new.FTSA_ID_ETSA;
		RForfait.FTSA_ID_RECU := :new.FTSA_ID_RECU;
		RForfait.FTSA_IND_MIN_FORFAIT := :new.FTSA_IND_MIN_FORFAIT;
		RForfait.FTSA_LET_CLE := :new.FTSA_LET_CLE;
		RForfait.FTSA_NUM_FT := :new.FTSA_NUM_FT;
		RForfait.FTSA_VLR_TARIF := :new.FTSA_VLR_TARIF;
		RForfait.FTSA_UTI_CRE := :new.FTSA_UTI_CRE;
		RForfait.FTSA_UTI_MOD := :new.FTSA_UTI_MOD;
		RForfait.FTSA_PRO_CRE := :new.FTSA_PRO_CRE;
		RForfait.FTSA_PRO_MOD := :new.FTSA_PRO_MOD;
		RForfait.FTSA_COL_COM := :new.FTSA_COL_COM;
	end if;

	-- Un évènement n'est posté qu'à compter du moment ou le forfait technique a été calculé par l'application : numéro > 0
	if (RForfait.FTSA_NUM_FT is null) or (RForfait.FTSA_NUM_FT <= 0) then
		return;
	end if;

	If SUBSTR(RForfait.FTSA_COL_COM, 4, 1) = '1' Then
		-- Pseudo Update de chargement de l'appli externe. Suppression à suivre. Pas d'évènement Modif à poster.
		return;
	End If;

	-- noPerroquet
	if Updating then
		If (:new.FTSA_IND_MIN_FORFAIT = :old.FTSA_IND_MIN_FORFAIT or (:new.FTSA_IND_MIN_FORFAIT is null and :old.FTSA_IND_MIN_FORFAIT is null)) and
		   (:new.FTSA_LET_CLE = :old.FTSA_LET_CLE or (:new.FTSA_LET_CLE is null and :old.FTSA_LET_CLE is null)) and
		   (:new.FTSA_ID_FOTE = :old.FTSA_ID_FOTE or (:new.FTSA_ID_FOTE is null and :old.FTSA_ID_FOTE is null)) and
		   (:new.FTSA_NUM_FT = :old.FTSA_NUM_FT or (:new.FTSA_NUM_FT is null and :old.FTSA_NUM_FT is null)) and
		   (:new.FTSA_VLR_TARIF = :old.FTSA_VLR_TARIF or (:new.FTSA_VLR_TARIF is null and :old.FTSA_VLR_TARIF is null)) and
		   (nvl(substr(:new.FTSA_COL_COM,7,1),'0') = nvl(substr(:old.FTSA_COL_COM,7,1),'0'))
		Then
			return;
		End If;
	End If;

	-- Calcul du code action
	If Deleting Then
		v_codeAction := 'S';
	Else
		If Updating then
			if (:old.FTSA_NUM_FT > 0) then
				v_codeAction := 'M';
			else
				v_codeAction := 'C';
			end if;
		else
			v_codeAction := 'C';
		end if;
	end if;

	-- Lecture des éléments parents (dossier technique, recueil, patient, séjour)
	If RForfait.FTSA_ID_RECU is not null Then
		Select /* T_EVT_SAD_FT_SAISIS_CMS */ *
		  Into RRecueil
		  From SAD_RECUEIL
		 Where RECU_ID_RECU = RForfait.FTSA_ID_RECU;
	End if;

	If RRecueil.RECU_ID_DOSS is not null Then
		Select /* T_EVT_SAD_FT_SAISIS_CMS */ *
		  Into RDossier
		  From SAD_DOSSIER_TECHNIQUE
		 Where DOSS_ID_DOSS = RRecueil.RECU_ID_DOSS;
	End if;

	If RDossier.DOSS_ID_PATI is not null Then
		Select /* T_EVT_SAD_FT_SAISIS_CMS */ *
		  Into RPatient
		  From IDE_PATIENT
		 Where PATI_ID_PATI = RDossier.DOSS_ID_PATI;
	End if;

	If RDossier.DOSS_ID_SEJO is not null Then
		Select /* T_EVT_SAD_FT_SAISIS_CMS */ *
		  Into RSejour
		  From IDE_SEJOUR
		 Where SEJO_ID_SEJO = RDossier.DOSS_ID_SEJO;
	End if;

	If RDossier.DOSS_ID_UNFO_TEC is not null Then
		Select /* T_EVT_SAD_FT_SAISIS_CMS */ *
		  Into RUfTec
		  From BAS_UF
		 Where UNFO_ID_UNFO = RDossier.DOSS_ID_UNFO_TEC;
	End if;

	If RForfait.FTSA_ID_ETSA is not null Then
		Begin
			Select /* T_EVT_SAD_FT_SAISIS_CMS */ PAD_APPAREIL.*
			  Into RAppareil
			  From SAD_ETAP_SAISIES
			  join PAD_LIT_POSTE on (LIPO_ID_LIPO = ETSA_ID_LIPO)
			  join PAD_APPAREIL on (APPA_ID_APPA = LIPO_ID_APPA)
			 Where ETSA_ID_ETSA = RForfait.FTSA_ID_ETSA;

		Exception
			When no_data_found then
				RAppareil.APPA_ID_APPA := null;
		End;
	End if;

	-- Gestion de l'écho
	pckTrigPfiHmM.PRO_appliExt(RForfait.FTSA_DAT_MOD, RForfait.FTSA_COL_COM, n_idApli, v_idTrameOrigine);

	Select /* T_EVT_SAD_FT_SAISIS_CMS */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

	Insert /* T_EVT_SAD_FT_SAISIS_CMS */
	Into scl_Evt (evnt_id_evnt, evnt_date, evnt_version, evnt_id_etab,
						 evnt_code_evt, evnt_id_obj, evnt_code_pere, evnt_id_pere,
						 evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej, evnt_num_rss,
						 evnt_action, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE,
						 EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
	Values (n_IdEvnt, current_date, '0', RSejour.SEJO_ID_ETAB_CON,
			'FORFAIT_TECH_OM', RForfait.FTSA_ID_FTSA, 'RECUEIL', RForfait.FTSA_ID_RECU,
			RPatient.PATI_ID_PATI, RPatient.PATI_NIP, RSejour.SEJO_ID_SEJO, RSejour.SEJO_NUM_SEJ_COMM, null,
			v_codeAction, n_IdApli, v_idTrameOrigine,
			current_date, current_date, 0, RForfait.FTSA_UTI_CRE, RForfait.FTSA_UTI_MOD, RForfait.FTSA_PRO_CRE, RForfait.FTSA_PRO_MOD, RForfait.FTSA_COL_COM);

	-- Segment Patient
	v_xml := null;
	v_xml := v_xml || '<PATIENT>';
	v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', RPatient.PATI_ID_PATI);
	v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', RPatient.PATI_NOM_USU);
	v_xml := v_xml || pckPfiHmM.val2xml('NOM_NAI', RPatient.PATI_NOM_NAI);
	v_xml := v_xml || pckPfiHmM.val2xml('PRENOM', RPatient.PATI_PRENOM);
	v_xml := v_xml || pckPfiHmM.val2xml('DAT_NAI', RPatient.PATI_DAT_NAI);
	v_xml := v_xml || pckPfiHmM.val2xml('SEXE', RPatient.PATI_SEXE);
	v_xml := v_xml || pckPfiHmM.val2xml('NIP', RPatient.PATI_NIP);
	v_xml := v_xml || '</PATIENT>';

	Select /* T_EVT_SAD_FT_SAISIS_CMS */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

	Insert /* T_EVT_SAD_FT_SAISIS_CMS */
	Into scl_evt_data (evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val
							, EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
	Values (n_IdEvnd, n_IdEvnt, 1, 'PATIENT', 'I', v_xml
		  , current_date, current_date, 0, 3, 3, 2, 2, NULL);

	-- Segment Sejour
	v_xml := null;
	v_xml := v_xml || '<SEJOUR>';
	v_xml := v_xml || pckPfiHmM.val2xml('ID_SEJO', RSejour.SEJO_ID_SEJO);
	v_xml := v_xml || pckPfiHmM.val2xml('ID_UNFO', RSejour.SEJO_ID_UNFO);
	v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', RSejour.SEJO_DAT_DEB);
	v_xml := v_xml || pckPfiHmM.val2xml('NUM_SEJ', RSejour.SEJO_NUM_SEJ_COMM);
	v_xml := v_xml || '</SEJOUR>';

	Insert /* T_EVT_SAD_FT_SAISIS_CMS */
	Into scl_evt_data (evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val
							, EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
	Values (n_IdEvnd+1, n_IdEvnt, 2, 'SEJOUR', 'I', v_xml
		  , current_date, current_date, 0, 3, 3, 2, 2, NULL);

	-- Segment Dossier
	v_xml := null;
	v_xml := v_xml || '<DOSSIER>';
	v_xml := v_xml || pckPfiHmM.val2xml('ID_DOSS', RDossier.DOSS_ID_DOSS);
	v_xml := v_xml || pckPfiHmM.val2xml('IND_ACTIVITE_PRIVE', RDossier.DOSS_IND_ACTIVITE_PRIVE);
	v_xml := v_xml || pckPfiHmM.val2xml('IND_GRATUIT', RDossier.DOSS_IND_GRATUIT);
	v_xml := v_xml || pckPfiHmM.val2xml('ID_RECU', RRecueil.RECU_ID_RECU);
	v_xml := v_xml || pckPfiHmM.val2xml('DATE', RRecueil.RECU_DATE);
	v_xml := v_xml || pckPfiHmM.val2xml('ID_UNFO_TEC', RUfTec.UNFO_ID_UNFO);
	v_xml := v_xml || pckPfiHmM.val2xml('CODE_UNFO_TEC', RUfTec.UNFO_CODE);
	v_xml := v_xml || pckPfiHmM.val2xml('LIBELLE_UNFO_TEC', RUfTec.UNFO_LIBELLE);
	v_xml := v_xml || pckPfiHmM.val2xml('TYP_FAC_UNFO_TEC', RUfTec.UNFO_TYP_FAC);
	v_xml := v_xml || '</DOSSIER>';

	Insert /* T_EVT_SAD_FT_SAISIS_CMS */
	Into scl_evt_data (evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val
							, EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
	Values (n_IdEvnd+2, n_IdEvnt, 3, 'DOSSIER', 'I', v_xml
		  , current_date, current_date, 0, 3, 3, 2, 2, NULL);

	-- Segment Forfait Technique
	v_xml := null;
	v_xml := v_xml || '<FORFAIT_TECHNIQUE>';
	v_xml := v_xml || pckPfiHmM.val2xml('ID_FTSA', RForfait.FTSA_ID_FTSA);
	v_xml := v_xml || pckPfiHmM.val2xml('ID_FOTE', RForfait.FTSA_ID_FOTE);
	v_xml := v_xml || pckPfiHmM.val2xml('IND_MIN_FORFAIT', RForfait.FTSA_IND_MIN_FORFAIT);
	v_xml := v_xml || pckPfiHmM.val2xml('LET_CLE', RForfait.FTSA_LET_CLE);
	v_xml := v_xml || pckPfiHmM.val2xml('NUM_FT', RForfait.FTSA_NUM_FT);
	v_xml := v_xml || pckPfiHmM.val2xml('VLR_TARIF', RForfait.FTSA_VLR_TARIF);
	v_xml := v_xml || pckPfiHmM.val2xml('APPA_NUMERO', RAppareil.APPA_NUMERO);
	v_xml := v_xml || pckPfiHmM.val2xml('APPA_DSC_PRESTATION', RAppareil.APPA_DSC_PRESTATION);
	v_xml := v_xml || pckPfiHmM.val2xml('APPA_MDE_TRAITEMENT', RAppareil.APPA_MDE_TRAITEMENT);
	v_xml := v_xml || '</FORFAIT_TECHNIQUE>';

	Insert /* T_EVT_SAD_FT_SAISIS_CMS */
	Into scl_evt_data (evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val
							, EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
	Values (n_IdEvnd+3, n_IdEvnt, 4, 'FORFAIT_TECHNIQUE', v_codeAction, v_xml
		  , current_date, current_date, 0, 3, 3, 2, 2, NULL);

EXCEPTION
	WHEN OTHERS THEN
		Begin
			pckTrigPfiHmM.PRO_traceScl(RPatient.PATI_NIP, RSejour.SEJO_NUM_SEJ_COMM, RSejour.SEJO_ID_ETAB_CON, 'Evt Forfait Tech One', 'T_EVT_SAD_FT_SAISIS_CMS', '0', SUBSTR(SQLERRM,1,2000));
		Exception
			When Others Then
				return;
		End;
END;
