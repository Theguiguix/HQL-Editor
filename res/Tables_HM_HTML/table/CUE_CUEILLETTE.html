TRIGGER_NAME TRIGGER_TYPE TRIGGER_OWNER TRIGGERING_EVENT STATUS TABLE_NAME
T_EVTHM_CUEILLETTE_NEW_M AFTER EACH ROW PCAP_XROUGE UPDATE DISABLED CUE_CUEILLETTE 



TRIGGER "PCAP_XROUGE".T_EVTHM_CUEILLETTE_NEW_M
   AFTER UPDATE
   ON cue_cueillette
   FOR EACH ROW
BEGIN
	DECLARE
		n_IdEvnt					SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd					SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_xml						scl_evt_data.evnd_val%Type;
		n_idapli					scl_appli.apli_id%Type;
		v_idtrameorigine			Varchar2(19);
		v_indEnvoiAuto				bas_parametre.para_code%Type;
		v_indStockeur				bas_parametre.para_code%Type;
		v_codeEvt					scl_evt.evnt_code_evt%Type;

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#

		v_indEnvoiAuto := pckPfiHmM.FUN_litValParam('CUE_ENVOI_DESTOCKAGE', :new.CUCU_ID_ETAB_D, current_date);
		if v_indEnvoiAuto is null then
			v_indEnvoiAuto := '0';
		end if;

		v_indStockeur := pckPfiHmM.FUN_litValParam('CUE_STOCKEUR', :new.CUCU_ID_ETAB_D, current_date);
		if v_indStockeur is null then
			v_indStockeur := '0';
		end if;

		--------------------------------------------------------------------------------------------------------------
		-- On envoie la cueillette
		--		Si Envoi auto Activé (CUE_ENVOI_DESTOCKAGE à '1')
		--				=> sur le passage vers un des états :
		--							2219='DISP_ENVOYEE' 2220='DISP_TERMINEE' 7888='DISP_PART_TERMINEE' 8071='DISP_PART_ENVOYEE'
		--				=> a chaque update sur l'état :
		--							7888='DISP_PART_TERMINEE' 8071='DISP_PART_ENVOYEE'
		--							(Une cueillette DJIN est réalisée jour par jour)
		-- 				=> si col_com position 7 a changé
		--
		--		Si Envoi auto NON Activé (CUE_ENVOI_DESTOCKAGE à '0') (l'utilisateur clique sur "Envoyer")
		--				=> sur le passage vers un des états : 2219='DISP_ENVOYEE' 8071='DISP_PART_ENVOYEE'
		--				=> a chaque update sur l'état : 8071='DISP_PART_ENVOYEE'
		--					(Une cueillette DJIN est réalisée jour par jour)
		-- 				=> si col_com position 7 a changé
		--
		--		Si Envoi vers Stockeur Activé (CUE_STOCKEUR à '1')
		--			Generation d'un EVT QTEALIVRER :
		--				=> sur le passage vers un des états : 10535:Dispensation envoyée stockeur ou 10534:Journée envoyée stockeur
		--				=> a chaque update sur l'état : 10534:Journée envoyée stockeur
		--					(Une cueillette DJIN est réalisée jour par jour)
		-- 				=> si col_com position 7 a changé
		--			Generation d'un EVT STOCKMED :
		--				=> sur le passage vers un des états : 10533:Dispensation livrée ou 10532:Journée livrée
		--				=> a chaque update sur l'état : 10532:Journée livrée
		--					(Une cueillette DJIN est réalisée jour par jour)
		--------------------------------------------------------------------------------------------------------------

		-- Pas d'envoi si l'état cueillette n'a pas varié.
		-- Sauf, eventuellement dans le cas d'une DJIN partiellement terminée (ou envoyée ou livrée), ou dans le cas d'une réémission forcée
		If (:NEW.CUCU_ID_CAGE_ETA_CUEILLETTE = :OLD.CUCU_ID_CAGE_ETA_CUEILLETTE) And
		   (:NEW.CUCU_ID_CAGE_ETA_CUEILLETTE not in(7888, 8071, 10534, 10532)) And
		   (nvl(SUBSTR(:NEW.CUCU_COL_COM, 7, 1),'X') = nvl(SUBSTR(:OLD.CUCU_COL_COM, 7, 1),'X'))
		Then
			Return;
		End If;

		-- Pas d'envoi si l'état cueillette n'est pas de type envoyéé, livrée, terminée ou partiellement terminée.
		If :NEW.CUCU_ID_CAGE_ETA_CUEILLETTE not in (2219, 2220, 7888, 8071, 10534, 10535, 10532, 10533) Or :NEW.CUCU_ID_CAGE_ETA_CUEILLETTE is null Then
			Return;
		End If;

		-- Hors Stockeur, Si envoi Manuel, Pas d'envoi si l'état cueillette n'est pas de type envoyéé, ou partiellement envoyée
		If (v_indStockeur = '0') And (v_indEnvoiAuto = '0') And :NEW.CUCU_ID_CAGE_ETA_CUEILLETTE not in(2219, 8071) Then
			Return;
		End If;

		If (v_indStockeur = '1') Then
			If :NEW.CUCU_ID_CAGE_ETA_CUEILLETTE not in(10532, 10533) Then
				v_codeEvt := 'QTEALIVRER';
			Else
				v_codeEvt := 'STOCKMED';
			End If;
		Else
			v_codeEvt := 'STOCKMED';
		End If;

		-- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
		pckTrigPfiHmM.pro_appliExt (:NEW.cucu_dat_mod, :NEW.CUCU_COL_COM, n_idapli, v_idtrameorigine);

		Select /* T_EVTHM_CUEILLETTE_NEW_M */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		INSERT /* T_EVTHM_CUEILLETTE_NEW_M */
		INTO scl_evt(evnt_id_evnt, evnt_date, evnt_version, evnt_id_etab, evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej
						  , evnt_num_rss, evnt_code_evt, evnt_id_obj, evnt_code_pere, evnt_id_pere, evnt_action, evnt_id_apli, evnt_id_evt_externe
						  , evnt_dat_cre, evnt_dat_mod, evnt_nmaj, evnt_uti_cre, evnt_uti_mod, evnt_pro_cre, evnt_pro_mod, evnt_col_com)
		VALUES (n_IdEvnt, current_date, '0', :NEW.CUCU_ID_ETAB_D, NULL, NULL, NULL, NULL
			  , NULL, v_codeEvt, :NEW.CUCU_ID_CUCU, NULL, NULL, 'C', n_idapli, v_idtrameorigine
			  , current_date, current_date, 0, :NEW.CUCU_UTI_CRE, :NEW.CUCU_UTI_MOD, :NEW.CUCU_PRO_CRE, :NEW.CUCU_PRO_MOD, :NEW.CUCU_COL_COM);

		v_xml := NULL;
		v_xml := v_xml || '<CUEILLETTE>';
		v_xml := v_xml || pckpfihmm.val2xml ('ID_CUEILLETTE', :NEW.CUCU_ID_CUCU);
		v_xml := v_xml || pckpfihmm.val2xml ('ID_UF', :NEW.CUCU_ID_UNFO);
		v_xml := v_xml || pckpfihmm.val2xml ('DATE_DEB', TO_CHAR (:NEW.CUCU_DAT_DEB, 'DD/MM/YYYY HH24:MI:SS'));
		v_xml := v_xml || pckpfihmm.val2xml ('DATE_FIN', TO_CHAR (:NEW.CUCU_DAT_FIN, 'DD/MM/YYYY HH24:MI:SS'));
		v_xml := v_xml || pckpfihmm.val2xml ('CAGE_DISP', :NEW.CUCU_ID_CAGE_DISPENSATION);
		v_xml := v_xml || pckpfihmm.val2xml ('CAGE_ETAT', :NEW.CUCU_ID_CAGE_ETA_CUEILLETTE);
		v_xml := v_xml || pckpfihmm.val2xml ('CUEILLETTE', 'Nouvelle');
		v_xml := v_xml || '</CUEILLETTE>';

		Select /* T_EVTHM_CUEILLETTE_NEW_M */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		INSERT /* T_EVTHM_CUEILLETTE_NEW_M */
		INTO scl_evt_data(evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val
							   , evnd_dat_cre, evnd_dat_mod, evnd_nmaj, evnd_uti_cre, evnd_uti_mod, evnd_pro_cre, evnd_pro_mod, evnd_col_com)
		VALUES (n_IdEvnd, n_IdEvnt, 1, 'CUEILLETTE', 'C', v_xml,
			  	current_date, current_date, 0, 3, 3, 2, 2, NULL);

	EXCEPTION
		WHEN OTHERS THEN
			Begin
				pckTrigPfiHmM.PRO_traceScl(null, null, :NEW.CUCU_ID_ETAB_D, 'Evt cueillette', 'T_EVTHM_CUEILLETTE_NEW_M', '0', SUBSTR (SQLERRM, 1, 2000));
			Exception
				When Others Then
					return;
			End;
	END;
END T_EVTHM_CUEILLETTE_NEW_M;
