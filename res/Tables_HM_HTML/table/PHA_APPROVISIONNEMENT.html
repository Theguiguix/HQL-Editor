TRIGGER "PCAP_XROUGE".T_EVTHM_APPROVISIONNEMENT_M
AFTER UPDATE
ON PHA_APPROVISIONNEMENT FOR EACH ROW
DECLARE
    n_idEvnt               SCL_EVT.EVNT_ID_EVNT%TYPE;
    n_idEvnd              SCL_EVT_DATA.EVND_ID_EVND%TYPE;
    v_xml                   SCL_EVT_DATA.EVND_VAL%TYPE;
    n_idApli                SCL_APPLI.APLI_ID%TYPE;
    n_idEtab               PRE_PRESCRIPTION_LISTE.PRME_ID_ETAB%TYPE;
    n_idPati                PRE_PRESCRIPTION_LISTE.PRME_ID_PATI%TYPE;
    n_ipp                    IDE_PATIENT.PATI_NIP%TYPE;
    n_idSejo               PRE_PRESCRIPTION_LISTE.PRME_ID_SEJO%TYPE;
    n_nda                   IDE_SEJOUR.SEJO_NUM_SEJ_COMM%TYPE;
    v_idTrameOrigine  VARCHAR2 (19);

BEGIN
    -- #Version# 1.16 du 07/06/2016 #Version#

    -- On déclenche le trigger si l'approvisionnement passe à "DISPENSE"
    IF NOT (:NEW.PAPP_STATUT = 'DISPENSE' AND :NEW.PAPP_STATUT <> :OLD.PAPP_STATUT) THEN
        RETURN;
    END IF;

    -- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
    pcktrigpfihmm.pro_appliext (:NEW.PAPP_DAT_MOD, :NEW.PAPP_COL_COM, n_idApli, v_idTrameOrigine);

    -- Lecture de l'établissement sur la prescription
    SELECT /* T_EVTHM_APPROVISIONNEMENT_M */ a.PRME_ID_ETAB, a.PRME_ID_PATI, c.PATI_NIP, a.PRME_ID_SEJO, d.SEJO_NUM_SEJ_COMM
      INTO n_idEtab, n_idPati, n_ipp, n_idSejo, n_nda
      FROM PRE_PRESCRIPTION_LISTE a, PRE_PRESCRIPTION_LIGNE b, IDE_PATIENT c, IDE_SEJOUR d
     WHERE a.PRME_ID_PRME = b.PRPL_ID_PRME
       AND b.PRPL_ID_PRPL = :NEW.PAPP_ID_PRPL
       AND a.PRME_ID_PATI = c.PATI_ID_PATI
       AND a.PRME_ID_SEJO = d.SEJO_ID_SEJO;

    Select /* T_EVTHM_APPROVISIONNEMENT_M */ SEQ_SCL_EVT.nextval Into n_idEvnt From Dual;

      INSERT /* T_EVTHM_APPROVISIONNEMENT_M */
      INTO scl_evt (evnt_id_evnt, evnt_date, evnt_version, evnt_id_etab,
            evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej,
            evnt_num_rss, evnt_code_evt, evnt_id_obj, evnt_code_pere,
            evnt_id_pere, evnt_action, evnt_id_apli, evnt_id_evt_externe,
            EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
    VALUES (n_idEvnt, current_date, '0', n_idEtab,
            n_idPati, n_ipp, n_idSejo, n_nda,
            NULL, 'APPROVISIONNEMENT', :NEW.PAPP_ID_PAPP, NULL,
            NULL, 'C', n_idApli, v_idTrameOrigine,
            current_date, current_date, 0, :NEW.PAPP_UTI_CRE, :NEW.PAPP_UTI_MOD, :NEW.PAPP_PRO_CRE, :NEW.PAPP_PRO_MOD, :NEW.PAPP_COL_COM);

    v_xml := NULL;
    v_xml := v_xml || '<APPROVISIONNEMENT>';
    v_xml := v_xml || pckpfihmm.val2xml ('ID_PRPL', :NEW.PAPP_ID_PRPL);
    v_xml := v_xml || pckpfihmm.val2xml ('ID_PHAC', :NEW.PAPP_ID_PHAC);
    v_xml := v_xml || pckpfihmm.val2xml ('ID_UTIL_DISPENSATION', :NEW.PAPP_ID_UTIL_DISPENSATION);
    v_xml := v_xml || pckpfihmm.val2xml ('ID_PLTH', :NEW.PAPP_ID_PLTH);
    v_xml := v_xml || pckpfihmm.val2xml ('ID_UNFO_HEB_COURANTE', :NEW.PAPP_ID_UNFO_HEB_COURANTE);
    v_xml := v_xml || pckpfihmm.val2xml ('STATUT', :NEW.PAPP_STATUT);
    v_xml := v_xml || pckpfihmm.val2xml ('DAT_DEB_DISTRIBUTION', :NEW.PAPP_DAT_DEB_DISTRIBUTION);
    v_xml := v_xml || pckpfihmm.val2xml ('DAT_FIN_DISTRIBUTION', :NEW.PAPP_DAT_FIN_DISTRIBUTION);
    v_xml := v_xml || pckpfihmm.val2xml ('DAT_DISPENSATION', :NEW.PAPP_DAT_DISPENSATION);
    v_xml := v_xml || pckpfihmm.val2xml ('COD_CIP', :NEW.PAPP_COD_CIP);
    v_xml := v_xml || '</APPROVISIONNEMENT>';

    Select /* T_EVTHM_APPROVISIONNEMENT_M */ SEQ_SCL_EVT_DATA.nextval Into n_idEvnd From Dual;

    INSERT /* T_EVTHM_APPROVISIONNEMENT_M */
    INTO scl_evt_data(evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val,
                    EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
    VALUES (n_idEvnd, n_idEvnt, 1, 'APPROVISIONNEMENT', 'C', v_xml,
              current_date, current_date, 0, 3, 3, 2, 2, NULL);

EXCEPTION
    WHEN OTHERS THEN
        Begin
            pckTrigPfiHmM.PRO_traceScl(null, null, n_idEtab, 'Evt_approvisionnement', 'T_EVTHM_APPROVISIONNEMENT_M', '0', SUBSTR(SQLERRM,1,2000));
        Exception
            When Others Then
                return;
        End;
END T_EVTHM_APPROVISIONNEMENT_M;
