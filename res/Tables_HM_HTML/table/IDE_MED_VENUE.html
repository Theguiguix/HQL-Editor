TRIGGER "PCAP_XROUGE".T_EVTHM_MED_CM
AFTER INSERT or UPDATE
ON IDE_MED_VENUE
FOR EACH ROW
WHEN (new.MESE_DAT_DEB Is Not Null)
BEGIN
	DECLARE
		n_CodRet				NUMBER;
		n_NoMvt					NUMBER;
		n_NoPat					NUMBER;
        v_IndConfirme     		IDE_MOUVEMENT.MOUV_IND_CONFIRMATION%TYPE ;
		v_idLipoSaisi           IDE_MOUVEMENT.MOUV_ID_LIPO_SAISI%TYPE;
		PAT						IDE_PATIENT%ROWTYPE;
		SEJ						IDE_SEJOUR%ROWTYPE;
		MVT						IDE_MOUVEMENT%ROWTYPE;
		v_MedeCode				VARCHAR2(40);
		n_IdEvnt				SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd				SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml					SCL_EVT_DATA.EVND_VAL%TYPE;
		v_TypSejBase 			BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		v_TypSej				BAS_CATALOGUE_PERS.CAPE_CODE%TYPE;
		v_UfCode				BAS_UF.UNFO_CODE%TYPE;
		v_UfTypSej				BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		v_TypMvt				BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		v_LitCode				PAD_LIT_POSTE.LIPO_CODE%TYPE;
		v_ChambreCode			PAD_CHAMBRE_SALLE.CHAS_CODE%TYPE;
		n_idApli				SCL_APPLI.APLI_ID%Type;
		v_idTrameOrigine		VARCHAR2(19);
		v_LibEvt				VARCHAR2(30);
		n_IdEvt					NUMBER;

		CURSOR C1 (DateMvt DATE, NoSej NUMBER) IS
			SELECT /* T_EVTHM_MED_CM */ MOUV_ID_MOUV, SEJO_ID_PATI, NVL(MOUV_IND_CONFIRMATION, '0'), MOUV_ID_LIPO_SAISI
			FROM IDE_MOUVEMENT, IDE_SEJOUR
			WHERE MOUV_ID_SEJO = SEJO_ID_SEJO
			AND MOUV_ID_SEJO = NoSej
			AND MOUV_DAT_MOU = DateMvt
			ORDER BY MOUV_DAT_MOU;

		v_evtHL7				SCL_EVT.EVNT_EVTHL7%Type;
		v_evtIndNoNda			Varchar2(1);
		v_evtTypMvt				Varchar2(1);
		v_evtIndRec				Varchar2(1);

		v_srvNip				scl_appli.APLI_SRVIPP%type;
		v_sejoStatut			bas_parametre.para_valeur%Type;

		v_CageCodeRoleMede		Varchar2(120);

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#
		v_evtIndNoNda	:= '-';
		v_evtTypMvt := '-';
		v_evtIndRec := '-';

		If Updating Then
			-- noPerroquet
			If (:new.MESE_ID_MEDE = :old.MESE_ID_MEDE or (:new.MESE_ID_MEDE is null and :old.MESE_ID_MEDE is null)) and
				 (:new.MESE_DAT_DEB = :old.MESE_DAT_DEB or (:new.MESE_DAT_DEB is null and :old.MESE_DAT_DEB is null)) and
				 (:new.MESE_DAT_FIN = :old.MESE_DAT_FIN or (:new.MESE_DAT_FIN is null and :old.MESE_DAT_FIN is null)) and
				 (nvl(substr(:new.MESE_COL_COM,7,1),'0') = nvl(substr(:old.MESE_COL_COM,7,1),'0')) and
				 (:new.MESE_ID_CAPE = :old.MESE_ID_CAPE or (:new.MESE_ID_CAPE is null and :old.MESE_ID_CAPE is null))
			Then
				return;
			End If;
		End If;

		/* On détermine sur quel mouvement on travaille par la date */
		OPEN C1 (:new.MESE_DAT_DEB, :new.MESE_ID_SEJO);
		FETCH C1 INTO n_NoMvt, n_NoPat, v_IndConfirme, v_idLipoSaisi;

		IF C1%ROWCOUNT = 0 THEN
			RETURN;
		END IF;

		CLOSE C1;

		/* Lecture du patient, du séjour, du mouvement et info rattachées */
		n_CodRet := pckPfiHmM.FUN_LirePAT(n_NoPat, PAT);
		n_CodRet := pckPfiHmM.FUN_LireSEJ(:new.MESE_ID_SEJO, SEJ);

		If SEJ.sejo_dat_sup is not null Then
			return;
		End If;

		If (SEJ.SEJO_IND_INCOMPLET = '1') OR (SEJ.SEJO_IND_INCOMPLET IS NULL) Then
			return;
		End If;


		v_sejoStatut := pckPfiHmM.FUN_litValParam('NUM_SEJO_STATUT', SEJ.SEJO_ID_ETAB_CON, current_date);

		If v_sejoStatut = 'J' Then
			v_srvNip := '0';
		Else
			Begin
				Select /* T_EVTHM_MED_CM */ Max(APLI_SRVIPP)
				  Into v_srvNip
				  From scl_appli
				 Where APLI_TYPE in ('HM','OM')
				   And APLI_ID_ETAB = SEJ.SEJO_ID_ETAB_CON;

			Exception
				When others Then
					v_srvNip := '0';
			End;
		End If;


		If SEJ.SEJO_NUM_SEJ_COMM is null Then
			v_evtIndNoNda := 'D';
			-- HM est il serveur d'ipp ?
			If v_srvNip = '1' Then
				return;
			End If;
		End If;

		If v_IndConfirme = '1' Then
		  IF SUBSTR(SEJ.SEJO_COL_COM,6,1) <> 'D' Then -- si pas de demande de nip faite pour le séjour
			return;
		  End If;
		End If;

		SELECT /* T_EVTHM_MED_CM */ * INTO MVT
		FROM IDE_MOUVEMENT
		WHERE MOUV_ID_MOUV = n_NoMvt;

		n_CodRet := pckPfiHmM.FUN_LireUF(MVT.MOUV_ID_UNFO, v_UfCode, v_UfTypSej);
		n_CodRet := pckPfiHmM.FUN_LireTypMvt(MVT.MOUV_ID_CAGE, v_TypMvt);
		n_codRet := pckPfiHmM.FUN_LireLIT(MVT.MOUV_ID_LIPO, v_ChambreCode, v_LitCode);

		v_evtTypMvt := v_TypMvt;

		-- Lecture type du séjour
		if SEJ.SEJO_ID_CAPE_VEN is not null then
			SELECT /* T_EVTHM_MED_CM */ CAGE_CODE, CAPE_CODE
				INTO v_TypSejBase, v_TypSej
				FROM BAS_CATALOGUE_SPE, BAS_CATALOGUE_GEN, BAS_CATALOGUE_PERS
			 WHERE CAPE_ID_CAPE = SEJ.SEJO_ID_CAPE_VEN
				 AND CASP_ID_CAPE = CAPE_ID_CAPE
				 AND CASP_ID_CAGE = CAGE_ID_CAGE
				 AND CAGE_ID_TYIN = 4;
		end if;

		-- Lecture du code médecin
		BEGIN
			SELECT /* T_EVTHM_MED_CM */ MEDE_CODE INTO v_MedeCode
			FROM IDE_MEDECIN
			WHERE MEDE_ID_MEDE = :new.MESE_ID_MEDE;
		EXCEPTION
			WHEN OTHERS THEN RETURN;
		END;

		-- WI 103044 : C'est ce trigger qui gère les évenements de changement de méd responsable (A54)
		-- Lecture du type de médecin (pour savoir s'il s'agit d'un médecin responsable)
		BEGIN
			SELECT /* T_EVTHM_MED_CM */ CAGE_COD_GES INTO v_CageCodeRoleMede
			FROM BAS_CATALOGUE_GEN
				Join BAS_CATALOGUE_SPE On CAGE_ID_CAGE = CASP_ID_CAGE
				Join BAS_CATALOGUE_PERS On CASP_ID_CAPE = CAPE_ID_CAPE
			WHERE CAPE_ID_CAPE = :new.MESE_ID_CAPE;
		EXCEPTION
			WHEN OTHERS THEN RETURN;
		END;
		If (v_CageCodeRoleMede = 'AUm' and v_evtTypMvt = 'D' and v_idLipoSaisi is null) then
			-- Médecin responsable sur mouvement de déplacement qui n'a pas de lit saisi
			If Updating Then
				v_evtHL7 := 'Z99';
			ELSE
				v_evtHL7 := 'A54';
			End If;
		ELSE
			v_evtHL7 := 'Z99';
		End If;
		-- Fin WI 103044

		/* Si le mouvement associé au médecin est le mouvement d'entrée, on générera un evt SEJOUR. Sinon Evt MOUVEMENT */
		IF MVT.MOUV_DAT_MOU = SEJ.SEJO_DAT_DEB  THEN
			v_LibEvt := 'SEJOUR';
			n_IdEvt := SEJ.SEJO_ID_SEJO;
		ELSE
			If v_IndConfirme = '1' Then
				v_LibEvt := 'MOUVEMENTP';
			else
				v_LibEvt := 'MOUVEMENT';
			END IF;
			n_IdEvt := MVT.MOUV_ID_MOUV;
		END IF;

    	If v_TypSejBase in ('S','J','N') Then
    	    v_evtIndRec := 'R';
    	End If;

		If v_evtHL7 is not null Then
			v_evtHL7 := v_evtHL7 || v_evtIndNoNda || v_evtTypMvt || v_evtIndRec;
		End If;

		If Updating Then
			-- En mise à jour, Pas d'evt sauf si changement de médecin (ou de type)
			If (:new.MESE_ID_MEDE = :old.MESE_ID_MEDE or (:new.MESE_ID_MEDE is null and :old.MESE_ID_MEDE is null)) and
				 (:new.MESE_ID_CAPE = :old.MESE_ID_CAPE or (:new.MESE_ID_CAPE is null and :old.MESE_ID_CAPE is null))
			Then
				v_evtHL7 := null;
			End If;
		End If;

		-- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
		If :new.MESE_UTI_MOD = 3 Then
			pckTrigPfiHmM.PRO_appliExt(:new.MESE_DAT_MOD, :new.MESE_COL_COM, n_idApli, v_idTrameOrigine);
		End If;

		Select /* T_EVTHM_MED_CM */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		INSERT /* T_EVTHM_MED_CM */
		INTO SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB,
					EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ,
					EVNT_NUM_RSS, EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE,
					EVNT_ID_PERE, EVNT_ACTION, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE,
					EVNT_EVTHL7, EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ,
        			EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD,
        			EVNT_COL_COM)
		VALUES (n_IdEvnt, current_date, '0', SEJ.SEJO_ID_ETAB_CON,
				PAT.PATI_ID_PATI, PAT.PATI_NIP, SEJ.SEJO_ID_SEJO, SEJ.SEJO_NUM_SEJ_COMM,
				NULL, v_LibEvt, n_IdEvt, NULL,
				NULL, 'M', n_IdApli, v_idTrameOrigine,
				v_evtHl7, current_date, current_date, 0,
        		:new.MESE_UTI_CRE, :new.MESE_UTI_MOD, :new.MESE_PRO_CRE, :new.MESE_PRO_MOD,
        		:new.MESE_COL_COM);

		v_xml := null;
		v_xml := v_xml || '<PATIENT>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', PAT.PATI_ID_PATI);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', PAT.PATI_NOM_USU);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_NAI', PAT.PATI_NOM_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('PRENOM', PAT.PATI_PRENOM);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_NAI', PAT.PATI_DAT_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('SEXE', PAT.PATI_SEXE);
		v_xml := v_xml || pckPfiHmM.val2xml('NIP', PAT.PATI_NIP);
		v_xml := v_xml || '</PATIENT>';

		Select /* T_EVTHM_MED_CM */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		INSERT /* T_EVTHM_MED_CM */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE,
						EVND_ACTION, EVND_VAL, EVND_DAT_CRE, EVND_DAT_MOD,
						EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE,
						EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd, n_IdEvnt, 1, 'PATIENT',
				'I', v_Xml, current_date, current_date,
				0, 3, 3, 2,
				2, NULL);

		IF v_LibEvt = 'SEJOUR' THEN
			v_xml := null;
			v_xml := v_xml || '<SEJOUR>';
			v_xml := v_xml || pckPfiHmM.val2xml('ID_SEJO', SEJ.SEJO_ID_SEJO);
			v_xml := v_xml || pckPfiHmM.val2xml('ID_UNFO', SEJ.SEJO_ID_UNFO);
			v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJBASE', v_TypSejBase);
			v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJ', v_TypSej);
			v_xml := v_xml || pckPfiHmM.val2xml('UNFO_CODE', v_UfCode);
			v_xml := v_xml || pckPfiHmM.val2xml('UNFO_TYPSEJ', v_UfTypSej);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', SEJ.SEJO_DAT_DEB);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_FIN', SEJ.SEJO_DAT_FIN);
			v_xml := v_xml || pckPfiHmM.val2xml('NUM_SEJ', SEJ.SEJO_NUM_SEJ_COMM);
			v_xml := v_xml || pckPfiHmM.val2xml('STATUT', SEJ.SEJO_STATUT);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_SUP', SEJ.SEJO_DAT_SUP);
			v_xml := v_xml || pckPfiHmM.val2xml('MED', to_char(:new.MESE_ID_MESE));
			v_xml := v_xml || '</SEJOUR>';

		ELSE
			v_xml := null;
			v_xml := v_xml || '<SEJOUR>';
			v_xml := v_xml || pckPfiHmM.val2xml('ID_SEJO', SEJ.SEJO_ID_SEJO);
			v_xml := v_xml || pckPfiHmM.val2xml('ID_UNFO', SEJ.SEJO_ID_UNFO);
			v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJBASE', v_TypSejBase);
			v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJ', v_TypSej);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', SEJ.SEJO_DAT_DEB);
			v_xml := v_xml || pckPfiHmM.val2xml('NUM_SEJ', SEJ.SEJO_NUM_SEJ_COMM);
			v_xml := v_xml || pckPfiHmM.val2xml('STATUT', SEJ.SEJO_STATUT);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_SUP', SEJ.SEJO_DAT_SUP);
			v_xml := v_xml || pckPfiHmM.val2xml('MED', to_char(:new.MESE_ID_MESE));
			v_xml := v_xml || '</SEJOUR>';
		END IF;

		INSERT /* T_EVTHM_MED_CM */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE,
						EVND_ACTION, EVND_VAL, EVND_DAT_CRE, EVND_DAT_MOD,
						EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE,
						EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd+1, n_IdEvnt, 2, 'SEJOUR',
				'M', v_Xml, current_date, current_date,
				0, 3, 3, 2,
				2, NULL);


		-- Insert SCL_EVT_DATA : Informations sur le mouvement
		IF v_LibEvt = 'MOUVEMENT' THEN
			v_xml := null;
			v_xml := v_xml || '<MOUVEMENT>';
			v_xml := v_xml || pckPfiHmM.val2xml('ID_MOUV', MVT.MOUV_ID_MOUV);
			v_xml := v_xml || pckPfiHmM.val2xml('UNFO_HEB_ID', MVT.MOUV_ID_MOUV);
			v_xml := v_xml || pckPfiHmM.val2xml('UNFO_HEB_CODE', v_UfCode);
			v_xml := v_xml || pckPfiHmM.val2xml('UNFO_HEB_TYPSEJ', v_UfTypSej);
			v_xml := v_xml || pckPfiHmM.val2xml('UNFO_MED_ID', MVT.MOUV_ID_UNFO_MED);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', MVT.MOUV_DAT_MOU);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_FIN', MVT.MOUV_DAT_FIN_PARTIELLE);
			v_xml := v_xml || pckPfiHmM.val2xml('TYPMVT', v_TypMvt);
			v_xml := v_xml || pckPfiHmM.val2xml('ACONFIRMER', MVT.MOUV_IND_CONFIRMATION);
			v_xml := v_xml || pckPfiHmM.val2xml('CHAMBRE', v_ChambreCode);
			v_xml := v_xml || pckPfiHmM.val2xml('LIT', v_LitCode);
			v_xml := v_xml || '</MOUVEMENT>';

			INSERT /* T_EVTHM_MED_CM */
			INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE,
							EVND_ACTION, EVND_VAL, EVND_DAT_CRE, EVND_DAT_MOD,
							EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE,
							EVND_PRO_MOD, EVND_COL_COM)
			VALUES (n_IdEvnd+2, n_IdEvnt, 3, 'MOUVEMENT',
					'M', v_Xml, current_date, current_date,
					0, 3, 3, 2,
					2, NULL);
		END IF;

	EXCEPTION
		WHEN OTHERS THEN
			Begin
				pckTrigPfiHmM.PRO_traceScl(PAT.PATI_NIP, SEJ.SEJO_NUM_SEJ_COMM, SEJ.SEJO_ID_ETAB_CON, 'Evt Medecin (Sej/Mvt)', 'T_EVTHM_MED_CM', '0', SUBSTR(SQLERRM,1,2000));
			Exception
				When Others Then
					return;
			End;
	END;
END T_EVTHM_MED_CM;
TRIGGER "PCAP_XROUGE".T_EVTHM_MED_S
AFTER DELETE
ON IDE_MED_VENUE
FOR EACH ROW
WHEN (old.MESE_DAT_DEB Is Not Null)
BEGIN
	DECLARE
		n_CodRet				NUMBER;
		n_NoMvt					NUMBER;
		n_NoPat					NUMBER;
        v_IndConfirme     		IDE_MOUVEMENT.MOUV_IND_CONFIRMATION%TYPE ;
		v_idLipoSaisi           IDE_MOUVEMENT.MOUV_ID_LIPO_SAISI%TYPE;
		PAT						IDE_PATIENT%ROWTYPE;
		SEJ						IDE_SEJOUR%ROWTYPE;
		MVT						IDE_MOUVEMENT%ROWTYPE;
		v_MedeCode				VARCHAR2(40);
		n_IdEvnt				SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd				SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml					SCL_EVT_DATA.EVND_VAL%TYPE;
		v_TypSejBase 			BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		v_TypSej				BAS_CATALOGUE_PERS.CAPE_CODE%TYPE;
		v_UfCode				BAS_UF.UNFO_CODE%TYPE;
		v_UfTypSej				BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		v_TypMvt				BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		v_LitCode				PAD_LIT_POSTE.LIPO_CODE%TYPE;
		v_ChambreCode			PAD_CHAMBRE_SALLE.CHAS_CODE%TYPE;
		n_idApli				SCL_APPLI.APLI_ID%Type;
		v_idTrameOrigine		VARCHAR2(19);
		v_LibEvt				VARCHAR2(30);
		n_IdEvt					NUMBER;

		CURSOR C1 (DateMvt DATE, NoSej NUMBER) IS
			SELECT /* T_EVTHM_MED_S */ MOUV_ID_MOUV, SEJO_ID_PATI, NVL(MOUV_IND_CONFIRMATION, '0'), MOUV_ID_LIPO_SAISI
			FROM IDE_MOUVEMENT, IDE_SEJOUR
			WHERE MOUV_ID_SEJO = SEJO_ID_SEJO
			AND MOUV_ID_SEJO = NoSej
			AND MOUV_DAT_MOU = DateMvt
			ORDER BY MOUV_DAT_MOU;

		v_evtHL7				SCL_EVT.EVNT_EVTHL7%Type;
		v_evtIndNoNda			Varchar2(1);
		v_evtTypMvt				Varchar2(1);
		v_evtIndRec				Varchar2(1);

		v_srvNip				scl_appli.APLI_SRVIPP%type;
		v_sejoStatut			bas_parametre.para_valeur%Type;

		v_CageCodeRoleMede		Varchar2(120);

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#
		v_evtIndNoNda	:= '-';
		v_evtTypMvt := '-';
		v_evtIndRec := '-';

		/* On détermine sur quel mouvement on travaille par la date */
		OPEN C1 (:old.MESE_DAT_DEB, :old.MESE_ID_SEJO);
		FETCH C1 INTO n_NoMvt, n_NoPat, v_IndConfirme, v_idLipoSaisi;

		IF C1%ROWCOUNT = 0 THEN
			RETURN;
		END IF;

		CLOSE C1;

		/* Lecture du patient, du séjour, du mouvement et info rattachées */
		n_CodRet := pckPfiHmM.FUN_LirePAT(n_NoPat, PAT);
		n_CodRet := pckPfiHmM.FUN_LireSEJ(:old.MESE_ID_SEJO, SEJ);

		If SEJ.sejo_dat_sup is not null Then
			return;
		End If;

		If (SEJ.SEJO_IND_INCOMPLET = '1') OR (SEJ.SEJO_IND_INCOMPLET IS NULL) Then
			return;
		End If;

		v_sejoStatut := pckPfiHmM.FUN_litValParam('NUM_SEJO_STATUT', SEJ.SEJO_ID_ETAB_CON, current_date);

		If v_sejoStatut = 'J' Then
			v_srvNip := '0';
		Else
			Begin
				Select /* T_EVTHM_MED_S */ Max(APLI_SRVIPP)
				  Into v_srvNip
				  From scl_appli
				 Where APLI_TYPE in ('HM','OM')
				   And APLI_ID_ETAB = SEJ.SEJO_ID_ETAB_CON;

			Exception
				When others Then
					v_srvNip := '0';
			End;
		End If;

		If SEJ.SEJO_NUM_SEJ_COMM is null Then
			v_evtIndNoNda := 'D';
			-- HM est il serveur d'ipp ?
			If v_srvNip = '1' Then
				return;
			End If;
		End If;

		If v_IndConfirme = '1' Then
		  IF SUBSTR(SEJ.SEJO_COL_COM,6,1) <> 'D' Then -- si pas de demande de nip faite pour le séjour
			return;
		  End If;
		End If;

		SELECT /* T_EVTHM_MED_S */ * INTO MVT
		FROM IDE_MOUVEMENT
		WHERE MOUV_ID_MOUV = n_NoMvt;

		n_CodRet := pckPfiHmM.FUN_LireUF(MVT.MOUV_ID_UNFO, v_UfCode, v_UfTypSej);
		n_CodRet := pckPfiHmM.FUN_LireTypMvt(MVT.MOUV_ID_CAGE, v_TypMvt);
		n_codRet := pckPfiHmM.FUN_LireLIT(MVT.MOUV_ID_LIPO, v_ChambreCode, v_LitCode);

		v_evtTypMvt := v_TypMvt;

		-- Lecture type du séjour
		if SEJ.SEJO_ID_CAPE_VEN is not null then
			SELECT /* T_EVTHM_MED_S */ CAGE_CODE, CAPE_CODE
				INTO v_TypSejBase, v_TypSej
				FROM BAS_CATALOGUE_SPE, BAS_CATALOGUE_GEN, BAS_CATALOGUE_PERS
			 WHERE CAPE_ID_CAPE = SEJ.SEJO_ID_CAPE_VEN
				 AND CASP_ID_CAPE = CAPE_ID_CAPE
				 AND CASP_ID_CAGE = CAGE_ID_CAGE
				 AND CAGE_ID_TYIN = 4;
		end if;

		-- Lecture du code médecin
		BEGIN
			SELECT /* T_EVTHM_MED_S */ MEDE_CODE INTO v_MedeCode
			FROM IDE_MEDECIN
			WHERE MEDE_ID_MEDE = :old.MESE_ID_MEDE;

		EXCEPTION
			WHEN OTHERS THEN RETURN;
		END;

		-- WI 103044 : Uniquement méd responsable
		-- Lecture du type de médecin (pour savoir s'il s'agit d'un médecin responsable)
		BEGIN
			SELECT /* T_EVTHM_MED_S */ CAGE_COD_GES INTO v_CageCodeRoleMede
			FROM BAS_CATALOGUE_GEN
				Join BAS_CATALOGUE_SPE On CAGE_ID_CAGE = CASP_ID_CAGE
				Join BAS_CATALOGUE_PERS On CASP_ID_CAPE = CAPE_ID_CAPE
			WHERE CAPE_ID_CAPE = :old.MESE_ID_CAPE;

		EXCEPTION
			WHEN OTHERS THEN RETURN;
		END;

		If (v_CageCodeRoleMede = 'AUm' and v_evtTypMvt = 'D' and v_idLipoSaisi is null) then
			-- Médecin responsable sur mouvement de déplacement qui n'a pas de lit saisi
			v_evtHL7 := 'A55';
		ELSE
			Return;
		End If;
		-- Fin WI 103044

		/* Uniquement Evt MOUVEMENT */
		IF MVT.MOUV_DAT_MOU = SEJ.SEJO_DAT_DEB  THEN
			Return;
		ELSE
			If v_IndConfirme = '1' Then
				v_LibEvt := 'MOUVEMENTP';
			else
				v_LibEvt := 'MOUVEMENT';
			END IF;
			n_IdEvt := MVT.MOUV_ID_MOUV;
		END IF;

    	If v_TypSejBase in ('S','J','N') Then
    	    v_evtIndRec := 'R';
    	End If;

		If v_evtHL7 is not null Then
			v_evtHL7 := v_evtHL7 || v_evtIndNoNda || v_evtTypMvt || v_evtIndRec;
		End If;

		-- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
		If :old.MESE_UTI_MOD = 3 Then
			pckTrigPfiHmM.PRO_appliExt(:old.MESE_DAT_MOD, :old.MESE_COL_COM, n_idApli, v_idTrameOrigine);
		End If;

		Select /* T_EVTHM_MED_S */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		INSERT /* T_EVTHM_MED_S */
		INTO SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB,
					EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ,
					EVNT_NUM_RSS, EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE,
					EVNT_ID_PERE, EVNT_ACTION, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE,
					EVNT_EVTHL7, EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ,
        			EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD,
        			EVNT_COL_COM)
		VALUES (n_IdEvnt, current_date, '0', SEJ.SEJO_ID_ETAB_CON,
				PAT.PATI_ID_PATI, PAT.PATI_NIP, SEJ.SEJO_ID_SEJO, SEJ.SEJO_NUM_SEJ_COMM,
				NULL, v_LibEvt, n_IdEvt, NULL,
				NULL, 'M', n_IdApli, v_idTrameOrigine,
				v_evtHl7, current_date, current_date, 0,
        		:old.MESE_UTI_CRE, :old.MESE_UTI_MOD, :old.MESE_PRO_CRE, :old.MESE_PRO_MOD,
        		:old.MESE_COL_COM);

		v_xml := null;
		v_xml := v_xml || '<PATIENT>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', PAT.PATI_ID_PATI);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', PAT.PATI_NOM_USU);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_NAI', PAT.PATI_NOM_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('PRENOM', PAT.PATI_PRENOM);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_NAI', PAT.PATI_DAT_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('SEXE', PAT.PATI_SEXE);
		v_xml := v_xml || pckPfiHmM.val2xml('NIP', PAT.PATI_NIP);
		v_xml := v_xml || '</PATIENT>';

		Select /* T_EVTHM_MED_S */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		INSERT /* T_EVTHM_MED_S */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE,
						EVND_ACTION, EVND_VAL, EVND_DAT_CRE, EVND_DAT_MOD,
						EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE,
						EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd, n_IdEvnt, 1, 'PATIENT',
				'I', v_Xml, current_date, current_date,
				0, 3, 3, 2,
				2, NULL);

		v_xml := null;
		v_xml := v_xml || '<SEJOUR>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_SEJO', SEJ.SEJO_ID_SEJO);
		v_xml := v_xml || pckPfiHmM.val2xml('ID_UNFO', SEJ.SEJO_ID_UNFO);
		v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJBASE', v_TypSejBase);
		v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJ', v_TypSej);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', SEJ.SEJO_DAT_DEB);
		v_xml := v_xml || pckPfiHmM.val2xml('NUM_SEJ', SEJ.SEJO_NUM_SEJ_COMM);
		v_xml := v_xml || pckPfiHmM.val2xml('STATUT', SEJ.SEJO_STATUT);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_SUP', SEJ.SEJO_DAT_SUP);
		v_xml := v_xml || pckPfiHmM.val2xml('MED', to_char(:old.MESE_ID_MESE));
		v_xml := v_xml || '</SEJOUR>';

		INSERT /* T_EVTHM_MED_S */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE,
						EVND_ACTION, EVND_VAL, EVND_DAT_CRE, EVND_DAT_MOD,
						EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE,
						EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd+1, n_IdEvnt, 2, 'SEJOUR',
				'M', v_Xml, current_date, current_date,
				0, 3, 3, 2,
				2, NULL);

		v_xml := null;
		v_xml := v_xml || '<MOUVEMENT>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_MOUV', MVT.MOUV_ID_MOUV);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_HEB_ID', MVT.MOUV_ID_MOUV);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_HEB_CODE', v_UfCode);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_HEB_TYPSEJ', v_UfTypSej);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_MED_ID', MVT.MOUV_ID_UNFO_MED);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', MVT.MOUV_DAT_MOU);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_FIN', MVT.MOUV_DAT_FIN_PARTIELLE);
		v_xml := v_xml || pckPfiHmM.val2xml('TYPMVT', v_TypMvt);
		v_xml := v_xml || pckPfiHmM.val2xml('ACONFIRMER', MVT.MOUV_IND_CONFIRMATION);
		v_xml := v_xml || pckPfiHmM.val2xml('CHAMBRE', v_ChambreCode);
		v_xml := v_xml || pckPfiHmM.val2xml('LIT', v_LitCode);
		v_xml := v_xml || '</MOUVEMENT>';

		INSERT /* T_EVTHM_MED_S */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE,
						EVND_ACTION, EVND_VAL, EVND_DAT_CRE, EVND_DAT_MOD,
						EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE,
						EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd+2, n_IdEvnt, 3, 'MOUVEMENT',
				'M', v_Xml, current_date, current_date,
				0, 3, 3, 2,
				2, NULL);

	EXCEPTION
		WHEN OTHERS THEN
			Begin
				pckTrigPfiHmM.PRO_traceScl(PAT.PATI_NIP, SEJ.SEJO_NUM_SEJ_COMM, SEJ.SEJO_ID_ETAB_CON, 'Evt Medecin (Mvt)', 'T_EVTHM_MED_S', '0', SUBSTR(SQLERRM,1,2000));
			Exception
				When Others Then
					return;
			End;
	END;
END T_EVTHM_MED_S;
