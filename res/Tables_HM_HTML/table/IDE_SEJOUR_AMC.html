TRIGGER "PCAP_XROUGE".T_EVTHM_SEJOUR_AMC_CMS
AFTER INSERT or UPDATE or DELETE
ON IDE_SEJOUR_AMC
FOR EACH ROW
BEGIN
	DECLARE
		Amc					IDE_SEJOUR_AMC%RowType;
		n_NoPat				IDE_PATIENT.PATI_ID_PATI%Type;
		n_NoSej				IDE_SEJOUR.SEJO_ID_SEJO%Type;
		Pat					IDE_PATIENT%ROWTYPE;
		Sej					IDE_SEJOUR%ROWTYPE;
		Ass					IDE_ASSURE%ROWTYPE;
		n_idApli			SCL_APPLI.APLI_ID%Type;
		v_idTrameOrigine	VARCHAR2(19);
		v_sejoStatut		bas_parametre.para_valeur%Type;
		v_srvNip			scl_appli.APLI_SRVIPP%Type;
		n_CodRet			Number;
		n_IdEvnt			SCL_EVT.EVNT_ID_EVNT%Type;
		n_IdEvnd			SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml				SCL_EVT_DATA.EVND_VAL%Type;
		v_TypSejBase 		BAS_CATALOGUE_GEN.CAGE_CODE%Type;
		v_TypSej			BAS_CATALOGUE_PERS.CAPE_CODE%Type;
		v_UfCode			BAS_UF.UNFO_CODE%Type;
		v_UfTypSej			BAS_CATALOGUE_GEN.CAGE_CODE%Type;
		v_CodGes          	BAS_CATALOGUE_GEN.CAGE_COD_GES%TYPE;
		v_evtIndNoNda		Varchar2(1);
		v_evtIndRec			Varchar2(1);
		v_evtHL7			SCL_EVT.EVNT_EVTHL7%Type;
		n_NoAssPech			IDE_PEC.PECH_ID_ASSU%Type;

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#
		v_evtIndNoNda	:= '-';
		v_evtIndRec := '-';

		If deleting Then
			Amc.seam_id_sejo := :old.seam_id_sejo;
			Amc.SEAM_UTI_CRE := :old.SEAM_UTI_CRE;
			Amc.SEAM_UTI_MOD := :old.SEAM_UTI_MOD;
			Amc.SEAM_PRO_CRE := :old.SEAM_PRO_CRE;
			Amc.SEAM_PRO_MOD := :old.SEAM_PRO_MOD;
			Amc.SEAM_DAT_MOD := :old.SEAM_DAT_MOD;
			Amc.SEAM_COL_COM := :old.SEAM_COL_COM;
		Else
			-- noPerroquet
			If (:new.SEAM_ID_ORGA_AMC = :old.SEAM_ID_ORGA_AMC or (:new.SEAM_ID_ORGA_AMC is null and :old.SEAM_ID_ORGA_AMC is null)) and
	           (:new.SEAM_TYP_PEC = :old.SEAM_TYP_PEC or (:new.SEAM_TYP_PEC is null and :old.SEAM_TYP_PEC is null)) and
	           (:new.SEAM_NUM_ADHERENT = :old.SEAM_NUM_ADHERENT or (:new.SEAM_NUM_ADHERENT is null and :old.SEAM_NUM_ADHERENT is null)) and
	           (:new.SEAM_DAT_DEB_VAL = :old.SEAM_DAT_DEB_VAL or (:new.SEAM_DAT_DEB_VAL is null and :old.SEAM_DAT_DEB_VAL is null)) and
			   (:new.SEAM_DAT_FIN_VAL = :old.SEAM_DAT_FIN_VAL or (:new.SEAM_DAT_FIN_VAL is null and :old.SEAM_DAT_FIN_VAL is null))
			Then
				return;
			End If;

			Amc.seam_id_sejo := :new.seam_id_sejo;
			Amc.SEAM_UTI_CRE := :new.SEAM_UTI_CRE;
			Amc.SEAM_UTI_MOD := :new.SEAM_UTI_MOD;
			Amc.SEAM_PRO_CRE := :new.SEAM_PRO_CRE;
			Amc.SEAM_PRO_MOD := :new.SEAM_PRO_MOD;
			Amc.SEAM_DAT_MOD := :new.SEAM_DAT_MOD;
			Amc.SEAM_COL_COM := :new.SEAM_COL_COM;
		End if;

		If (Amc.seam_id_sejo is null) Then
			return;
		End If;

		-- Lecture du séjour et du patient
		Begin
			Select /* T_EVTHM_SEJOUR_AMC_CMS */ sejo_id_sejo, sejo_id_pati
			  Into n_NoSej,n_NoPat
			  From IDE_SEJOUR
			 Where sejo_id_sejo = Amc.seam_id_sejo;

		Exception
			When Others Then
				return;
		End;

		-- Lecture du séjour et du patient
		n_CodRet := pckPfiHmM.FUN_LirePAT(n_NoPat, Pat);
		n_CodRet := pckPfiHmM.FUN_LireSEJ(n_NoSej, Sej);

		If Sej.sejo_dat_sup is not null Then
			return;
		End If;

		If (Sej.SEJO_IND_INCOMPLET = '1') OR (Sej.SEJO_IND_INCOMPLET IS NULL) Then
			return;
		End If;

		v_sejoStatut := pckPfiHmM.FUN_litValParam('NUM_SEJO_STATUT', Sej.SEJO_ID_ETAB_CON, current_date);

		If v_sejoStatut = 'J' Then
			v_srvNip := '0';
		Else
			Begin
				Select /* T_EVTHM_SEJOUR_AMC_CMS */ Max(APLI_SRVIPP)
				  Into v_srvNip
				  From scl_appli
				 Where APLI_TYPE in ('HM','OM')
				   And APLI_ID_ETAB = Sej.SEJO_ID_ETAB_CON;

			Exception
				When others Then
					v_srvNip := '0';
			End;
		End If;

		If Sej.SEJO_NUM_SEJ_COMM is null Then
			v_evtIndNoNda := 'D';
			-- HM est il serveur d'ipp ?
			If v_srvNip = '1' Then
				return;
			End If;
		End If;

		-- Chargement des propriétés des UF
		If Sej.SEJO_ID_UNFO is not null Then
			n_CodRet := pckPfiHmM.FUN_LireUF(Sej.SEJO_ID_UNFO, v_UfCode, v_TypSejBase);
			Begin
				SELECT /* T_EVTHM_SEJOUR_AMC_CMS */ CAPE_CODE, CAGE_COD_GES
					INTO v_TypSej, v_CodGes
					FROM BAS_UF, BAS_CATEGORIE_UFM, BAS_CATALOGUE_PERS, BAS_CATALOGUE_SPE, BAS_CATALOGUE_GEN
				 WHERE UNFO_ID_CUFM = CUFM_ID_CUFM
					 AND CUFM_ID_CAPE_TYP_SEJ = CAPE_ID_CAPE
					 AND CAPE_ID_CAPE = CASP_ID_CAPE
					 AND CASP_ID_CAGE = CAGE_ID_CAGE
					 AND CAGE_ID_TYIN = 4
					 AND UNFO_ID_UNFO = Sej.SEJO_ID_UNFO;

			Exception
				When others then
					v_TypSej := null;
					v_CodGes := null;
			end;
		End If;

		-- Pas le mieux qu'on ait vu : pour compatibilité avec source ScLite. on dupplique le Cage_code
		v_UfTypSej := v_TypSejBase;

		If v_CodGes = 'P' Then
			v_evtIndRec := 'R';
		End If;

		-- Recherche du numéro de l'assuré
		Begin
			Select /* T_EVTHM_SEJOUR_AMC_CMS */ pech_id_assu
			  Into n_NoAssPech
			  From IDE_PEC
			 Where pech_id_sejo = Amc.seam_id_sejo;

		Exception
			When Others Then
				return;
		End;

		-- Lecture de l'assuré
		BEGIN
			SELECT /* T_EVTHM_SEJOUR_AMC_CMS */ * INTO Ass
			  FROM IDE_ASSURE
			 WHERE ASSU_ID_ASSU = n_NoAssPech;
		EXCEPTION
			WHEN OTHERS THEN
				Ass.ASSU_ID_ASSU := NULL;
				Return;
		END;

		-- Chargement du code evt HL7
		v_evtHL7 := 'Z99' || v_evtIndNoNda || '-' || v_evtIndRec;

		-- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
		If Amc.SEAM_UTI_MOD = 3 Then
			pckTrigPfiHmM.PRO_appliExt(Amc.SEAM_DAT_MOD, Amc.SEAM_COL_COM, n_idApli, v_idTrameOrigine);
		End If;

		Select /* T_EVTHM_SEJOUR_AMC_CMS */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		INSERT /* T_EVTHM_SEJOUR_AMC_CMS */
		INTO SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB, EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ,
					EVNT_NUM_RSS, EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION,EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
					EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
		VALUES (n_IdEvnt, current_date, '0', Sej.SEJO_ID_ETAB_CON, Pat.PATI_ID_PATI, Pat.PATI_NIP, Sej.SEJO_ID_SEJO, Sej.SEJO_NUM_SEJ_COMM,
				NULL, 'COUVSOC', Sej.SEJO_ID_SEJO, NULL, NULL, 'M', n_IdApli, v_idTrameOrigine, v_evtHl7,
				current_date, current_date, 0, Amc.SEAM_UTI_CRE, Amc.SEAM_UTI_MOD, Amc.SEAM_PRO_CRE, Amc.SEAM_PRO_MOD, Amc.SEAM_COL_COM);

		v_xml := null;
		v_xml := v_xml || '<PATIENT>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', Pat.PATI_ID_PATI);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', Pat.PATI_NOM_USU);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_NAI', Pat.PATI_NOM_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('PRENOM', Pat.PATI_PRENOM);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_NAI', Pat.PATI_DAT_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('SEXE', Pat.PATI_SEXE);
		v_xml := v_xml || pckPfiHmM.val2xml('NIP', Pat.PATI_NIP);
		v_xml := v_xml || '</PATIENT>';

		Select /* T_EVTHM_SEJOUR_AMC_CMS */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		INSERT /* T_EVTHM_SEJOUR_AMC_CMS */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
								  EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd, n_IdEvnt, 1, 'PATIENT', 'I', v_Xml,
				current_date, current_date, 0, 3, 3, 2, 2, NULL);

		v_xml := null;
		v_xml := v_xml || '<SEJOUR>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_SEJO', Sej.SEJO_ID_SEJO);
		v_xml := v_xml || pckPfiHmM.val2xml('ID_UNFO', Sej.SEJO_ID_UNFO);
		v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJBASE', v_TypSejBase);
		v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJ', v_TypSej);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_CODE', v_UfCode);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_TYPSEJ', v_UfTypSej);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', Sej.SEJO_DAT_DEB);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_FIN', Sej.SEJO_DAT_FIN);
		v_xml := v_xml || pckPfiHmM.val2xml('NUM_SEJ', Sej.SEJO_NUM_SEJ_COMM);
		v_xml := v_xml || pckPfiHmM.val2xml('STATUT', Sej.SEJO_STATUT);

				current_date, current_date, 0, 3, 3, 2, 2, NULL);

		v_xml := null;
		v_xml := v_xml || '<ASSURE>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_ASSU', Ass.ASSU_ID_ASSU);

		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,

		WHEN OTHERS THEN
