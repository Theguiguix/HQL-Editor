TRIGGER "PCAP_XROUGE".T_EVT_BAS_DOCUMENT_M
AFTER UPDATE
ON BAS_DOCUMENT
FOR EACH ROW
BEGIN
	-- #Version# 1.16 du 30/03/2016 #Version#
	If :new.DOCU_TYP_MIME = 'application/pdf' And (:old.DOCU_TYP_MIME != 'application/pdf' Or :old.DOCU_TYP_MIME is null) Then
		UPDATE /* T_EVT_BAS_DOCUMENT_M */ SCL_EVT_ATTENTE
			 SET EVNA_READY_TO_GO = '1'
		 WHERE EVNA_ID_EVNT IN (
								SELECT EVNT_ID_EVNT
									FROM SCL_EVT
								 WHERE EVNT_ID_OBJ = :new.DOCU_ID_DOCU
								   AND EVNT_CODE_EVT = 'EVENEMENTDOCUMENTOM'
							   )
			 AND EVNA_READY_TO_GO = '0'
			 AND EVNA_CAUSE = 'NO_DOC_PDF';
	End If;

EXCEPTION
	WHEN OTHERS THEN
		Begin
			pckTrigPfiHmM.PRO_traceScl(null, null, null, 'Evt Bas Document', 'T_EVT_BAS_DOCUMENT_M', '0', SUBSTR(SQLERRM,1,2000));
		Exception
			When Others Then
				return;
		End;
END;
