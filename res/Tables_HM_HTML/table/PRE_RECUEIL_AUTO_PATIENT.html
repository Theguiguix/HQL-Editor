Statistiques de colonne


TRIGGER_NAME TRIGGER_TYPE TRIGGER_OWNER TRIGGERING_EVENT STATUS TABLE_NAME
T_EVTHM_AUTONOMIE_PATIENT_CMS AFTER EACH ROW PCAP_XROUGE INSERT OR UPDATE OR DELETE DISABLED PRE_RECUEIL_AUTO_PATIENT 



TRIGGER "PCAP_XROUGE".T_EVTHM_AUTONOMIE_PATIENT_CMS
AFTER INSERT OR UPDATE OR DELETE
ON PRE_RECUEIL_AUTO_PATIENT
FOR EACH ROW
BEGIN
    DECLARE
        PAT                            	IDE_PATIENT%ROWTYPE;
        v_patiStatut                	BAS_PARAMETRE.PARA_VALEUR%Type;
        v_srvNip                    	SCL_APPLI.APLI_SRVIPP%type;
        n_idApli                    	SCL_APPLI.APLI_ID%Type;
        v_action                    	SCL_EVT.EVNT_ACTION%Type;
		n_IdEvnt						SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd						SCL_EVT_DATA.EVND_ID_EVND%TYPE;
        v_codeEvt						SCL_EVT.EVNT_CODE_EVT%TYPE;
        v_evtHL7                   	 	SCL_EVT.EVNT_EVTHL7%Type;
        v_Xml                        	SCL_EVT_DATA.EVND_VAL%TYPE;
        v_idPati						PRE_RECUEIL_AUTO_PATIENT.PRAP_ID_PATI%TYPE;
		v_datMod						PRE_RECUEIL_AUTO_PATIENT.PRAP_DAT_MOD%TYPE;
		v_colCom						PRE_RECUEIL_AUTO_PATIENT.PRAP_COL_COM%TYPE;
    	v_idPrap						PRE_RECUEIL_AUTO_PATIENT.PRAP_ID_PRAP%TYPE;
    	v_utiCre						PRE_RECUEIL_AUTO_PATIENT.PRAP_UTI_CRE%TYPE;
    	v_utiMod						PRE_RECUEIL_AUTO_PATIENT.PRAP_UTI_MOD%TYPE;
    	v_proCre						PRE_RECUEIL_AUTO_PATIENT.PRAP_PRO_CRE%TYPE;
    	v_proMod						PRE_RECUEIL_AUTO_PATIENT.PRAP_PRO_MOD%TYPE;
    	v_idEtab						PRE_RECUEIL_AUTO_PATIENT.PRAP_ID_ETAB%TYPE;
		v_idTrameOrigine            	VARCHAR2(19);

    BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#

      	-- noPerroquet
        If Updating Then
            If (:new.PRAP_ID_CAGE_NIVEAU = :old.PRAP_ID_CAGE_NIVEAU or (:new.PRAP_ID_CAGE_NIVEAU is null and :old.PRAP_ID_CAGE_NIVEAU is null)) And
               (:new.PRAP_DAT_DEB_VAL = :old.PRAP_DAT_DEB_VAL or (:new.PRAP_DAT_DEB_VAL is null and :old.PRAP_DAT_DEB_VAL is null)) And
               (:new.PRAP_DAT_FIN_VAL = :old.PRAP_DAT_FIN_VAL or (:new.PRAP_DAT_FIN_VAL is null and :old.PRAP_DAT_FIN_VAL is null)) And
               (:new.PRAP_OBSERVATIONS = :old.PRAP_OBSERVATIONS or (:new.PRAP_OBSERVATIONS is null and :old.PRAP_OBSERVATIONS is null))
            Then
            	return;
            End If;
        End If;

        -- Chargement des variables
        If Deleting Then
        	v_idPati := :old.PRAP_ID_PATI;						-- Identifiant du patient
        	v_Action := 'S';									-- Action
        	v_datMod := :old.PRAP_DAT_MOD;						-- Date modification
        	v_colCom := :old.PRAP_COL_COM;						-- Colonne communication
        	v_idPrap := :old.PRAP_ID_PRAP;						-- Pk de la table PRE_RECUEIL_AUTO_PATIENT
        	v_utiCre := :old.PRAP_UTI_CRE;						-- Utilisateur créateur
			v_utiMod := :old.PRAP_UTI_MOD;						-- Utilisateur modificateur
			v_proCre := :old.PRAP_PRO_CRE;						-- Profil créateur
			v_proMod := :old.PRAP_PRO_MOD;						-- Profil modificateur
			v_idEtab := :old.PRAP_ID_ETAB;						-- Identifiant établissement
        Else
        	v_idPati := :new.PRAP_ID_PATI;						-- Identifiant du patient
        	If Inserting Then
				v_Action := 'C';								-- Action
			Else
            	v_Action := 'M';								-- Action
            End If;
            v_datMod := :new.PRAP_DAT_MOD;						-- Date modification
            v_colCom := :new.PRAP_COL_COM;						-- Colonne communication
            v_idPrap := :new.PRAP_ID_PRAP;						-- Pk de la table PRE_RECUEIL_AUTO_PATIENT
            v_utiCre := :new.PRAP_UTI_CRE;						-- Utilisateur créateur
            v_utiMod := :new.PRAP_UTI_MOD;						-- Utilisateur modificateur
            v_proCre := :new.PRAP_PRO_CRE;						-- Profil créateur
            v_proMod := :new.PRAP_PRO_MOD;						-- Profil modificateur
            v_idEtab := :new.PRAP_ID_ETAB;						-- Identifiant établissement
        End If;

        v_evtHL7 := 'A31' || '---';							 	-- Evénement HL7
		v_codeEvt := 'AUTONOMIE_PATIENT';						-- Code événement

        -- Lecture du patient
		Select /* T_EVTHM_AUTONOMIE_PATIENT_CMS */ * Into PAT
		  From IDE_PATIENT
		 Where PATI_ID_PATI = v_idPati;

        -- Vérifie que le nip est renseigné quand HM est serveur d'IPP. Si ce n'est pas le cas on s'arrête là.
        v_patiStatut := pckPfiHmM.FUN_litValParam('NUM_PATI_STATUT', PAT.PATI_ID_ETAB, current_date);
        If v_patiStatut = 'J' Then
            v_srvNip := '0';
        Else
            Begin
                Select /* T_EVTHM_AUTONOMIE_PATIENT_CMS */ max(APLI_SRVIPP)
                  Into v_srvNip
                  From scl_appli
                 Where APLI_TYPE in ('HM','OM')
                   And APLI_ID_ETAB = PAT.PATI_ID_ETAB;

				Exception
					When others Then
						v_srvNip := '0';
            End;
        End If;
        -- Teste si HM est serveur d'IPP
        If v_srvNip = '1' Then
            -- Teste si le NIP est renseigné
            If PAT.PATI_NIP is null Then
                -- HM est serveur d'ipp et le NIP n'est pas renseigné. On s'arrête là.
                return;
            End If;
        End If;

        -- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
        pckTrigPfiHmM.PRO_appliExt(v_datMod, v_colCom, n_idApli, v_idTrameOrigine);

		Select /* T_EVTHM_AUTONOMIE_PATIENT_CMS */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		Insert /* T_EVTHM_AUTONOMIE_PATIENT_CMS */
		Into SCL_EVT (EVNT_ID_EVNT,EVNT_DATE,EVNT_VERSION,EVNT_ID_ETAB,EVNT_ID_PATIENT,
							 EVNT_NIP,EVNT_ID_SEJOUR,EVNT_NUM_SEJ,EVNT_NUM_RSS,EVNT_CODE_EVT,
							 EVNT_ID_OBJ,EVNT_CODE_PERE,EVNT_ID_PERE,EVNT_ACTION,EVNT_ID_APLI,
							 EVNT_ID_EVT_EXTERNE,EVNT_EVTHL7,EVNT_DAT_CRE,EVNT_DAT_MOD,EVNT_NMAJ,
							 EVNT_UTI_CRE,EVNT_UTI_MOD,EVNT_PRO_CRE,EVNT_PRO_MOD,EVNT_COL_COM
							)
		Values (n_IdEvnt,current_date,'0',v_idEtab,PAT.PATI_ID_PATI,
				PAT.PATI_NIP,NULL,NULL,NULL, v_codeEvt,
				v_idPrap,NULL,NULL,v_action,n_IdApli,
				v_idTrameOrigine,v_evtHl7,current_date,current_date,0,
				v_utiCre,v_utiMod,v_proCre,v_proMod,v_colCom
			   );

        v_xml := null;
        v_xml := v_xml || '<PATIENT>';
        v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', PAT.PATI_ID_PATI);
        v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', PAT.PATI_NOM_USU);
        v_xml := v_xml || pckPfiHmM.val2xml('NOM_NAI', PAT.PATI_NOM_NAI);
        v_xml := v_xml || pckPfiHmM.val2xml('PRENOM', PAT.PATI_PRENOM);
        v_xml := v_xml || pckPfiHmM.val2xml('DAT_NAI', PAT.PATI_DAT_NAI);
        v_xml := v_xml || pckPfiHmM.val2xml('SEXE', PAT.PATI_SEXE);
        v_xml := v_xml || pckPfiHmM.val2xml('NIP', PAT.PATI_NIP);
        v_xml := v_xml || '</PATIENT>';

		Select /* T_EVTHM_AUTONOMIE_PATIENT_CMS */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

        INSERT /* T_EVTHM_AUTONOMIE_PATIENT_CMS */
        INTO SCL_EVT_DATA (EVND_ID_EVND,EVND_ID_EVNT,EVND_INDEX,EVND_TYPE,EVND_ACTION,
                                  EVND_VAL,EVND_DAT_CRE,EVND_DAT_MOD,EVND_NMAJ,EVND_UTI_CRE,
                                  EVND_UTI_MOD,EVND_PRO_CRE,EVND_PRO_MOD,EVND_COL_COM
                                )
        VALUES (n_IdEvnd, n_IdEvnt, 1, 'PATIENT', v_action,
                v_Xml, current_date, current_date, 0, 3,
                3, 2, 2, NULL
               );

	EXCEPTION
		WHEN OTHERS THEN
			Begin
				pckTrigPfiHmM.PRO_traceScl(PAT.PATI_NIP, null, PAT.PATI_ID_ETAB, 'Evt constantes', 'T_EVTHM_AUTONOMIE_PATIENT_CMS', '0', SUBSTR(SQLERRM,1,2000));
			Exception
				When Others Then
					return;
			End;
    END;

END T_EVTHM_AUTONOMIE_PATIENT_CMS;
