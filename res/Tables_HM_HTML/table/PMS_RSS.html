TRIGGER "PCAP_XROUGE".T_EVT_PMS_RSS2_M
AFTER UPDATE
ON PMS_RSS
FOR EACH ROW
DECLARE
	RPatient		IDE_PATIENT%RowType;
	RSejour			IDE_SEJOUR%RowType;
	RUfEntree		BAS_UF%RowType;
	RErrGro			PMS_ERR_GROUPAGE%RowType;
	RGhm    		PMS_GHM%RowType;
	RVersion   		PMS_VERSION_GROUP%RowType;
	v_xml			SCL_EVT_DATA.EVND_VAL%Type;
	n_IdEvnt		SCL_EVT.EVNT_ID_EVNT%TYPE;
	n_IdEvnd		SCL_EVT_DATA.EVND_ID_EVND%TYPE;
	RRum			PMS_RUM%RowType;
	RUm				SAD_UM%RowType;
	RDiagPrin		SAD_DIAG_SAISIS%RowType;

	cursor cRum(idRss number) is
		Select /* T_EVT_PMS_RSS2_M */ *
		  From pms_rum
		 Where TRUM_ID_TRSS = idRss
			 And TRUM_IND_ATT = '0'
			 And TRUM_IND_SIM = '0'
			 And TRUM_IND_FIN = '1';

BEGIN
	-- #Version# 1.16 du 01/03/2016 #Version#

	-- On n'émet que si le rss est à transmettre
	If pckTrigPfiHmM.FUN_isRss2Send(:new.TRSS_ID_TRSS, :new.TRSS_SARGH_NUM_GHS)='0' Then
		return;
	end if;

	-- Lecture du séjour
	Select /* T_EVT_PMS_RSS2_M */ *
		Into RSejour
		From IDE_SEJOUR
		 Where SEJO_ID_SEJO in (Select s.SEJO_ID_SEJO
							From IDE_SEJOUR s, PMS_RUM r
								 Where s.SEJO_ID_SEJO = r.TRUM_ID_SEJO
								 And r.TRUM_ID_TRSS = :new.TRSS_ID_TRSS);

	-- Lecture de l'UF d'entrée
	If RSejour.SEJO_ID_UNFO is not null Then
		Select /* T_EVT_PMS_RSS2_M */ *
			Into RUfEntree
			From BAS_UF
		 Where UNFO_ID_UNFO = RSejour.SEJO_ID_UNFO;
	End if;

	-- Lecture du patient
	Select /* T_EVT_PMS_RSS2_M */ *
		Into RPatient
		From IDE_PATIENT
	 Where PATI_ID_PATI = RSejour.SEJO_ID_PATI;

	-- Lecture du GHM
	Select /* T_EVT_PMS_RSS2_M */ *
		Into RGhm
		From PMS_GHM
	 Where TGHM_ID_TGHM = :new.TRSS_ID_TGHM;

	-- Lecture du code erreur
	Select /* T_EVT_PMS_RSS2_M */ *
		Into RErrGro
		From PMS_ERR_GROUPAGE
	 Where ERGR_ID_ERGR = :new.TRSS_ID_ERGR;

	-- Lecture de la version de groupage
	Select /* T_EVT_PMS_RSS2_M */ *
		Into RVersion
		From PMS_VERSION_GROUP
	 Where VEGR_ID_VEGR = :new.TRSS_ID_VEGR;

	Select /* T_EVT_PMS_RSS2_M */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

	Insert /* T_EVT_PMS_RSS2_M */
	Into scl_Evt (evnt_id_evnt, evnt_date, evnt_version, evnt_id_etab,
				evnt_code_evt, evnt_id_obj, evnt_code_pere, evnt_id_pere,
				evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej, evnt_num_rss,
				evnt_action,
				EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
	Values (n_IdEvnt, sysdate, '0', RSejour.SEJO_ID_ETAB_CON,
			'RSS', :new.TRSS_ID_TRSS, null, null,
			RPatient.PATI_ID_PATI, RPAtient.PATI_NIP, RSejour.SEJO_ID_SEJO, RSejour.SEJO_NUM_SEJ_COMM, :new.TRSS_NUMERO, 'M',
		  	current_date, current_date, 0, :new.TRSS_UTI_CRE, :new.TRSS_UTI_MOD, :new.TRSS_PRO_CRE, :new.TRSS_PRO_MOD, :new.TRSS_COL_COM);

	-- Segment Patient
	v_xml := null;
	v_xml := v_xml || '<PATIENT>';
	v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', RPatient.PATI_ID_PATI);
	v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', RPatient.PATI_NOM_USU);
	v_xml := v_xml || pckPfiHmM.val2xml('NOM_NAI', RPatient.PATI_NOM_NAI);
	v_xml := v_xml || pckPfiHmM.val2xml('PRENOM', RPatient.PATI_PRENOM);
	v_xml := v_xml || pckPfiHmM.val2xml('DAT_NAI', RPatient.PATI_DAT_NAI);
	v_xml := v_xml || pckPfiHmM.val2xml('SEXE', RPatient.PATI_SEXE);
	v_xml := v_xml || pckPfiHmM.val2xml('NIP', RPatient.PATI_NIP);
	v_xml := v_xml || '</PATIENT>';

	Select /* T_EVT_PMS_RSS2_M */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

	Insert /* T_EVT_PMS_RSS2_M */
	Into scl_evt_data (evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val,
						   EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
	Values (n_IdEvnd, n_IdEvnt, 1, 'PATIENT', 'I', v_xml,
			current_date, current_date, 0, 3, 3, 2, 2, NULL);

	-- Segment Séjour
	v_xml := null;
	v_xml := v_xml || '<SEJOUR>';
	v_xml := v_xml || pckPfiHmM.val2xml('ID_SEJO', RSejour.SEJO_ID_SEJO);
	v_xml := v_xml || pckPfiHmM.val2xml('ID_UNFO', RSejour.SEJO_ID_UNFO);
	v_xml := v_xml || pckPfiHmM.val2xml('UNFO_CODE', RUfEntree.UNFO_CODE);
	v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', RSejour.SEJO_DAT_DEB);
	v_xml := v_xml || pckPfiHmM.val2xml('NUM_SEJ', RSejour.SEJO_NUM_SEJ_COMM);
	v_xml := v_xml || '</SEJOUR>';

	Insert /* T_EVT_PMS_RSS2_M */
	Into scl_evt_data (evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val,
					EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
	Values (n_IdEvnd+1, n_IdEvnt, 2, 'SEJOUR', 'I', v_xml,
			current_date, current_date, 0, 3, 3, 2, 2, NULL);

	-- Segment RSS
	v_xml := null;
	v_xml := v_xml || '<RSS>';
	v_xml := v_xml || pckPfiHmM.val2xml('ID_TRSS', :new.TRSS_ID_TRSS);
	v_xml := v_xml || pckPfiHmM.val2xml('SARGH_NUM_GHS', :new.TRSS_SARGH_NUM_GHS);
	v_xml := v_xml || pckPfiHmM.val2xml('NUMERO', :new.TRSS_NUMERO);
	v_xml := v_xml || pckPfiHmM.val2xml('ERGR_COD_ERR_GRO', RErrGro.ERGR_COD_ERR_GRO);
	v_xml := v_xml || pckPfiHmM.val2xml('TGHM_NUMERO', RGhm.TGHM_NUMERO);
	v_xml := v_xml || pckPfiHmM.val2xml('TGHM_CMD', RGhm.TGHM_CMD);
	v_xml := v_xml || pckPfiHmM.val2xml('VEGR_FORMAT', RVersion.VEGR_FORMAT);
	v_xml := v_xml || '</RSS>';

	Insert /* T_EVT_PMS_RSS2_M */
	Into scl_evt_data (evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val,
					EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
	Values (n_IdEvnd+2, n_IdEvnt, 3, 'RSS', 'M', v_xml,
			current_date, current_date, 0, 3, 3, 2, 2, NULL);

EXCEPTION
	WHEN OTHERS THEN
		Begin
			pckTrigPfiHmM.PRO_traceScl(RPatient.PATI_NIP, null, RPatient.PATI_ID_ETAB, 'Evts RSS', 'T_EVT_PMS_RSS2_M', '0', SUBSTR(SQLERRM,1,2000));
		Exception
			When Others Then
				return;
		End;
END;
TRIGGER "PCAP_XROUGE".T_EVT_PMS_RSS_S
AFTER DELETE
ON PMS_RSS
FOR EACH ROW
DECLARE
	RPatient		IDE_PATIENT%RowType;
	RSejour			IDE_SEJOUR%RowType;
	RUfEntree		BAS_UF%RowType;
	RErrGro			PMS_ERR_GROUPAGE%RowType;
	v_xml			SCL_EVT_DATA.EVND_VAL%Type;
	n_IdEvnt		SCL_EVT.EVNT_ID_EVNT%TYPE;
	n_IdEvnd		SCL_EVT_DATA.EVND_ID_EVND%TYPE;
	n_idOldEvt		scl_evt.evnt_id_evnt%Type;

BEGIN
	-- #Version# 1.16 du 01/03/2016 #Version#

	-- Au moment de la suppression, le RSS est orphelin.
	-- Tous ses RUM ayant été dégagés, on ne peut plus faire de lien vers le séjour
	-- On le recherche donc dans les évènements précédants, et si on le trouve, on le poste à Suppression

	select /* T_EVT_PMS_RSS_S */ nvl(max(evnt_id_evnt), 0)
	  into n_idOldEvt
	  From scl_evt
	 where evnt_id_obj = :old.TRSS_ID_TRSS;

	if n_idOldEvt <> 0 then
		Select /* T_EVT_PMS_RSS_S */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		Insert /* T_EVT_PMS_RSS_S */
		Into scl_Evt (evnt_id_evnt, evnt_date, evnt_version, evnt_id_etab,
						evnt_code_evt, evnt_id_obj, evnt_code_pere, evnt_id_pere,
						evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej, evnt_num_rss, evnt_action,
	      				EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
		Select n_IdEvnt, sysdate, '0', evnt_id_etab,
					'RSS', :old.TRSS_ID_TRSS, null, null,
					evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej, :old.TRSS_NUMERO, 'S',
