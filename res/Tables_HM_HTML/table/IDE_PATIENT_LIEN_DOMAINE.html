TRIGGER_NAME TRIGGER_TYPE TRIGGER_OWNER TRIGGERING_EVENT STATUS TABLE_NAME
T_EVTHM_PATIENT_LIEN_DOMAINE_C AFTER EACH ROW PCAP_XROUGE INSERT ENABLED IDE_PATIENT_LIEN_DOMAINE 



TRIGGER "PCAP_XROUGE".T_EVTHM_PATIENT_LIEN_DOMAINE_C
AFTER INSERT
ON IDE_PATIENT_LIEN_DOMAINE
FOR EACH ROW
BEGIN
	DECLARE
		n_IdEvnt					SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd					SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml						SCL_EVT_DATA.EVND_VAL%TYPE;
		n_idApli					SCL_APPLI.APLI_ID%Type;
		v_idTrameOrigine			VARCHAR2(19);
		v_action					scl_evt.EVNT_ACTION%Type;
		v_evtHL7					SCL_EVT.EVNT_EVTHL7%Type;
		PAT							IDE_PATIENT%ROWTYPE;
		v_srvNip					scl_appli.APLI_SRVIPP%type;
		v_patiStatut				bas_parametre.para_valeur%Type;

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#
		SELECT /* T_EVTHM_PATIENT_LIEN_DOMAINE_C */ * INTO PAT FROM IDE_PATIENT
		WHERE PATI_ID_PATI = :new.DOML_ID_PATI;

		-- Vérifier si nip renseigné
		v_patiStatut := pckPfiHmM.FUN_litValParam('NUM_PATI_STATUT', PAT.PATI_ID_ETAB, current_date);
		If v_patiStatut = 'J' Then
			v_srvNip := '0';
		Else
			Begin
				Select /* T_EVTHM_PATIENT_LIEN_DOMAINE_C */ max(APLI_SRVIPP)
				  Into v_srvNip
				  From scl_appli
				 Where APLI_TYPE in ('HM','OM')
				   And APLI_ID_ETAB = PAT.PATI_ID_ETAB;

			Exception
				When others Then
					v_srvNip := '0';
			End;
		End If;

		If v_srvNip = '1' Then
			-- HM est serveur d'IPP
			If PAT.PATI_NIP is null Then
				return;
			End If;
		End If;

		-- Action
		v_action := 'C'	;

		-- Evénement HL7
		v_evtHL7 := 'A47' || '---';

		-- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
		pckTrigPfiHmM.PRO_appliExt(:new.DOML_DAT_MOD, :new.DOML_COL_COM, n_idApli, v_idTrameOrigine);

		Select /* T_EVTHM_PATIENT_LIEN_DOMAINE_C */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		INSERT /* T_EVTHM_PATIENT_LIEN_DOMAINE_C */
		INTO SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB, EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ,
					EVNT_NUM_RSS, EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
					EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
		VALUES (n_IdEvnt, current_date, '0', PAT.PATI_ID_ETAB, PAT.PATI_ID_PATI, PAT.PATI_NIP, NULL,NULL,
				NULL, 'IDENT_PATIENT', :new.DOML_ID_DOML, NULL, NULL, v_action, n_IdApli, v_idTrameOrigine, v_evtHl7,
				current_date, current_date, 0, :new.DOML_UTI_CRE, :new.DOML_UTI_MOD, :new.DOML_PRO_CRE, :new.DOML_PRO_MOD, :new.DOML_COL_COM);

		-- Segment Patient
		v_xml := null;
	    v_xml := v_xml || '<PATIENT>';
	    v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', PAT.PATI_ID_PATI);
	    v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', PAT.PATI_NOM_USU);
	    v_xml := v_xml || pckPfiHmM.val2xml('NOM_NAI', PAT.PATI_NOM_NAI);
	    v_xml := v_xml || pckPfiHmM.val2xml('PRENOM', PAT.PATI_PRENOM);
	    v_xml := v_xml || pckPfiHmM.val2xml('DAT_NAI', PAT.PATI_DAT_NAI);
	    v_xml := v_xml || pckPfiHmM.val2xml('SEXE', PAT.PATI_SEXE);
	    v_xml := v_xml || pckPfiHmM.val2xml('NIP', PAT.PATI_NIP);
	    v_xml := v_xml || '</PATIENT>';

		Select /* T_EVTHM_PATIENT_LIEN_DOMAINE_C */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		INSERT /* T_EVTHM_PATIENT_LIEN_DOMAINE_C */
		INTO SCL_EVT_DATA (EVND_ID_EVND, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
    					EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd, n_IdEvnt, 1, 'PATIENT', v_action, v_Xml,
    			current_date, current_date, 0, 3, 3, 2, 2, NULL);

	EXCEPTION
		WHEN OTHERS THEN
			Begin
				pckTrigPfiHmM.PRO_traceScl(PAT.PATI_NIP, null, PAT.PATI_ID_ETAB, 'Evt Patient', 'T_EVTHM_PATIENT_LIEN_DOMAINE_C', '0', SUBSTR(SQLERRM,1,2000));
			Exception
				When Others Then
					return;
			End;
	END;

END T_EVTHM_PATIENT_LIEN_DOMAINE_C;
