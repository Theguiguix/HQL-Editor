ON IDE_PATIENT
FOR EACH ROW
BEGIN
	DECLARE
		n_IdEvnt				SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd				SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml					SCL_EVT_DATA.EVND_VAL%TYPE;
		n_idApli				SCL_APPLI.APLI_ID%Type;

		v_srvNip				scl_appli.APLI_SRVIPP%type;
		v_patiStatut			bas_parametre.para_valeur%Type;

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#
		v_evtIndNoIpp := '-';

		-- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
		pckTrigPfiHmM.PRO_appliExt(:new.PATI_DAT_MOD, :new.PATI_COL_COM, n_idApli, v_idTrameOrigine);

		v_patiStatut := pckPfiHmM.FUN_litValParam('NUM_PATI_STATUT', :new.PATI_ID_ETAB, current_date);

		If v_patiStatut = 'J' Then
			v_srvNip := '0';
		Else
			Begin
				Select /* T_EVTHM_PATIENT_C */ max(APLI_SRVIPP)
				  Into v_srvNip
				  From scl_appli
				 Where APLI_TYPE in ('HM','OM')
				   And APLI_ID_ETAB = :new.PATI_ID_ETAB;

			Exception
				When others Then
					v_srvNip := '0';
			End;
		End If;

		-- HM est serveur d'IPP
		If v_srvNip = '1' Then
			If :new.pati_nip is not null Then
				v_action := 'C'	;
				v_evtHL7 := 'A28';
			Else
				return;
			End If;
		-- HM n'est pas serveur d'IPP
		Else
			If :new.pati_nip is not null Then
				v_action := 'C'	;
				v_evtHL7 := 'A28';
			Else
				v_evtIndNoIpp := 'D';
				v_Action := 'D';
				v_evtHL7 := 'A28';
				:new.PATI_COL_COM := SUBSTR( SUBSTR(:new.PATI_COL_COM, 1, 5)  || '     ', 1, 5) || 'D' || SUBSTR(:new.PATI_COL_COM, 7);
			End If;
		End If;

		If v_evtHL7 is not null Then
			v_evtHL7 := v_evtHL7 || v_evtIndNoIpp || '--';
		End If;

		Select /* T_EVTHM_PATIENT_C */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		INSERT /* T_EVTHM_PATIENT_C */
		INTO SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB, EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ,
				EVNT_NUM_RSS, EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
      			EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
		VALUES (n_IdEvnt, current_date, '0', :new.PATI_ID_ETAB, :new.PATI_ID_PATI, :new.PATI_NIP, NULL, NULL,
				NULL, 'PATIENT', :new.PATI_ID_PATI, NULL, NULL, v_action, n_IdApli, v_idTrameOrigine, v_evtHl7,
      			current_date, current_date, 0, :new.PATI_UTI_CRE, :new.PATI_UTI_MOD, :new.PATI_PRO_CRE, :new.PATI_PRO_MOD, :new.PATI_COL_COM);

		v_Xml := null;
		v_Xml := v_Xml || '<PATIENT>';
		v_Xml := v_Xml || pckPfiHmM.val2xml('ID_PATI', :new.PATI_ID_PATI);
		v_Xml := v_Xml || pckPfiHmM.val2xml('NOM_USU', :new.PATI_NOM_USU);
		v_Xml := v_Xml || pckPfiHmM.val2xml('NOM_NAI', :new.PATI_NOM_NAI);
		v_Xml := v_Xml || pckPfiHmM.val2xml('PRENOM', :new.PATI_PRENOM);
		v_Xml := v_Xml || pckPfiHmM.val2xml('DAT_NAI', :new.PATI_DAT_NAI);
		v_Xml := v_Xml || pckPfiHmM.val2xml('SEXE', :new.PATI_SEXE);
		v_Xml := v_Xml || pckPfiHmM.val2xml('NIP', :new.PATI_NIP);
		v_Xml := v_Xml || pckPfiHmM.val2xml('NIP_OLD', :new.PATI_NIP);
		v_Xml := v_Xml || pckPfiHmM.val2xml('ID_PATI_MAI', :new.PATI_ID_PATI_MAI);
		v_Xml := v_Xml || pckPfiHmM.val2xml('DATE_SUP', :new.PATI_DAT_SUP);
		v_Xml := v_Xml || '</PATIENT>';

		Select /* T_EVTHM_PATIENT_C */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		INSERT /* T_EVTHM_PATIENT_C */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
    			EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd, n_IdEvnt, 1, 'PATIENT', v_action, v_Xml,
      		current_date, current_date, 0, 3, 3, 2, 2, NULL);

	EXCEPTION
		WHEN OTHERS THEN
			Begin
				pckTrigPfiHmM.PRO_traceScl(:new.PATI_NIP, null, :new.PATI_ID_ETAB, 'Evt Patient', 'T_EVTHM_PATIENT_C', '0', SUBSTR(SQLERRM,1,2000));
			Exception
				When Others Then
					return;
			End;
	END;

END T_EVTHM_PATIENT_C;
