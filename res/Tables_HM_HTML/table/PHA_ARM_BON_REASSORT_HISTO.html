T_EVTHM_REASSORT_ARMOIRE_C AFTER EACH ROW PCAP_XROUGE UPDATE DISABLED PHA_ARM_BON_REASSORT_HISTO 



TRIGGER "PCAP_XROUGE".T_EVTHM_REASSORT_ARMOIRE_C
AFTER UPDATE
ON PHA_ARM_BON_REASSORT_HISTO
FOR EACH ROW
DECLARE
	n_IdEvnt			SCL_EVT.EVNT_ID_EVNT%TYPE;
	n_IdEvnd			SCL_EVT_DATA.EVND_ID_EVND%TYPE;
	v_xml              	scl_evt_data.evnd_val%TYPE;
    n_idapli           	scl_appli.apli_id%TYPE;
    v_idtrameorigine   	VARCHAR2 (19);
    n_idcage_e         	bas_catalogue_gen.cage_id_cage%TYPE;
    n_idcage_ep        	bas_catalogue_gen.cage_id_cage%TYPE;
    n_idcage_s         	bas_catalogue_gen.cage_id_cage%TYPE;
    n_idcage_t         	bas_catalogue_gen.cage_id_cage%TYPE;
    n_nb_envois        	NUMBER;

BEGIN
	-- #Version# 1.16 du 01/03/2016 #Version#

	-- Si l'état de le bon de commande n'est pas à "validée pharmacie" et état n'est pas "livrée" ou "livraisonpartielle" -> on ne génère pas d'évènement
    SELECT /* T_EVTHM_REASSORT_ARMOIRE_C */ a.cage_id_cage
	  INTO n_idcage_e
	  FROM bas_catalogue_gen a, bas_type_info b
     WHERE a.cage_id_tyin = b.tyin_id_tyin
	   AND a.cage_code ='LP'
	   AND b.tyin_code = 'PHARM_ETAT';

  	SELECT /* T_EVTHM_REASSORT_ARMOIRE_C */ a.cage_id_cage
      INTO n_idcage_ep
      FROM bas_catalogue_gen a, bas_type_info b
     WHERE a.cage_id_tyin = b.tyin_id_tyin
       AND a.cage_code ='LI'
       AND b.tyin_code = 'PHARM_ETAT';

  	SELECT /* T_EVTHM_REASSORT_ARMOIRE_C */ a.cage_id_cage
      INTO n_idcage_s
      FROM bas_catalogue_gen a, bas_type_info b
     WHERE a.cage_id_tyin = b.tyin_id_tyin
       AND a.cage_code = 'VP'
       AND b.tyin_code = 'PHARM_STATUT';

 	-- On déclenche le trigger si le bon de commande passe à "livré" ou si col_com position 7 a changé
    IF NOT (:NEW.pabh_id_cage_statut = n_idcage_s AND :NEW.pabh_id_cage_statut <> :OLD.pabh_id_cage_statut) THEN
    	RETURN;
    END IF;

	If (:NEW.pabh_id_cage_etat  <> n_idcage_e  and :NEW.pabh_id_cage_etat  <> n_idcage_ep)  or :NEW.pabh_id_cage_etat is null Then
    	Return;
    End If;

  	-- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
    pcktrigpfihmm.pro_appliext (:NEW.pabh_dat_mod, :NEW.pabh_col_com, n_idapli, v_idtrameorigine);

	Select /* T_EVTHM_REASSORT_ARMOIRE_C */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

  	INSERT /* T_EVTHM_REASSORT_ARMOIRE_C */
  	INTO scl_evt (evnt_id_evnt, evnt_date, evnt_version, evnt_id_etab,
    		evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej,
			evnt_num_rss, evnt_code_evt, evnt_id_obj, evnt_code_pere,
			evnt_id_pere, evnt_action, evnt_id_apli, evnt_id_evt_externe,
			EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
	VALUES (n_IdEvnt, current_date, '0', :NEW.pabh_id_etab,
            NULL, NULL, NULL, NULL,
            NULL, 'REASSORTARM', :NEW.pabh_id_pabh, NULL,
            NULL, 'C', n_idapli, v_idtrameorigine,
            current_date, current_date, 0, :NEW.pabh_UTI_CRE, :NEW.pabh_UTI_MOD, :NEW.pabh_PRO_CRE, :NEW.pabh_PRO_MOD, :NEW.pabh_COL_COM);

	v_xml := NULL;
    v_xml := v_xml || '<REASSORTARM>';
    v_xml := v_xml || pckpfihmm.val2xml ('CODECOMM', :NEW.pabh_cod_commande);
    v_xml := v_xml || pckpfihmm.val2xml ('DATECOMM', TO_CHAR (:NEW.pabh_dat_commande,'DD/MM/YYYYY HH24:MI:SS'));
    v_xml := v_xml || pckpfihmm.val2xml ('NATURE', :NEW.pabh_nature);
    v_xml := v_xml || '</REASSORTARM>';

	Select /* T_EVTHM_REASSORT_ARMOIRE_C */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

    INSERT /* T_EVTHM_REASSORT_ARMOIRE_C */
    INTO scl_evt_data(evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val,
	    			EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
	VALUES (n_IdEvnd, n_IdEvnt, 1, 'REASSORTARM', 'C', v_xml,
	      	current_date, current_date, 0, 3, 3, 2, 2, NULL);

EXCEPTION
	WHEN OTHERS THEN
		Begin
			pckTrigPfiHmM.PRO_traceScl(null, null, :NEW.pabh_id_etab, 'Evt_reassort_armoire', 'T_EVTHM_REASSORT_ARMOIRE_C', '0', SUBSTR(SQLERRM,1,2000));
		Exception
			When Others Then
				return;
		End;
END T_EVTHM_REASSORT_ARMOIRE_C;
