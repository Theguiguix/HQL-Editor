TRIGGER "PCAP_XROUGE".T_EVTHM_SCHG_M
AFTER UPDATE
ON SOI_JOUR_CHIRURGIE_TEMPS_OP
FOR EACH ROW
BEGIN
	DECLARE
		n_IdEvnt					SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd					SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml						SCL_EVT_DATA.EVND_VAL%TYPE;
		n_idApli					SCL_APPLI.APLI_ID%Type;
		v_idTrameOrigine			VARCHAR2(19);
		v_idPati					IDE_PATIENT.PATI_ID_PATI%Type;
		v_nip					    IDE_PATIENT.PATI_NIP%Type;
		v_idSejo					SOI_JOUR_CHIRURGIE.SCHI_ID_SEJO%Type;
		v_nda					    IDE_SEJOUR.SEJO_NUM_SEJ_COMM%Type;
		v_idEtab					BAS_ETABLISSEMENT.ETAB_ID_ETAB%Type;
		v_boolTechAnest				boolean;
		v_boolTopHoraire			boolean;

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#
		If   (:new.SCHG_TEC_ANESTHESIE = :old.SCHG_TEC_ANESTHESIE or (:new.SCHG_TEC_ANESTHESIE is null and :old.SCHG_TEC_ANESTHESIE is null)) and
			 (SUBSTR(:new.SCHG_COL_COM, 7, 1) = SUBSTR(:old.SCHG_COL_COM, 7, 1) or (SUBSTR(:new.SCHG_COL_COM,7,1) is null and SUBSTR(:old.SCHG_COL_COM,7,1) is null))
		Then
			v_boolTechAnest := false;
		Else
			v_boolTechAnest := true;
		End If;

		If   (:new.SCHG_ENT_INDUCTION = :old.SCHG_ENT_INDUCTION or (:new.SCHG_ENT_INDUCTION is null and :old.SCHG_ENT_INDUCTION is null)) and
			 (:new.SCHG_SOR_INDUCTION = :old.SCHG_SOR_INDUCTION or (:new.SCHG_SOR_INDUCTION is null and :old.SCHG_SOR_INDUCTION is null)) and
			 (:new.SCHG_ENT_SAL_OPERATION = :old.SCHG_ENT_SAL_OPERATION or (:new.SCHG_ENT_SAL_OPERATION is null and :old.SCHG_ENT_SAL_OPERATION is null)) and
			 (:new.SCHG_SOR_SAL_OPERATION = :old.SCHG_SOR_SAL_OPERATION or (:new.SCHG_SOR_SAL_OPERATION is null and :old.SCHG_SOR_SAL_OPERATION is null)) and
			 (:new.SCHG_ENT_SSPI = :old.SCHG_ENT_SSPI or (:new.SCHG_ENT_SSPI is null and :old.SCHG_ENT_SSPI is null)) and
			 (:new.SCHG_PRE_SOR_SSPI = :old.SCHG_PRE_SOR_SSPI or (:new.SCHG_PRE_SOR_SSPI is null and :old.SCHG_PRE_SOR_SSPI is null)) and
			 (:new.SCHG_SOR_EFF_SSPI = :old.SCHG_SOR_EFF_SSPI or (:new.SCHG_SOR_EFF_SSPI is null and :old.SCHG_SOR_EFF_SSPI is null)) and
			 (:new.SCHG_OUVERTURE = :old.SCHG_OUVERTURE or (:new.SCHG_OUVERTURE is null and :old.SCHG_OUVERTURE is null)) and
			 (:new.SCHG_FERMETURE = :old.SCHG_FERMETURE or (:new.SCHG_FERMETURE is null and :old.SCHG_FERMETURE is null)) and
			 (:new.SCHG_ENT_BLOC = :old.SCHG_ENT_BLOC or (:new.SCHG_ENT_BLOC is null and :old.SCHG_ENT_BLOC is null)) and
			 (:new.SCHG_DAT_SOR_BLOC = :old.SCHG_DAT_SOR_BLOC or (:new.SCHG_DAT_SOR_BLOC is null and :old.SCHG_DAT_SOR_BLOC is null)) and
			 (SUBSTR(:new.SCHG_COL_COM, 7, 1) = SUBSTR(:old.SCHG_COL_COM, 7, 1) or (SUBSTR(:new.SCHG_COL_COM,7,1) is null and SUBSTR(:old.SCHG_COL_COM,7,1) is null))
		Then
			v_boolTopHoraire := false;
		Else
			v_boolTopHoraire := true;
		End If;

		-- noPerroquet
		If Not v_boolTechAnest And Not v_boolTopHoraire Then
			return;
		End If;

		-- Relecture du patient
		SELECT /* T_EVTHM_SCHG_M */ PATI_ID_PATI, PATI_NIP, SCHI_ID_SEJO, PATI_ID_ETAB, SEJO_NUM_SEJ_COMM
		INTO v_idPati, v_nip, v_idSejo, v_idEtab, v_nda
		FROM SOI_JOUR_CHIRURGIE, IDE_PATIENT, IDE_SEJOUR
		WHERE SCHI_ID_PATI = PATI_ID_PATI
		AND SCHI_ID_SCHI = :new.SCHG_ID_SCHI
		AND SCHI_ID_SEJO = SEJO_ID_SEJO;

		-- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
		pckTrigPfiHmM.PRO_appliExt(:new.SCHG_DAT_MOD, :new.SCHG_COL_COM, n_idApli, v_idTrameOrigine);

		-- évènement technique anesthésie
		If v_boolTechAnest Then
			Select /* T_EVTHM_SCHG_M */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

			INSERT /* T_EVTHM_SCHG_M */
			INTO SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB, EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ,
					EVNT_NUM_RSS, EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
	      			EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', v_idEtab, v_idPati, v_nip, v_idSejo, v_nda,
					NULL, 'TECH_ANEST', :new.SCHG_ID_SCHG, NULL, NULL, 'M', n_IdApli, v_idTrameOrigine, 'S03',
	      			current_date, current_date, 0, :new.SCHG_UTI_CRE, :new.SCHG_UTI_MOD, :new.SCHG_PRO_CRE, :new.SCHG_PRO_MOD, :new.SCHG_COL_COM);

			-- Segment technique anesthésie
		    v_xml := null;
		    v_xml := v_xml || '<TECH_ANEST>';
		    v_xml := v_xml || pckPfiHmM.val2xml('ID_SCHG', :new.SCHG_ID_SCHG);
		    v_xml := v_xml || pckPfiHmM.val2xml('TEC_ANESTHESIE', :new.SCHG_TEC_ANESTHESIE);
		    v_xml := v_xml || '</TECH_ANEST>';

			Select /* T_EVTHM_SCHG_M */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		    INSERT /* T_EVTHM_SCHG_M */
		    INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
						   EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
		    VALUES (n_IdEvnd, n_IdEvnt, 1, 'TECH_ANEST', 'M', v_Xml,
					current_date, current_date, 0, 3, 3, 2, 2, NULL);
		End If;

		-- évènement top horaire
		If v_boolTopHoraire Then
			Select /* T_EVTHM_SCHG_M */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

			INSERT /* T_EVTHM_SCHG_M */
			INTO SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB, EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ,
					EVNT_NUM_RSS, EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
	      			EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', v_idEtab, v_idPati, v_nip, v_idSejo, v_nda,
					NULL, 'TOP_HORAIRE', :new.SCHG_ID_SCHG, NULL, NULL, 'M', n_IdApli, v_idTrameOrigine, 'R01',
	      			current_date, current_date, 0, :new.SCHG_UTI_CRE, :new.SCHG_UTI_MOD, :new.SCHG_PRO_CRE, :new.SCHG_PRO_MOD, :new.SCHG_COL_COM);

			-- Segment top horaire
		    v_xml := null;
		    v_xml := v_xml || '<TOP_HORAIRE>';
		    v_xml := v_xml || pckPfiHmM.val2xml('ID_SCHG', :new.SCHG_ID_SCHG);
		    v_xml := v_xml || pckPfiHmM.val2xml('ENT_INDUCTION', :new.SCHG_ENT_INDUCTION);
			v_xml := v_xml || pckPfiHmM.val2xml('SOR_INDUCTION', :new.SCHG_SOR_INDUCTION);
			v_xml := v_xml || pckPfiHmM.val2xml('ENT_SAL_OPERATION', :new.SCHG_ENT_SAL_OPERATION);
			v_xml := v_xml || pckPfiHmM.val2xml('SOR_SAL_OPERATION', :new.SCHG_SOR_SAL_OPERATION);
		    v_xml := v_xml || pckPfiHmM.val2xml('ENT_SSPI', :new.SCHG_ENT_SSPI);
			v_xml := v_xml || pckPfiHmM.val2xml('PRE_SOR_SSPI', :new.SCHG_PRE_SOR_SSPI);
			v_xml := v_xml || pckPfiHmM.val2xml('SOR_EFF_SSPI', :new.SCHG_SOR_EFF_SSPI);
			v_xml := v_xml || pckPfiHmM.val2xml('OUVERTURE', :new.SCHG_OUVERTURE);
			v_xml := v_xml || pckPfiHmM.val2xml('FERMETURE', :new.SCHG_FERMETURE);
			v_xml := v_xml || pckPfiHmM.val2xml('ENT_BLOC', :new.SCHG_ENT_BLOC);
			v_xml := v_xml || pckPfiHmM.val2xml('DAT_SOR_BLOC', :new.SCHG_DAT_SOR_BLOC);
		    v_xml := v_xml || '</TOP_HORAIRE>';

			Select /* T_EVTHM_SCHG_M */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		    INSERT /* T_EVTHM_SCHG_M */
		    INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
						   EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
		    VALUES (n_IdEvnd, n_IdEvnt, 1, 'TOP_HORAIRE', 'M', v_Xml,
					current_date, current_date, 0, 3, 3, 2, 2, NULL);
		End If;

	EXCEPTION
		WHEN OTHERS THEN
			Begin
				pckTrigPfiHmM.PRO_traceScl(v_nip, null, v_idEtab, 'Technique anesthésie et tops horaires', 'T_EVTHM_JOUR_CHIRURGIE_TEMPS_OP_M', '0', SUBSTR(SQLERRM,1,2000));
			Exception
				When Others Then
					return;
			End;
	END;
END T_EVTHM_SCHG_M;
