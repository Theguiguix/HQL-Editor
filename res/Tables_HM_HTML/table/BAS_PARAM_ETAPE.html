TRIGGER_NAME TRIGGER_TYPE TRIGGER_OWNER TRIGGERING_EVENT STATUS TABLE_NAME
T_EVTHM_PARAM_ETAPE_CMS AFTER EACH ROW PCAP_XROUGE INSERT OR UPDATE OR DELETE DISABLED BAS_PARAM_ETAPE 



TRIGGER "PCAP_XROUGE".T_EVTHM_PARAM_ETAPE_CMS
AFTER INSERT or UPDATE or DELETE
ON BAS_PARAM_ETAPE
FOR EACH ROW
BEGIN
	DECLARE
		n_IdEvnt				SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd				SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml					SCL_EVT_DATA.EVND_VAL%TYPE;
		n_idApli				SCL_APPLI.APLI_ID%Type;
		v_idTrameOrigine		VARCHAR2(19);
		v_Action				VARCHAR2(1);
		n_idPaet				BAS_PARAM_ETAPE.PAET_ID_PAET%Type;
		n_idPaam				BAS_PARAM_ETAPE.PAET_ID_PAAM%Type;
		n_utiCre				BAS_PARAM_ETAPE.PAET_UTI_CRE%Type;
		n_utiMod				BAS_PARAM_ETAPE.PAET_UTI_MOD%Type;
		n_proCre				BAS_PARAM_ETAPE.PAET_PRO_CRE%Type;
		n_proMod				BAS_PARAM_ETAPE.PAET_PRO_MOD%Type;
		v_colCom				BAS_PARAM_ETAPE.PAET_COL_COM%Type;
		n_idMoth				RGU_MOTIF.MOTH_ID_MOTH%Type;
		v_codeMoth				RGU_MOTIF.MOTH_CODE%Type;
		v_libelleMoth			RGU_MOTIF.MOTH_LIBELLE%Type;
		n_idEtab				RGU_MOTIF.MOTH_ID_ETAB%Type;

		CURSOR c_moth(idPaam number) IS
		SELECT /* T_EVTHM_PARAM_ETAPE_CMS */ moth_id_moth, moth_id_etab, moth_code, moth_libelle
		  FROM rgu_motif
		 WHERE moth_id_moth IN (SELECT lmam_id_moth
		                          FROM bas_lien_motif_acte, bas_param_acte_medical
		                         WHERE lmam_id_paam = paam_id_paam
		                           AND paam_id_paam = idPaam);

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#

		-- on ne crée un événement que sur modification de la durée de l'étape
		If updating Then
			If (:new.PAET_DUR_ETAPE = :old.PAET_DUR_ETAPE or (:new.PAET_DUR_ETAPE is null and :old.PAET_DUR_ETAPE is null)) and
				 (nvl(substr(:new.PAET_COL_COM,7,1),'0') = nvl(substr(:old.PAET_COL_COM,7,1),'0'))
			Then
				return;
			End If;
		End If;

		If Deleting Then
			IF SUBSTR(:old.PAET_COL_COM, 4, 1) = '1' THEN
				pckTrigPfiHmM.PRO_appliExt(:old.PAET_DAT_MOD, :old.PAET_COL_COM, n_idApli, v_idTrameOrigine);
			End If;
		Else
			pckTrigPfiHmM.PRO_appliExt(:new.PAET_DAT_MOD, :new.PAET_COL_COM, n_idApli, v_idTrameOrigine);
		End If;

		v_Action := 'M';

		If Deleting Then
			n_idPaet := :old.PAET_ID_PAET;
			n_idPaam := :old.PAET_ID_PAAM;
			n_utiCre := :old.PAET_UTI_CRE;
			n_utiMod := :old.PAET_UTI_MOD;
			n_proCre := :old.PAET_PRO_CRE;
			n_proMod := :old.PAET_PRO_MOD;
			v_colCom := :old.PAET_COL_COM;
		Else
			n_idPaet := :new.PAET_ID_PAET;
			n_idPaam := :new.PAET_ID_PAAM;
			n_utiCre := :new.PAET_UTI_CRE;
			n_utiMod := :new.PAET_UTI_MOD;
			n_proCre := :new.PAET_PRO_CRE;
			n_proMod := :new.PAET_PRO_MOD;
			v_colCom := :new.PAET_COL_COM;
		End If;

		OPEN c_moth(n_idPaam);
	 	LOOP
		 	FETCH c_moth INTO n_idMoth, n_idEtab, v_codeMoth, v_libelleMoth;
		 	EXIT WHEN c_moth%NOTFOUND;

			Select /* T_EVTHM_PARAM_ETAPE_CMS */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

			Insert /* T_EVTHM_PARAM_ETAPE_CMS */
			Into SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB, EVNT_ID_PATIENT,
								 EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ, EVNT_NUM_RSS, EVNT_CODE_EVT,
								 EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION, EVNT_ID_APLI,
								 EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7, EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ,
								 EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', n_idEtab, NULL,
					NULL, NULL, NULL, NULL,	'MOTIF_HOSPI',
					n_idMoth, NULL, NULL, v_Action,	n_IdApli,
					v_idTrameOrigine, NULL, current_date, current_date, 0,
					n_utiCre, n_utiMod, n_proCre, n_proMod, v_colCom);

			v_xml := null;
			v_xml := v_xml || '<MOTIF_HOSPI>';
			v_xml := v_xml || pckPfiHmM.val2xml('ID_MOTH', n_idMoth);
			v_xml := v_xml || pckPfiHmM.val2xml('CODE', v_codeMoth);
			v_xml := v_xml || pckPfiHmM.val2xml('LIBELLE',v_libelleMoth);
			v_xml := v_xml || pckPfiHmM.val2xml('ID_PAET', n_idPaet);
			v_xml := v_xml || '</MOTIF_HOSPI>';

			Select /* T_EVTHM_PARAM_ETAPE_CMS */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

			INSERT /* T_EVTHM_PARAM_ETAPE_CMS */
			INTO SCL_EVT_DATA (EVND_ID_EVND, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION,
									  EVND_VAL, EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE,
									  EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
			VALUES (n_IdEvnd, n_IdEvnt, 1, 'MOTIF_HOSPI', 'I',
					v_Xml, current_date, current_date, 0, 3,
					3, 2, 2, NULL);
		END LOOP;
	 	CLOSE c_moth;

	EXCEPTION
		WHEN OTHERS THEN
			Begin
				pckTrigPfiHmM.PRO_traceScl(NULL, NULL, n_idEtab, 'Evt Motif d hospitalisation', 'T_EVTHM_PARAM_ETAPE_CMS', '0', SUBSTR(SQLERRM,1,2000));
			Exception
				When Others Then
					return;
			End;
	END;
END T_EVTHM_PARAM_ETAPE_CMS;
