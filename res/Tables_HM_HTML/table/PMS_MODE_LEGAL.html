TRIGGER_NAME TRIGGER_TYPE TRIGGER_OWNER TRIGGERING_EVENT STATUS TABLE_NAME
T_EVTHM_MODE_LEGAL_CMS AFTER EACH ROW PCAP_XROUGE INSERT OR UPDATE OR DELETE ENABLED PMS_MODE_LEGAL 



TRIGGER "PCAP_XROUGE".T_EVTHM_MODE_LEGAL_CMS
AFTER INSERT or UPDATE or DELETE
ON PMS_MODE_LEGAL
FOR EACH ROW
BEGIN
	DECLARE
		n_CodRet				NUMBER;
		n_NoSej					NUMBER;
		PAT						IDE_PATIENT%ROWTYPE;
		SEJ						IDE_SEJOUR%ROWTYPE;
		v_TypSejBase 			BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		v_TypSej				BAS_CATALOGUE_PERS.CAPE_CODE%TYPE;
		v_UfCode				BAS_UF.UNFO_CODE%TYPE;
		v_UfTypSej				BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		v_LitCode				PAD_LIT_POSTE.LIPO_CODE%TYPE;
		v_ChambreCode			PAD_CHAMBRE_SALLE.CHAS_CODE%TYPE;
		n_IdEvnt				SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd				SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml					SCL_EVT_DATA.EVND_VAL%TYPE;
		n_idApli				SCL_APPLI.APLI_ID%Type;
		v_idTrameOrigine		VARCHAR2(19);
		v_LibEvt				VARCHAR2(30);
		n_IdEvt					NUMBER;

		-- Recherche du premier séjour de temps plein (Forme activité 01) psy postérieur au mode légal
		CURSOR C1 (dateModeLegal Date, noPatient Number) IS
			SELECT /* T_EVTHM_MODE_LEGAL_CMS */ SEJO_ID_SEJO
			  FROM IDE_SEJOUR, BAS_UF, BAS_CATEGORIE_UFM
			 WHERE SEJO_ID_PATI = noPatient
			   AND SEJO_DAT_DEB >= dateModeLegal
			   AND SEJO_DAT_SUP is Null
			   AND SEJO_IND_INCOMPLET = '0'
			   AND SEJO_NUM_SEJ_COMM is not null
			   AND UNFO_ID_UNFO = SEJO_ID_UNFO
			   AND CUFM_ID_CUFM = UNFO_ID_CUFM
			   AND CUFM_ID_CAGE_NAT_PEC = 2587
		ORDER BY SEJO_DAT_DEB;

		v_evtHL7				SCL_EVT.EVNT_EVTHL7%Type;
		v_evtIndNoNda			Varchar2(1);
		v_evtTypMvt				Varchar2(1);
		v_evtIndRec				Varchar2(1);
		v_srvNip				scl_appli.APLI_SRVIPP%type;
		v_sejoStatut			bas_parametre.para_valeur%Type;

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#
		v_evtIndNoNda	:= '-';
		v_evtTypMvt := 'E';
		v_evtIndRec := '-';

		If Updating Then
			-- noPerroquet
			If (:new.MLGL_DAT_DEB_VAL = :old.MLGL_DAT_DEB_VAL or (:new.MLGL_DAT_DEB_VAL is null and :old.MLGL_DAT_DEB_VAL is null)) and
				 (:new.MLGL_DAT_FIN_VAL = :old.MLGL_DAT_FIN_VAL or (:new.MLGL_DAT_FIN_VAL is null and :old.MLGL_DAT_FIN_VAL is null)) and
				 (:new.MLGL_ID_CAGE = :old.MLGL_ID_CAGE or (:new.MLGL_ID_CAGE is null and :old.MLGL_ID_CAGE is null)) and
				 (nvl(substr(:new.MLGL_COL_COM,7,1),'0') = nvl(substr(:old.MLGL_COL_COM,7,1),'0'))
			Then
				return;
			End If;
		End If;

		-- On détermine sur quel Séjour on va simuler un Update
		If Deleting Then
			Open C1 (:old.MLGL_DAT_DEB_VAL, :old.MLGL_ID_PATI);
		Else
			Open C1 (:new.MLGL_DAT_DEB_VAL, :new.MLGL_ID_PATI);
		End if;
		Fetch C1 Into n_NoSej;
		If C1%ROWCOUNT = 0 Then
			Return;
		End If;
		Close C1;

		-- Lecture du patient, du séjour, du mouvement et info rattachées
		If Deleting Then
			n_CodRet := pckPfiHmM.FUN_LirePAT(:old.MLGL_ID_PATI, PAT);
		Else
			n_CodRet := pckPfiHmM.FUN_LirePAT(:new.MLGL_ID_PATI, PAT);
		End if;
		n_CodRet := pckPfiHmM.FUN_LireSEJ(n_NoSej, SEJ);
		v_sejoStatut := pckPfiHmM.FUN_litValParam('NUM_SEJO_STATUT', SEJ.SEJO_ID_ETAB_CON, current_date);

		If v_sejoStatut = 'J' Then
			v_srvNip := '0';
		Else
			Begin
				Select /* T_EVTHM_MODE_LEGAL_CMS */ Max(APLI_SRVIPP)
				  Into v_srvNip
				  From scl_appli
				 Where APLI_TYPE in ('HM','OM')
				   And APLI_ID_ETAB = SEJ.SEJO_ID_ETAB_CON;

			Exception
				When others Then
					v_srvNip := '0';
			End;
		End If;

		If SEJ.SEJO_NUM_SEJ_COMM is null Then
			-- HM est il serveur d'ipp ?
			If v_srvNip = '1' Then
				return;
			End If;
		End If;


		n_CodRet := pckPfiHmM.FUN_LireUF(SEJ.SEJO_ID_UNFO, v_UfCode, v_UfTypSej);
		n_codRet := pckPfiHmM.FUN_LireLIT(SEJ.SEJO_ID_LIPO, v_ChambreCode, v_LitCode);

		-- Lecture type du séjour
		If SEJ.SEJO_ID_CAPE_VEN is not null Then
			SELECT /* T_EVTHM_MODE_LEGAL_CMS */ CAGE_CODE, CAPE_CODE
				INTO v_TypSejBase, v_TypSej
				FROM BAS_CATALOGUE_SPE, BAS_CATALOGUE_GEN, BAS_CATALOGUE_PERS
			 WHERE CAPE_ID_CAPE = SEJ.SEJO_ID_CAPE_VEN
				 AND CASP_ID_CAPE = CAPE_ID_CAPE
				 AND CASP_ID_CAGE = CAGE_ID_CAGE
				 AND CAGE_ID_TYIN = 4;
		End If;


		v_evtHL7 := 'Z99';

		If v_evtHL7 is not null Then
			v_evtHL7 := v_evtHL7 || v_evtIndNoNda || v_evtTypMvt || v_evtIndRec;
		End If;

		-- On stocke l'appli externe qui a généré l'évènement et l'Id de l'évènement Origine reçu, le cas échéant
		If Deleting Then
			If :old.MLGL_UTI_MOD = 3 Then
				pckTrigPfiHmM.PRO_appliExt(:old.MLGL_DAT_MOD, :old.MLGL_COL_COM, n_idApli, v_idTrameOrigine);
			End If;
		else
			If :new.MLGL_UTI_MOD = 3 Then
				pckTrigPfiHmM.PRO_appliExt(:new.MLGL_DAT_MOD, :new.MLGL_COL_COM, n_idApli, v_idTrameOrigine);
			End If;
		End If;

		Select /* T_EVTHM_MODE_LEGAL_CMS */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		If Deleting Then
			INSERT /* T_EVTHM_MODE_LEGAL_CMS */
			INTO SCL_EVT (
					EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB, EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ, EVNT_NUM_RSS, EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE,
					EVNT_ID_PERE, EVNT_ACTION, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
					EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', SEJ.SEJO_ID_ETAB_CON, PAT.PATI_ID_PATI, PAT.PATI_NIP, SEJ.SEJO_ID_SEJO, SEJ.SEJO_NUM_SEJ_COMM, NULL, 'SEJOUR', n_IdEvt, NULL,
					NULL, 'M', n_IdApli, v_idTrameOrigine, v_evtHl7,
					current_date, current_date, 0, :old.MLGL_UTI_CRE, :old.MLGL_UTI_MOD, :old.MLGL_PRO_CRE, :old.MLGL_PRO_MOD, :old.MLGL_COL_COM);

		Else
			INSERT /* T_EVTHM_MODE_LEGAL_CMS */
			INTO SCL_EVT (
					EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB, EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ, EVNT_NUM_RSS, EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE,
					EVNT_ID_PERE, EVNT_ACTION, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
					EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', SEJ.SEJO_ID_ETAB_CON, PAT.PATI_ID_PATI, PAT.PATI_NIP, SEJ.SEJO_ID_SEJO, SEJ.SEJO_NUM_SEJ_COMM, NULL, 'SEJOUR', n_IdEvt, NULL,
					NULL, 'M', n_IdApli, v_idTrameOrigine, v_evtHl7,
					current_date, current_date, 0, :new.MLGL_UTI_CRE, :new.MLGL_UTI_MOD, :new.MLGL_PRO_CRE, :new.MLGL_PRO_MOD, :new.MLGL_COL_COM);
		End if;

		v_xml := null;
		v_xml := v_xml || '<PATIENT>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', PAT.PATI_ID_PATI);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', PAT.PATI_NOM_USU);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_NAI', PAT.PATI_NOM_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('PRENOM', PAT.PATI_PRENOM);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_NAI', PAT.PATI_DAT_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('SEXE', PAT.PATI_SEXE);
		v_xml := v_xml || pckPfiHmM.val2xml('NIP', PAT.PATI_NIP);
		v_xml := v_xml || '</PATIENT>';

		Select /* T_EVTHM_MODE_LEGAL_CMS */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		INSERT /* T_EVTHM_MODE_LEGAL_CMS */
		INTO SCL_EVT_DATA (evnd_id_evnd,EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
    					EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd, n_IdEvnt, 1, 'PATIENT', 'I', v_Xml,
    			current_date, current_date, 0, 3, 3, 2, 2, NULL);


		v_xml := null;
		v_xml := v_xml || '<SEJOUR>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_SEJO', SEJ.SEJO_ID_SEJO);
		v_xml := v_xml || pckPfiHmM.val2xml('ID_UNFO', SEJ.SEJO_ID_UNFO);
		v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJBASE', v_TypSejBase);
		v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJ', v_TypSej);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_CODE', v_UfCode);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_TYPSEJ', v_UfTypSej);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', SEJ.SEJO_DAT_DEB);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_FIN', SEJ.SEJO_DAT_FIN);
		v_xml := v_xml || pckPfiHmM.val2xml('NUM_SEJ', SEJ.SEJO_NUM_SEJ_COMM);
		v_xml := v_xml || pckPfiHmM.val2xml('STATUT', SEJ.SEJO_STATUT);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_SUP', SEJ.SEJO_DAT_SUP);
		v_xml := v_xml || '</SEJOUR>';

		INSERT /* T_EVTHM_MODE_LEGAL_CMS */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
    					EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
		VALUES (n_IdEvnd+1, n_IdEvnt, 2, 'SEJOUR', 'M', v_Xml,
    			current_date, current_date, 0, 3, 3, 2, 2, NULL);

	EXCEPTION
		WHEN OTHERS THEN
			Begin
				pckTrigPfiHmM.PRO_traceScl(PAT.PATI_NIP, SEJ.SEJO_NUM_SEJ_COMM, SEJ.SEJO_ID_ETAB_CON, 'Evt Mode Legal', 'T_EVTHM_MODE_LEGAL_CMS', '0', SUBSTR(SQLERRM,1,2000));
			Exception
				When Others Then
					return;
			End;
	END;
END T_EVTHM_MODE_LEGAL_CMS;
