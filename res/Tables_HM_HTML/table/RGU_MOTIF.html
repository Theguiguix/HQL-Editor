TRIGGER "PCAP_XROUGE".T_EVTHM_MOTIF_CMS
AFTER INSERT or UPDATE or DELETE
ON RGU_MOTIF
FOR EACH ROW
BEGIN
	DECLARE
		n_IdEvnt				SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd				SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml					SCL_EVT_DATA.EVND_VAL%TYPE;
		n_idApli				SCL_APPLI.APLI_ID%Type;
		v_idTrameOrigine		VARCHAR2(19);
		v_Action				VARCHAR2(1);

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#
		If Deleting Then
			IF SUBSTR(:old.MOTH_COL_COM, 4, 1) = '1' THEN
				pckTrigPfiHmM.PRO_appliExt(:old.MOTH_DAT_MOD, :old.MOTH_COL_COM, n_idApli, v_idTrameOrigine);
			End If;
		Else
			pckTrigPfiHmM.PRO_appliExt(:new.MOTH_DAT_MOD, :new.MOTH_COL_COM, n_idApli, v_idTrameOrigine);
		End If;

		If Inserting Then
			v_Action := 'C';
		ElsIf Updating Then
			v_Action := 'M';
		Else
			v_Action := 'S';
		End If;

		Select /* T_EVTHM_MOTIF_CMS */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		If Deleting Then
			Insert /* T_EVTHM_MOTIF_CMS */
			Into SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB,
					EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ, EVNT_NUM_RSS,
					EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION,
					EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
	      			EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', :old.MOTH_ID_ETAB,
					NULL, NULL, NULL, NULL, NULL,
					'MOTIF_HOSPI',:old.MOTH_ID_MOTH, NULL, NULL, v_Action,
					n_IdApli,  v_idTrameOrigine, NULL,
	      			current_date, current_date, 0, :old.MOTH_UTI_CRE, :old.MOTH_UTI_MOD, :old.MOTH_PRO_CRE, :old.MOTH_PRO_MOD, :old.MOTH_COL_COM);
		Else
			Insert /* T_EVTHM_MOTIF_CMS */
			Into SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB,
							EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ, EVNT_NUM_RSS,
							EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION,
							EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
	      			EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', :new.MOTH_ID_ETAB,
					NULL, NULL,NULL, NULL, NULL,
					'MOTIF_HOSPI', :new.MOTH_ID_MOTH, NULL, NULL, v_Action,
					n_IdApli,  v_idTrameOrigine, NULL,
	      			current_date, current_date, 0, :new.MOTH_UTI_CRE, :new.MOTH_UTI_MOD, :new.MOTH_PRO_CRE, :new.MOTH_PRO_MOD, :new.MOTH_COL_COM);
		End If;

		-- Segment Patient
		If Deleting Then
			v_xml := null;
			v_xml := v_xml || '<MOTIF_HOSPI>';
			v_xml := v_xml || pckPfiHmM.val2xml('ID_MOTH', :old.MOTH_ID_MOTH);
			v_xml := v_xml || pckPfiHmM.val2xml('CODE', :old.MOTH_CODE);
			v_xml := v_xml || pckPfiHmM.val2xml('LIBELLE', :old.MOTH_LIBELLE);
			v_xml := v_xml || '</MOTIF_HOSPI>';
		Else
			v_xml := null;
			v_xml := v_xml || '<MOTIF_HOSPI>';
			v_xml := v_xml || pckPfiHmM.val2xml('ID_MOTH', :new.MOTH_ID_MOTH);
			v_xml := v_xml || pckPfiHmM.val2xml('CODE', :new.MOTH_CODE);
			v_xml := v_xml || pckPfiHmM.val2xml('LIBELLE', :new.MOTH_LIBELLE);
			v_xml := v_xml || '</MOTIF_HOSPI>';
		End If;

		Select /* T_EVTHM_MOTIF_CMS */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		INSERT /* T_EVTHM_MOTIF_CMS */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL,
						   EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
			VALUES (n_IdEvnd, n_IdEvnt, 1, 'MOTIF_HOSPI', 'I', v_Xml,
					current_date, current_date, 0, 3, 3, 2, 2, NULL);

	EXCEPTION
		WHEN OTHERS THEN
			Begin
				If Deleting Then
					pckTrigPfiHmM.PRO_traceScl(NULL, NULL, :old.MOTH_ID_ETAB, 'Evt Motif d hospitalisation', 'T_EVTHM_MOTIF_CMS', '0', SUBSTR(SQLERRM,1,2000));
				Else
					pckTrigPfiHmM.PRO_traceScl(NULL, NULL, :new.MOTH_ID_ETAB, 'Evt Motif d hospitalisation', 'T_EVTHM_MOTIF_CMS', '0', SUBSTR(SQLERRM,1,2000));
				End If;
			Exception
				When Others Then
					return;
			End;
	END;
END T_EVTHM_MOTIF_CMS;
