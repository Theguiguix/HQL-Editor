TRIGGER "PCAP_XROUGE".T_EVT_PRESCRIPTION_PERIODE_CM
AFTER INSERT or UPDATE
ON PRE_PRESCRIPTION_PERIODE
FOR EACH ROW
DECLARE
	Periode				PRE_PRESCRIPTION_PERIODE%RowType;
	Ligne				PRE_PRESCRIPTION_LIGNE%RowType;
	TypeLigne			BAS_CATALOGUE_GEN%RowType;
	Prescription		PRE_PRESCRIPTION_LISTE%RowType;
	Patient				IDE_PATIENT%RowType;
	Sejour				IDE_SEJOUR%RowType;
	v_action			SCL_EVT.EVNT_ACTION%Type;
	v_libAction			Varchar2(200);
	v_xml				SCL_EVT_DATA.EVND_VAL%Type;
	n_IdEvnt			SCL_EVT.EVNT_ID_EVNT%TYPE;
	n_IdEvnd			SCL_EVT_DATA.EVND_ID_EVND%TYPE;
	idLigDebSeq			PRE_PRESCRIPTION_LIGNE.PRPL_ID_PRPL%Type;

BEGIN
	-- #Version# 1.16 du 16/09/2016 #Version#

	Periode.PRPE_ID_PRPE := :new.PRPE_ID_PRPE;

	Periode.PRPE_ID_PRPL := :new.PRPE_ID_PRPL;
	Periode.PRPE_DAT_DEB_POS := :new.PRPE_DAT_DEB_POS;
	Periode.PRPE_DAT_FIN_POS := :new.PRPE_DAT_FIN_POS;
	Periode.PRPE_DAT_FIN_GENERATION := :new.PRPE_DAT_FIN_GENERATION;
	Periode.PRPE_TYPE := :new.PRPE_TYPE;
	Periode.PRPE_IND_AFF_CYCLE := :new.PRPE_IND_AFF_CYCLE;
	Periode.PRPE_NBR_DEB_HEURE := :new.PRPE_NBR_DEB_HEURE;
	Periode.PRPE_NBR_DEB_JOUR := :new.PRPE_NBR_DEB_JOUR;
	Periode.PRPE_NBR_DEB_MOIS := :new.PRPE_NBR_DEB_MOIS;
	Periode.PRPE_NBR_FIN_HEURE := :new.PRPE_NBR_FIN_HEURE;
	Periode.PRPE_NBR_FIN_JOUR := :new.PRPE_NBR_FIN_JOUR;
	Periode.PRPE_NBR_FIN_MOIS := :new.PRPE_NBR_FIN_MOIS;
	Periode.PRPE_IND_DEB_ANTERIEUR := :new.PRPE_IND_DEB_ANTERIEUR;
	Periode.PRPE_DUR_GAZ := :new.PRPE_DUR_GAZ;
	Periode.PRPE_UTI_CRE := :new.PRPE_UTI_CRE;
	Periode.PRPE_UTI_MOD := :new.PRPE_UTI_MOD;
	Periode.PRPE_PRO_CRE := :new.PRPE_PRO_CRE;
	Periode.PRPE_PRO_MOD := :new.PRPE_PRO_MOD;
	Periode.PRPE_COL_COM := :new.PRPE_COL_COM;

	if Periode.PRPE_ID_PRPL is not null then
		select /* T_EVT_PRESCRIPTION_PERIODE_CM */ *
		  into Ligne
		  From PRE_PRESCRIPTION_LIGNE
		 Where PRPL_ID_PRPL=Periode.PRPE_ID_PRPL;

		Prescription.PRME_ID_PRME := Ligne.PRPL_ID_PRME;
	end if;

	-- Oui si
	-- 2228 Fiche produit
	-- 1419 Médicament
	-- 1420 Perfusion
	-- 1421 Aérosol
	-- 1422 Préparations
	-- 2536 Perfusion en Y
	-- 2537 Perfusion en injection
	-- 6304 PSE
	-- 6305 PCA
	-- 6416 PCEA
	-- 6417 PCPA
	-- 7184 Patch
	-- 11372 Perfusion par gravite
	-- 11373 PSE par injectable
	-- 11374 PCA par injectable
	--
	-- Non si
	-- 5324 Dialyse
	-- 7081 Gaz
	If Ligne.PRPL_ID_CAGE_TYPE is null or Ligne.PRPL_ID_CAGE_TYPE not in (2228, 1419, 1420, 1421, 1422, 2536, 2537, 6304, 6305, 6416, 6417, 7184 ,11372, 11373, 11374) then
		Return;
	End If;

	If Prescription.PRME_ID_PRME is not null Then
		Select /* T_EVT_PRESCRIPTION_PERIODE_CM */ *
		  Into Prescription
		  From PRE_PRESCRIPTION_LISTE
		 Where PRME_ID_PRME=Prescription.PRME_ID_PRME;
	End If;

	-- On ne trace que les prescriptions médicamenteuses
	if nvl(Prescription.PRME_ID_CAGE_TYP_PRESCRIPT,0) <> 1424 then
		return;
	end if;

	-- On ne trace que les Traitement en hospitalisation et les Traitement en séance
	if nvl(Prescription.PRME_ID_CAGE_TYP_TRT,0) not in (1412, 1414) then
		return;
	end if;

	Patient.PATI_ID_PATI := Prescription.PRME_ID_PATI;
	Sejour.SEJO_ID_SEJO := Prescription.PRME_ID_SEJO;

	-- On ne trace pas une prescription non encore associée à un séjour
	if Sejour.SEJO_ID_SEJO is null then
		return;
	end if;


	-- En insertion : Oui Seulement si on est sur des cycles indéfinis, avec génération des périodes suivantes au fil de l'administration
	If inserting Then
		If Ligne.PRPL_NBR_CYCLES is null or Ligne.PRPL_NBR_CYCLES <> -1 Then
			Return;
		End If;

		-- Nb : pour les séquences, on ne va traquer que la validation de la prpl mère pour éviter la multiplication des évènements
		-- => seule la maman nous interesse
		if (Ligne.PRPL_ID_PRPL_DEB_SEQUENCE is not null) and (Ligne.PRPL_ID_PRPL_DEB_SEQUENCE <> Ligne.PRPL_ID_PRPL) then
			return;
		end if;

	else
		-- En mise à jour : Oui seulement si on a planifié un "arrêt" dans le futur (modification de la date de fin de prescription).
		If Ligne.PRPL_MED_VAL_STATUT = '2' And Ligne.PRPL_IND_ALE_PRI_COMPTE = '2' then
			If (:new.PRPE_DAT_FIN_AVT_ARRET = :old.PRPE_DAT_FIN_AVT_ARRET or (:new.PRPE_DAT_FIN_AVT_ARRET is null and :old.PRPE_DAT_FIN_AVT_ARRET is null)) And
			   (:new.PRPE_DAT_FIN_POS = :old.PRPE_DAT_FIN_POS or (:new.PRPE_DAT_FIN_POS is null and :old.PRPE_DAT_FIN_POS is null))
			Then
				return;
			End If;
		Else
			return;
		end if;

		-- Nb : pour les séquences, on ne simuler systématiquement une suppression de la prpl mère pour éviter la multiplication des évènements
		-- => seule la maman nous interesse
		if (Ligne.PRPL_ID_PRPL_DEB_SEQUENCE is not null) and (Ligne.PRPL_ID_PRPL_DEB_SEQUENCE <> Ligne.PRPL_ID_PRPL) then
			idLigDebSeq := Ligne.PRPL_ID_PRPL_DEB_SEQUENCE;
			select /* T_EVT_PRESCRIPTION_PERIODE_CM */ *
			  into Ligne
			  From PRE_PRESCRIPTION_LIGNE
			 Where PRPL_ID_PRPL=idLigDebSeq;
		end if;
	End If;

	-- Nb : pour les perfusions en Y, on ne va traquer que la validation ou suppression de la prpl mère pour éviter la multiplication des évènements
	-- => seule la maman nous interesse
	if (Ligne.PRPL_ID_PRPL_PERFUSION is not null) then
		return;
	end if;

	If Ligne.PRPL_ID_CAGE_TYPE is not null Then
		Select /* T_EVT_PRESCRIPTION_PERIODE_CM */ *
		  into TypeLigne
		  From BAS_CATALOGUE_GEN
		 Where CAGE_ID_CAGE=Ligne.PRPL_ID_CAGE_TYPE;
	End If;

	if Patient.PATI_ID_PATI is not null then
		select /* T_EVT_PRESCRIPTION_PERIODE_CM */ *
		  into Patient
		  From IDE_PATIENT
		 Where PATI_ID_PATI=Patient.PATI_ID_PATI;
	end if;

	if Sejour.SEJO_ID_SEJO is not null then
		select /* T_EVT_PRESCRIPTION_PERIODE_CM */ *
		  into Sejour
		  From IDE_SEJOUR
		 Where SEJO_ID_SEJO=Sejour.SEJO_ID_SEJO;
	end if;

	v_action := null;
	-- Les etats possibles, donnant lieu à émission seront :
	--
	-- VALMEDTMP	Validation médicale temporaire
	-- MODMEDTMP	Arrêt d'une prescription validée médicale temporaire dans le futur
	-- ARRMEDTMP	Arrêt d'une prescription validée médicale temporaire
	-- VALMED		Validation médicale
	-- MODVALMED	Arrêt d'une prescription validée médicalement dans le futur
	-- ARRVALMED	Arrêt d'une prescription validée médicalement
	-- VALPHATMP	Validation pharmacie temporaire
	-- MODPHATMP	Arrêt d'une prescription validée pharmacie temporaire dans le futur
	-- ARRPHATMP	Arrêt d'une prescription validée pharmacie temporaire
	-- VALPHAALT    Validation pharmacie avec Alerte
	-- MODPHAALT	Arrêt d'une prescription validée pharmacie avec Alerte dans le futur
	-- ARRPHAALT	Arrêt d'une prescription validée pharmacie avec Alerte
	-- VALPHA    	Validation pharmacie
	-- MODVALPHA	Arrêt d'une prescription validée pharmacie dans le futur
	-- ARRVALPHA	Arrêt d'une prescription validée pharmacie
	--
	--
	-- PRME_IND_PRE_VAL 	0=en attente de validation, 1=validé par le médecin, 2=Prescription annulée
	-- PRME_PHA_VAL_STATUT  0=non validée, 1=validée, 3=en attente de prise en compte, 4=Prescription éligible, 5=validée temporairement,
	--						6=validation pharma hétérogène,7=validation pharma hétérogène avec réserves
	--
	-- PRPL_MED_VAL_STATUT 	0=arrêté, 1=en attente de validation, 2=validé par le medecin
	-- PRPL_PHA_VAL_STATUT	0=non validé, 1=validé, 2=validé avec remarque, 3=validé en alerte, 6=validé et alerte prise en compte
	-- PRPL_ALE_PRI_COMPTE
	If Ligne.PRPL_MED_VAL_STATUT = '2' And Ligne.PRPL_IND_ALE_PRI_COMPTE = '2' Then
		If Ligne.PRPL_IND_TMP_MED_VAL = '1' Then
			If inserting Then
				-- Validation médicale temporaire
				v_action := 'VALMEDTMP';
				v_libAction := 'Validation médicale temporaire';
			Else
				-- Modification(Arrêt), Validation médicale temporaire
				v_action := 'MODMEDTMP';
				v_libAction := 'Modification (Validée médicale temporaire)';
			End if;
		Else
			If inserting Then
				-- Validation médicale
				v_action := 'VALMED';
				v_libAction := 'Validation médicale';
			Else
				-- Modification (Arrêt), Validation médicale
				v_action := 'MODVALMED';
				v_libAction := 'Modification (Validée  médicale)';
			End if;
		End If;
		If Ligne.PRPL_PHA_VAL_STATUT = '3' Then
			If inserting Then
				-- Validation pharmacie avec Alerte
				v_action := 'VALPHAALT';
				v_libAction := 'Validation pharmacie avec Alerte';
			Else
				-- Modification (Arrêt), Validation pharmacie avec Alerte
				v_action := 'MODPHAALT';
				v_libAction := 'Modification (Validée pharmacie avec Alerte)';
			End If;
		End If;
		If Ligne.PRPL_PHA_VAL_STATUT = '1' Then
			If (Ligne.PRPL_IND_TMP_PHA_VAL = '1') then
				If inserting Then
					-- Validation pharmacie temporaire
					v_action := 'VALPHATMP';
					v_libAction := 'Validation pharmacie temporaire';
				Else
					-- Modification (Arrêt), Validation pharmacie temporaire
					v_action := 'MODPHATMP';
					v_libAction := 'Modification (Validée pharmacie temporaire)';
				End If;
			Else
				If inserting Then
					-- Validation pharmacie
					v_action := 'VALPHA';
					v_libAction := 'Validation pharmacie';
				Else
					-- Modification (Arrêt), Validation pharmacie
					v_action := 'MODVALPHA';
					v_libAction := 'Modification (Validée pharmacie)';
				End If;
			End If;
		End If;
	End If;
	If Ligne.PRPL_MED_VAL_STATUT = '0' Then
		-- Création d'une ligne Arrêtée : aucun sens : on dégage
		return;
	End If;

	-- Si pas d'action remarquable, on ne poste pas d'évènement et on quitte
	If v_action is null Then
		return;
	End If;

	Select /* T_EVT_PRESCRIPTION_PERIODE_CM */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

	If inserting Then
		-- Cycles infinis : on poste sur la période elle même
		Insert /* T_EVT_PRESCRIPTION_PERIODE_CM */
		Into scl_Evt (evnt_id_evnt, evnt_date, evnt_version, evnt_id_etab
						   , evnt_code_evt, evnt_id_obj, evnt_code_pere, evnt_id_pere
						   , evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej, evnt_num_rss
						   , evnt_action, evnt_id_apli, evnt_id_evt_externe
						   , EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM, EVNT_EVTHL7)
					Values (n_IdEvnt, current_date, '0', Prescription.PRME_ID_ETAB
						  , 'PRESCMEDICPERIODE', Periode.PRPE_ID_PRPE, null, Ligne.PRPL_ID_PRPL
						  , Patient.PATI_ID_PATI, Patient.PATI_NIP, Sejour.SEJO_ID_SEJO, Sejour.SEJO_NUM_SEJ_COMM, null
						  , v_action, null, null
						  , current_date, current_date, 0, Periode.PRPE_UTI_CRE, Periode.PRPE_UTI_MOD, Periode.PRPE_PRO_CRE, Periode.PRPE_PRO_MOD, Periode.PRPE_COL_COM, null);

	Else
		-- Autres cas (Arrêt dans le futurà : on poste sur la ligne de prescription
		Insert /* T_EVT_PRESCRIPTION_PERIODE_CM */
		Into scl_Evt (evnt_id_evnt, evnt_date, evnt_version, evnt_id_etab
						   , evnt_code_evt, evnt_id_obj, evnt_code_pere, evnt_id_pere
						   , evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej, evnt_num_rss
						   , evnt_action, evnt_id_apli, evnt_id_evt_externe
						   , EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM, EVNT_EVTHL7)
					Values (n_IdEvnt, current_date, '0', Prescription.PRME_ID_ETAB
						  , 'PRESCRIPTIONMEDIC', Ligne.PRPL_ID_PRPL, null, Prescription.PRME_ID_PRME
						  , Patient.PATI_ID_PATI, Patient.PATI_NIP, Sejour.SEJO_ID_SEJO, Sejour.SEJO_NUM_SEJ_COMM, null
						  , v_action, null, null
						  , current_date, current_date, 0, Periode.PRPE_UTI_CRE, Periode.PRPE_UTI_MOD, Periode.PRPE_PRO_CRE, Periode.PRPE_PRO_MOD, Periode.PRPE_COL_COM, null);
	End If;

	-- Segment Action
	v_xml := null;
	v_xml := v_xml || '<ACTION>';
	v_xml := v_xml || pckPfiHmM.val2xml('CODE', v_action);
	v_xml := v_xml || pckPfiHmM.val2xml('LIBELLE', v_libAction);
	v_xml := v_xml || '</ACTION>';

	Select /* T_EVT_PRESCRIPTION_PERIODE_CM */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

	Insert /* T_EVT_PRESCRIPTION_PERIODE_CM */
	Into scl_evt_data (evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val
							, EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
			Values (n_IdEvnd, n_IdEvnt, 0, 'ACTION', 'I', v_xml
				  , current_date, current_date, 0, 3, 3, 2, 2, NULL);


	-- Segment Patient
