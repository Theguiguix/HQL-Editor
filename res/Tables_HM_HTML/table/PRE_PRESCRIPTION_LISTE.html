		  From pre_prescription_ligne
		 Where PRPL_ID_PRME = idPrme
		   And PRPL_ID_PRPL_PERFUSION is null
		   And coalesce(PRPL_ID_PRPL_DEB_SEQUENCE, PRPL_ID_PRPL) = PRPL_ID_PRPL
		   And PRPL_ID_CAGE_TYPE in (2228, 1419, 1420, 1421, 1422, 2536, 2537, 6304, 6305, 6416, 6417, 7184, 11372, 11373, 11374);

BEGIN
	-- #Version# 1.16 du 16/09/2016 #Version#

	-- On ne trace que les Traitement en hospitalisation et les Traitement en séance
	if nvl(:new.PRME_ID_CAGE_TYP_TRT,0) not in (1412, 1414) then
		return;
	end if;

	-- On cherche à tracer la prescription à compter de son association au séjour du patient
	If (:old.PRME_ID_SEJO is not null) Or (:new.PRME_ID_SEJO is null) Then
		return;
	End If;

	Patient.PATI_ID_PATI := :new.PRME_ID_PATI;
	Sejour.SEJO_ID_SEJO := :new.PRME_ID_SEJO;

	if Patient.PATI_ID_PATI is not null then
		select /* T_EVT_PRESCRIPTION_LISTE_M */ *
		  into Patient
		  From IDE_PATIENT
		 Where PATI_ID_PATI=Patient.PATI_ID_PATI;
	end if;

	if Sejour.SEJO_ID_SEJO is not null then
		select /* T_EVT_PRESCRIPTION_LISTE_M */ *
		  into Sejour
		  From IDE_SEJOUR
		 Where SEJO_ID_SEJO=Sejour.SEJO_ID_SEJO;
	end if;

	Open CLignes(:new.PRME_ID_PRME);
	Loop
		Fetch CLignes Into Ligne;

		If Ligne.PRPL_ID_CAGE_TYPE is not null Then

					v_libAction := 'Arrêt d''une prescription validée pharmacie temporaire';
				Else
					v_action := 'ARRVALPHA';
					v_libAction := 'Arrêt d''une prescription validée pharmacie';
				End If;
			ElsIf Ligne.PRPL_PHA_VAL_STATUT = '3' Then
				v_action := 'ARRPHAALT';
				v_libAction := 'Arrêt d''une prescription validée pharmacie avec Alerte';
			else
				If Ligne.PRPL_IND_TMP_MED_VAL = '1' Then
					v_action := 'ARRMEDTMP';
					v_libAction := 'Arrêt d''une prescription validée médicale temporaire';
				Else

					v_libAction := 'Arrêt d''une prescription validée médicalement';
				End If;
			End If;
		End If;
