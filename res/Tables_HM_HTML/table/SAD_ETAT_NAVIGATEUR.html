TRIGGER "PCAP_XROUGE".T_EVT_SAD_ETAT_NAVIGATEUR_CM
AFTER INSERT or UPDATE
ON SAD_ETAT_NAVIGATEUR
FOR EACH ROW
DECLARE
	-- Seuls les actes Valides sont pris en compte
	Cursor CActivitesCcam(idRec Number) is
		Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ *
		  From SAD_ACTIVITE_SAISIES
		 Where ACSA_ID_RECU=idRec
		   And ACSA_IND_VALIDE = '1'
		   And ACSA_ID_ASAI IS NOT NULL;

	Cursor CActivitesNgap(idRec Number) is
		Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ *
		  From SAD_ACTIVITE_SAISIES
		 Where ACSA_ID_RECU=idRec
		   And ACSA_IND_VALIDE = '1'
		   And ACSA_ID_ASAI IS NULL
		   And ACSA_ID_PRHO IS NOT NULL And ACSA_ID_FAVO IS NOT NULL;

	RActivite					    SAD_ACTIVITE_SAISIES%RowType;
	RActe							SAD_ACTES_SAISIS%RowType;
	RUfExe							BAS_UF%RowType;
	RUfResp							BAS_UF%RowType;
	RMedExe							IDE_MEDECIN%RowType;
	RRecueil						SAD_RECUEIL%RowType;
	RPatient						IDE_PATIENT%RowType;
	RSejour							IDE_SEJOUR%RowType;
	RDossier                        SAD_DOSSIER_TECHNIQUE%RowType;
	RUfEntree						BAS_UF%RowType;
	v_xml							SCL_EVT_DATA.EVND_VAL%Type;
	n_IdEvnt						SCL_EVT.EVNT_ID_EVNT%TYPE;
	n_IdEvnd						SCL_EVT_DATA.EVND_ID_EVND%TYPE;
	v_codeAction					varchar2(1);
	n_idApli						SCL_APPLI.APLI_ID%Type;
	v_idTrameOrigine				VARCHAR2(19);
	v_GoEvt							Varchar2(1);

BEGIN
	-- #Version# 1.16 du 01/03/2016 #Version#
	if :new.ETNA_ID_RECU is null then
		return;
	end if;

	if ((:old.ETNA_IND_STT_ACC is null or :old.ETNA_IND_STT_ACC = '0') And :new.ETNA_IND_STT_ACC = '1') Or
	   (:old.ETNA_DAT_ENT_PAT_SALLE is null And :new.ETNA_DAT_ENT_PAT_SALLE is not null) Or
	   (:old.ETNA_DAT_SOR_PAT_SALLE is null And :new.ETNA_DAT_SOR_PAT_SALLE is not null)
	then
		-- Au moins un indicateur remarquable a été activé : on poste des évenements Actes
		v_GoEvt := '1';

	else
		-- Aucun indicateur remarquable n'a bougé
		v_GoEvt := '0';
	end if;

	If (v_GoEvt = '0') Then
		return;
	End If;

	Open CActivitesCcam(:new.ETNA_ID_RECU);
	Loop
		FETCH CActivitesCcam INTO RActivite;
		EXIT WHEN CActivitesCcam%NOTFOUND OR CActivitesCcam%NOTFOUND IS NULL;

		v_GoEvt := '1';
		if RActivite.ACSA_IND_BLO= '1' then
			v_GoEvt := '0';
		end if;

		if v_GoEvt = '1' Then
			v_codeAction := 'M';

			If (((RActivite.ACSA_MON_TOT_NON_REMB is not null) Or (RActivite.ACSA_BAS_REMB is not null))
					And RActivite.ACSA_ID_MEDE is not null) then

				-- Lecture du l'acte
				Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ *
					Into RActe
					From SAD_ACTES_SAISIS
				Where ASAI_ID_ASAI = RActivite.ACSA_ID_ASAI;

				-- Lecture du recueil
				If RActe.ASAI_ID_RECU is not null Then
					Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ *
						Into RRecueil
						From SAD_RECUEIL
					Where RECU_ID_RECU = RActe.ASAI_ID_RECU;
				End if;

				-- Les Hypothèses initiales ne sont pas émises
				If nvl(RRecueil.RECU_IND_HYP_INI,'0') = '0' Then

					-- Lecture de l'UF exécutante
					If RActe.ASAI_ID_UNFO is not null Then
						Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ *
							Into RUfExe
							From BAS_UF
						Where UNFO_ID_UNFO = RActe.ASAI_ID_UNFO;
					End if;

					-- Lecture de l'UF responsable
					If RActe.ASAI_ID_UNFO_RES is not null Then
						Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ *
							Into RUfResp
							From BAS_UF
						Where UNFO_ID_UNFO = RActe.ASAI_ID_UNFO_RES;
					End if;

					-- Lecture du médecin exécutant
					If RActivite.ACSA_ID_MEDE is not null Then
						Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ *
							Into RMedExe
							From IDE_MEDECIN
						 Where MEDE_ID_MEDE = RActivite.ACSA_ID_MEDE;
					End if;

					-- Lecture du séjour
					Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ *
					  Into RSejour
					  From IDE_SEJOUR
					Where SEJO_ID_SEJO = RActe.ASAI_ID_SEJO;

					-- Lecture de l'UF d'entrée
					If RSejour.SEJO_ID_UNFO is not null Then
						Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ *
						  Into RUfEntree
						  From BAS_UF
						 Where UNFO_ID_UNFO = RSejour.SEJO_ID_UNFO;
					End if;

					-- Lecture du patient
					Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ *
					  Into RPatient
					  From IDE_PATIENT
					 Where PATI_ID_PATI = RSejour.SEJO_ID_PATI;

					-- Pour les Actes Issus d'un dossier technique, Lecture de l'état validation
			        Begin
						Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ *
 						  Into RDossier
						  From SAD_DOSSIER_TECHNIQUE
						 Where DOSS_ID_DOSS = RRecueil.RECU_ID_DOSS
						   And Rownum = 1;

					Exception
						When No_data_found Then
							RDossier.DOSS_IND_DOSSIER_VALIDE := null;
							RDossier.DOSS_ID_DOSS:=null;
					End;

					Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

					Insert /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */
					Into scl_Evt (evnt_id_evnt, evnt_date, evnt_version, evnt_id_etab,
										 evnt_code_evt, evnt_id_obj, evnt_code_pere, evnt_id_pere,
										 evnt_id_patient, evnt_nip, evnt_id_sejour, evnt_num_sej, evnt_num_rss,
										 evnt_action, EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE,
										 EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
					Values (n_IdEvnt, current_date, '0', RSejour.SEJO_ID_ETAB_CON,
							'ACTE_CCAM_NAV', RActivite.ACSA_ID_ACSA, 'RECUEIL', RActe.ASAI_ID_RECU,
							RPatient.PATI_ID_PATI, RPAtient.PATI_NIP, RActe.ASAI_ID_SEJO, RSejour.SEJO_NUM_SEJ_COMM, null,
							v_codeAction, n_IdApli, v_idTrameOrigine,
							current_date, current_date, 0, :new.ETNA_UTI_CRE, :new.ETNA_UTI_MOD, :new.ETNA_PRO_CRE, :new.ETNA_PRO_MOD, :new.ETNA_COL_COM);

					-- Segment Patient
					v_xml := null;
					v_xml := v_xml || '<PATIENT>';
					v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', RPatient.PATI_ID_PATI);
					v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', RPatient.PATI_NOM_USU);

					v_xml := v_xml || pckPfiHmM.val2xml('SEXE', RPatient.PATI_SEXE);
					v_xml := v_xml || pckPfiHmM.val2xml('NIP', RPatient.PATI_NIP);
					v_xml := v_xml || '</PATIENT>';

					Select /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

					Insert /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */
					Into scl_evt_data (evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val
											, EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
					Values (n_IdEvnd, n_IdEvnt, 1, 'PATIENT', 'I', v_xml
						  , current_date, current_date, 0, 3, 3, 2, 2, NULL);

					-- Segment Sejour
					v_xml := null;
					v_xml := v_xml || '<SEJOUR>';
					v_xml := v_xml || pckPfiHmM.val2xml('ID_SEJO', RSejour.SEJO_ID_SEJO);
					v_xml := v_xml || pckPfiHmM.val2xml('ID_UNFO', RSejour.SEJO_ID_UNFO);
					v_xml := v_xml || pckPfiHmM.val2xml('UNFO_CODE', RUfEntree.UNFO_CODE);
					v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', RSejour.SEJO_DAT_DEB);
					v_xml := v_xml || pckPfiHmM.val2xml('NUM_SEJ', RSejour.SEJO_NUM_SEJ_COMM);

					Insert /* T_EVT_SAD_ETAT_NAVIGATEUR_CM */
					Into scl_evt_data (evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val,

					Values (n_IdEvnd+1, n_IdEvnt, 2, 'SEJOUR', 'I', v_xml,
							current_date, current_date, 0, 3, 3, 2, 2, NULL);

					v_xml := null;
					v_xml := v_xml || '<RECUEIL>';
					v_xml := v_xml || pckPfiHmM.val2xml('ID_RECU', RRecueil.RECU_ID_RECU);

					v_xml := v_xml || pckPfiHmM.val2xml('ACCUEILLI_OLD', :old.ETNA_IND_STT_ACC);
					v_xml := v_xml || pckPfiHmM.val2xml('DAT_ENT_SALLE', :new.ETNA_DAT_ENT_PAT_SALLE);
					v_xml := v_xml || pckPfiHmM.val2xml('DAT_ENT_SALLE_OLD', :old.ETNA_DAT_ENT_PAT_SALLE);
					v_xml := v_xml || pckPfiHmM.val2xml('DAT_SOR_SALLE', :new.ETNA_DAT_SOR_PAT_SALLE);
					v_xml := v_xml || pckPfiHmM.val2xml('DAT_SOR_SALLE_OLD', :old.ETNA_DAT_SOR_PAT_SALLE);
					v_xml := v_xml || pckPfiHmM.val2xml('IDDOSSIER',RDossier.DOSS_ID_DOSS);
                   	v_xml := v_xml || pckPfiHmM.val2xml('VALIDE_TECHNIQUE',RDossier.DOSS_IND_DOSSIER_VALIDE);
                    v_xml := v_xml || pckPfiHmM.val2xml('VALIDE_TECHNIQUE_OLD',RDossier.DOSS_IND_DOSSIER_VALIDE);
					v_xml := v_xml || '</RECUEIL>';

					Into scl_evt_data (evnd_id_evnd, evnd_id_evnt, evnd_index, evnd_type, evnd_action, evnd_val

					Values (n_IdEvnd+2, n_IdEvnt, 3, 'RECUEIL', 'I', v_xml
						  , current_date, current_date, 0, 3, 3, 2, 2, NULL);

					-- Segment Acte CCAM
					v_xml := null;
					v_xml := v_xml || '<ACTE_CCAM>';
					v_xml := v_xml || pckPfiHmM.val2xml('ID_ACSA', RActivite.ACSA_ID_ACSA);
