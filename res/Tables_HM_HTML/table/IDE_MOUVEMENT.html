			ElsIf v_TypMvt = 'F' Then

			begin
				Select /* T_EVTHM_MVT_M */ count(evnt_id_evnt)
					Into n_nb
					From scl_evt
				 Where evnt_id_obj = :new.MOUV_ID_MOUV
				   And evnt_code_evt in ('MOUVEMENT', 'MOUVEMENTP', 'MOUVEMENTG');

				When others Then
					return;
			end;

			If v_TypMvt = 'S' Then
				v_evtHL7 := 'A13';
			ElsIf v_TypMvt = 'C' Then
				v_evtHL7 := 'A11';
			ElsIf v_TypMvt = 'P' Then
				v_evtHL7 := 'A13';
			ElsIf v_TypMvt = 'A' Then
				v_evtHL7 := 'A52';
			ElsIf v_TypMvt = 'R' Then

					v_evtHL7 := null;

					-- Est on sur une Bascule Externe/Hospit? Si oui, si A06 ou A07, il a été géré par le trigger SEJOUR (voire le Z99 si statut A ou P), pas de raison de balancer un Z99 sur le mouvement d'entrée.
