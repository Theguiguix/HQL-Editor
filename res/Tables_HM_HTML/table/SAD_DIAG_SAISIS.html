		-- Pseudo Update de chargement de l'appli externe. Suppression à suivre. Pas d'évènement Modif à poster.
		return;
	End If;

    -- noPerroquet
	If  (:new.DSAI_ID_ADIA = :old.DSAI_ID_ADIA or (:new.DSAI_ID_ADIA is null and :old.DSAI_ID_ADIA is null)) and

	Then
		return;
	End If;

	-- Si le diag  a déja été émis : modif sinon création
	Select /* T_EVT_SAD_DIAG_SAISIS_CM */ count(EVNT_ID_EVNT)
		Into n_nb
		From SCL_EVT
	Where EVNT_ID_OBJ = :new.DSAI_ID_DSAI
	  And EVNT_CODE_EVT = 'DIAGCIM10';

	if (n_nb > 0) then
		v_codeAction := 'M';
	else
		v_codeAction := 'C';
	end if;

	-- Lecture du recueil
	If :new.DSAI_ID_RECU is not null Then
		Select /* T_EVT_SAD_DIAG_SAISIS_CM */ *
			Into RRecueil
			From SAD_RECUEIL
		 Where RECU_ID_RECU = :new.DSAI_ID_RECU;

		If RRecueil.RECU_ID_UNFO is not null Then
			Select /* T_EVT_SAD_DIAG_SAISIS_CM */ *
				Into RUfExe
				From BAS_UF
			 Where UNFO_ID_UNFO = RRecueil.RECU_ID_UNFO;
		End if;

		-- Lecture de l'UF resp
		If RRecueil.RECU_ID_UNFO_RESP is not null Then
			Select /* T_EVT_SAD_DIAG_SAISIS_CM */ *
				Into RUfResp
				From BAS_UF
			 Where UNFO_ID_UNFO = RRecueil.RECU_ID_UNFO_RESP;
		End if;
	End if;

	-- Les Hypothèses initiales ne sont émises
	If nvl(RRecueil.RECU_IND_HYP_INI,'0') = '0' Then
		-- Lecture du séjour
		Select /* T_EVT_SAD_DIAG_SAISIS_CM */ *
		Into RSejour
		From IDE_SEJOUR
		Where SEJO_ID_SEJO = RRecueil.RECU_ID_SEJO;

		-- Lecture de l'UF d'entrée
		If RSejour.SEJO_ID_UNFO is not null Then
			Select /* T_EVT_SAD_DIAG_SAISIS_CM */ *
				Into RUfEntree
				From BAS_UF
			 Where UNFO_ID_UNFO = RSejour.SEJO_ID_UNFO;
		End if;

		-- Lecture du patient
		Select /* T_EVT_SAD_DIAG_SAISIS_CM */ *
		Into RPatient
		From IDE_PATIENT
		Where PATI_ID_PATI = RSejour.SEJO_ID_PATI;

		-- Gestion de l'écho
		pckTrigPfiHmM.PRO_appliExt(:new.DSAI_DAT_MOD, :new.DSAI_COL_COM, n_idApli, v_idTrameOrigine);
