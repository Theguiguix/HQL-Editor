TRIGGER "PCAP_XROUGE".T_EVTHM_ETAP_SAISIE_CMS
AFTER INSERT or UPDATE or DELETE
ON SAD_ETAP_SAISIES
FOR EACH ROW
BEGIN
	DECLARE
		n_CodRet				NUMBER;
		n_IdEvnt				SCL_EVT.EVNT_ID_EVNT%TYPE;
		n_IdEvnd				SCL_EVT_DATA.EVND_ID_EVND%TYPE;
		v_Xml					SCL_EVT_DATA.EVND_VAL%TYPE;
		RPatient				IDE_PATIENT%ROWTYPE;
		RSejour					IDE_SEJOUR%ROWTYPE;
		RRecueil				SAD_RECUEIL%ROWTYPE;
		RPaam					BAS_PARAM_ACTE_MEDICAL%ROWTYPE;
		v_TypSejBase 			BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		v_TypSej				BAS_CATALOGUE_PERS.CAPE_CODE%TYPE;
		v_UfCode				BAS_UF.UNFO_CODE%TYPE;
		v_UfTypSej				BAS_CATALOGUE_GEN.CAGE_CODE%TYPE;
		n_idApli				SCL_APPLI.APLI_ID%Type;
		v_idTrameOrigine		VARCHAR2(19);
		v_Action				VARCHAR2(1);
		v_srvNip				scl_appli.APLI_SRVIPP%type;
		v_sejoStatut			bas_parametre.para_valeur%Type;
		v_patiStatut			bas_parametre.para_valeur%Type;
		v_evtHL7				SCL_EVT.EVNT_EVTHL7%Type;

	BEGIN
		-- #Version# 1.16 du 01/03/2016 #Version#

		If Updating Then
			-- noPerroquet
			If (:new.ETSA_CODE = :old.ETSA_CODE or (:new.ETSA_CODE is null and :old.ETSA_CODE is null)) and
			   (:new.ETSA_DAT_RDV = :old.ETSA_DAT_RDV or (:new.ETSA_DAT_RDV is null and :old.ETSA_DAT_RDV is null)) and
			   (:new.ETSA_DUR_PREVUE = :old.ETSA_DUR_PREVUE or (:new.ETSA_DUR_PREVUE is null and :old.ETSA_DUR_PREVUE is null)) and
			   (:new.ETSA_DAT_SUP = :old.ETSA_DAT_SUP or (:new.ETSA_DAT_SUP is null and :old.ETSA_DAT_SUP is null))
			Then
				return;
			End If;
		End If;

		If :old.ETSA_DAT_SUP is not null And :new.ETSA_DAT_SUP is not null Then
			return;
		end If;

		-- On chope le recueil lié à l'étape
		Begin
			If Deleting Then
				Select /* T_EVTHM_ETAP_SAISIE_CMS */ *
				  Into RRecueil
				  From SAD_RECUEIL
				 Where recu_id_recu = :old.etsa_id_recu;

			Else
				Select /* T_EVTHM_ETAP_SAISIE_CMS */ *
				  Into RRecueil
				  From SAD_RECUEIL
				 Where recu_id_recu = :new.etsa_id_recu;
			End if;

		Exception
			When Others then
				return;
		End;


		Begin
			Select /* T_EVTHM_ETAP_SAISIE_CMS */ *
			  Into RPaam
			  From BAS_PARAM_ACTE_MEDICAL
			 Where paam_id_paam = RRecueil.recu_id_paam;

		Exception
			When Others then
				return;
		End;


		If RRecueil.recu_id_pati is null or RRecueil.recu_id_sejo is null Then
			-- On ne poste pas d'évènement si pas de patient ou de séjour
			return;
		End If;


		If RPaam.paam_id_tyin = 910 Then
			-- Il s'agit bien d'une intervention
			-- Lecture du patient et du séjour
			Begin
				Select /* T_EVTHM_ETAP_SAISIE_CMS */ *
				  Into RPatient
				  From IDE_PATIENT
				 Where pati_id_pati = RRecueil.recu_id_pati;

			Exception
				When Others then
					return;
			End;

			Begin
				Select /* T_EVTHM_ETAP_SAISIE_CMS */ *
				  Into RSejour
				  From IDE_SEJOUR
				 Where sejo_id_sejo = RRecueil.recu_id_sejo;

			Exception
				When Others then
					return;
			End;

		Else
			-- L'étape ne porte pas sur une intervention. Pas d'évènement à poster.
			return;
		End if;


		If RSejour.sejo_dat_sup is not null Then
			return;
		End If;

		-- Séjour incomplet = séjour mal formé au sens HM = séjour inexploitable = Séjour en cours de saisie
		If (RSejour.SEJO_IND_INCOMPLET = '1') OR (RSejour.SEJO_IND_INCOMPLET IS NULL) Then
			return;
		End If;

		If RSejour.SEJO_NUM_SEJ_COMM is null or RPatient.PATI_NIP is null Then
			-- HM est il serveur d'IPP ?
			v_patiStatut := pckPfiHmM.FUN_litValParam('NUM_PATI_STATUT', RSejour.SEJO_ID_ETAB_CON, current_date);

			-- HM est il serveur de NDA ?
			v_sejoStatut := pckPfiHmM.FUN_litValParam('NUM_SEJO_STATUT', RSejour.SEJO_ID_ETAB_CON, current_date);

			If v_sejoStatut = 'J' Then
				v_srvNip := '0';
			Else
				Begin
					Select /* T_EVTHM_ETAP_SAISIE_CMS */ Max(APLI_SRVIPP)
					  Into v_srvNip
					  From scl_appli
					 Where APLI_TYPE in ('HM','OM')
					   And APLI_ID_ETAB = RSejour.SEJO_ID_ETAB_CON;

				Exception
					When others Then
						v_srvNip := '0';
				End;
			End If;

			If v_srvNip = '1' Then
				return;
			End If;
		End If;

		n_CodRet := pckPfiHmM.FUN_LireUF(RSejour.SEJO_ID_UNFO, v_UfCode, v_UfTypSej);

		-- Lecture type du séjour
		If RSejour.SEJO_ID_CAPE_VEN is not null Then
			SELECT /* T_EVTHM_ETAP_SAISIE_CMS */ CAGE_CODE, CAPE_CODE
			  INTO v_TypSejBase, v_TypSej
			  FROM BAS_CATALOGUE_SPE, BAS_CATALOGUE_GEN, BAS_CATALOGUE_PERS
			 WHERE CAPE_ID_CAPE = RSejour.SEJO_ID_CAPE_VEN
			   AND CASP_ID_CAPE = CAPE_ID_CAPE
			   AND CASP_ID_CAGE = CAGE_ID_CAGE
			   AND CAGE_ID_TYIN = 4;
		End If;


		If inserting Then
			v_Action := 'C';
		ElsIf updating Then
			If :old.ETSA_DAT_SUP is null And :new.ETSA_DAT_SUP is not null Then
				v_Action := 'S';
			Else
				v_Action := 'M';
			End If;
		Else
			v_Action := 'S';
		End If;


		If deleting Then
			IF SUBSTR(:old.ETSA_COL_COM, 4, 1) = '1' THEN
				pckTrigPfiHmM.PRO_appliExt(:old.ETSA_DAT_MOD, :old.ETSA_COL_COM, n_idApli, v_idTrameOrigine);
			End If;
		Else
			If :new.ETSA_UTI_MOD = 3 Then
				pckTrigPfiHmM.PRO_appliExt(:new.ETSA_DAT_MOD, :new.ETSA_COL_COM, n_idApli, v_idTrameOrigine);
			End If;
		End If;

		Select /* T_EVTHM_ETAP_SAISIE_CMS */ SEQ_SCL_EVT.nextval Into n_IdEvnt From Dual;

		If Deleting Then
			Insert /* T_EVTHM_ETAP_SAISIE_CMS */
			Into SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB,
							EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ, EVNT_NUM_RSS,
							EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION,
							EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
							EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', RSejour.SEJO_ID_ETAB_CON,
							RSejour.SEJO_ID_PATI, RPatient.PATI_NIP, RSejour.SEJO_ID_SEJO, RSejour.SEJO_NUM_SEJ_COMM, NULL,
							'AGD-INTERV-PAT', :old.ETSA_ID_ETSA, NULL, NULL, v_Action,
							n_IdApli,  v_idTrameOrigine, v_evtHl7,
							current_date, current_date, 0, :old.ETSA_UTI_CRE, :old.ETSA_UTI_MOD, :old.ETSA_PRO_CRE, :old.ETSA_PRO_MOD, :old.ETSA_COL_COM);

		else
			Insert /* T_EVTHM_ETAP_SAISIE_CMS */
			Into SCL_EVT (EVNT_ID_EVNT, EVNT_DATE, EVNT_VERSION, EVNT_ID_ETAB,
							EVNT_ID_PATIENT, EVNT_NIP, EVNT_ID_SEJOUR, EVNT_NUM_SEJ, EVNT_NUM_RSS,
							EVNT_CODE_EVT, EVNT_ID_OBJ, EVNT_CODE_PERE, EVNT_ID_PERE, EVNT_ACTION,
							EVNT_ID_APLI, EVNT_ID_EVT_EXTERNE, EVNT_EVTHL7,
							EVNT_DAT_CRE, EVNT_DAT_MOD, EVNT_NMAJ, EVNT_UTI_CRE, EVNT_UTI_MOD, EVNT_PRO_CRE, EVNT_PRO_MOD, EVNT_COL_COM)
			VALUES (n_IdEvnt, current_date, '0', RSejour.SEJO_ID_ETAB_CON,
							RSejour.SEJO_ID_PATI, RPatient.PATI_NIP, RSejour.SEJO_ID_SEJO, RSejour.SEJO_NUM_SEJ_COMM, NULL,
							'AGD-INTERV-PAT', :new.ETSA_ID_ETSA, NULL, NULL, v_Action,
							n_IdApli,  v_idTrameOrigine, v_evtHl7,
							current_date, current_date, 0, :new.ETSA_UTI_CRE, :new.ETSA_UTI_MOD, :new.ETSA_PRO_CRE, :new.ETSA_PRO_MOD, :new.ETSA_COL_COM);
		end if;

		-- Segment Patient
		v_xml := null;
		v_xml := v_xml || '<PATIENT>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_PATI', RPatient.PATI_ID_PATI);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_USU', RPatient.PATI_NOM_USU);
		v_xml := v_xml || pckPfiHmM.val2xml('NOM_NAI', RPatient.PATI_NOM_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('PRENOM', RPatient.PATI_PRENOM);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_NAI', RPatient.PATI_DAT_NAI);
		v_xml := v_xml || pckPfiHmM.val2xml('SEXE', RPatient.PATI_SEXE);
		v_xml := v_xml || pckPfiHmM.val2xml('NIP', RPatient.PATI_NIP);
		v_xml := v_xml || '</PATIENT>';

		Select /* T_EVTHM_ETAP_SAISIE_CMS */ SEQ_SCL_EVT_DATA.nextval Into n_IdEvnd From Dual;

		INSERT /* T_EVTHM_ETAP_SAISIE_CMS */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL
												    , EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
			VALUES (n_IdEvnd, n_IdEvnt, 1, 'PATIENT', 'I', v_Xml
				    , current_date, current_date, 0, 3, 3, 2, 2, NULL);

		v_xml := null;
		v_xml := v_xml || '<SEJOUR>';
		v_xml := v_xml || pckPfiHmM.val2xml('ID_SEJO', RSejour.SEJO_ID_SEJO);
		v_xml := v_xml || pckPfiHmM.val2xml('ID_UNFO', RSejour.SEJO_ID_UNFO);
		v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJBASE', v_TypSejBase);
		v_xml := v_xml || pckPfiHmM.val2xml('TYPSEJ', v_TypSej);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_CODE', v_UfCode);
		v_xml := v_xml || pckPfiHmM.val2xml('UNFO_TYPSEJ', v_UfTypSej);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_DEB', RSejour.SEJO_DAT_DEB);
		v_xml := v_xml || pckPfiHmM.val2xml('DAT_FIN', RSejour.SEJO_DAT_FIN);
		v_xml := v_xml || pckPfiHmM.val2xml('NUM_SEJ', RSejour.SEJO_NUM_SEJ_COMM);

		v_xml := v_xml || '</SEJOUR>';

		INSERT /* T_EVTHM_ETAP_SAISIE_CMS */
		INTO SCL_EVT_DATA (evnd_id_evnd, EVND_ID_EVNT, EVND_INDEX, EVND_TYPE, EVND_ACTION, EVND_VAL
												    , EVND_DAT_CRE, EVND_DAT_MOD, EVND_NMAJ, EVND_UTI_CRE, EVND_UTI_MOD, EVND_PRO_CRE, EVND_PRO_MOD, EVND_COL_COM)
			VALUES (n_IdEvnd+1, n_IdEvnt, 2, 'SEJOUR', 'I', v_Xml
				    , current_date, current_date, 0, 3, 3, 2, 2, NULL);

		v_xml := null;
		v_xml := v_xml || '<INTERV>';
		If deleting Then
			v_xml := v_xml || pckPfiHmM.val2xml('ID_ETSA', :old.ETSA_ID_ETSA);
			v_xml := v_xml || pckPfiHmM.val2xml('CODE', :old.ETSA_CODE);

		else
			v_xml := v_xml || pckPfiHmM.val2xml('ID_ETSA', :new.ETSA_ID_ETSA);
			v_xml := v_xml || pckPfiHmM.val2xml('CODE', :new.ETSA_CODE);
